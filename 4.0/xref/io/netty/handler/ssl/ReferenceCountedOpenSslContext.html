65  /**
66   * An implementation of {@link SslContext} which works with libraries that support the
67   * <a href="https://www.openssl.org/">OpenSsl</a> C library API.
68   * <p>Instances of this class must be {@link #release() released} or else native memory will leak!
69   *
70   * <p>Instances of this class <strong>must not</strong> be released before any {@link ReferenceCountedOpenSslEngine}
71   * which depends upon the instance of this class is released. Otherwise if any method of
72   * {@link ReferenceCountedOpenSslEngine} is called which uses this class's JNI resources the JVM may crash.
73   */
74  public abstract class ReferenceCountedOpenSslContext extends SslContext implements ReferenceCounted {
75      private static final InternalLogger logger =
76              InternalLoggerFactory.getInstance(ReferenceCountedOpenSslContext.class);
77      /**
78       * To make it easier for users to replace JDK implementation with OpenSsl version we also use
79       * {@code jdk.tls.rejectClientInitiatedRenegotiation} to allow disabling client initiated renegotiation.
80       * Java8+ uses this system property as well.
81       * <p>
82       * See also <a href="http://blog.ivanristic.com/2014/03/ssl-tls-improvements-in-java-8.html">
83       * Significant SSL/TLS improvements in Java 8</a>
84       */
85      private static final boolean JDK_REJECT_CLIENT_INITIATED_RENEGOTIATION =
86              AccessController.doPrivileged(new PrivilegedAction<Boolean>() {
87                  @Override
88                  public Boolean run() {
89                      return SystemPropertyUtil.getBoolean("jdk.tls.rejectClientInitiatedRenegotiation", false);
90                  }
91              });

219     ReferenceCountedOpenSslContext(Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
220                                    OpenSslApplicationProtocolNegotiator apn, long sessionCacheSize,
221                                    long sessionTimeout, int mode, Certificate[] keyCertChain,
222                                    ClientAuth clientAuth, String[] protocols, boolean startTls, boolean enableOcsp,
223                                    boolean leakDetection) throws SSLException {
224         super(startTls);
225 
226         OpenSsl.ensureAvailability();
227 
228         if (enableOcsp && !OpenSsl.isOcspSupported()) {
229             throw new IllegalStateException("OCSP is not supported.");
230         }
231 
232         if (mode != SSL.SSL_MODE_SERVER && mode != SSL.SSL_MODE_CLIENT) {
233             throw new IllegalArgumentException("mode most be either SSL.SSL_MODE_SERVER or SSL.SSL_MODE_CLIENT");
234         }
235         leak = leakDetection ? leakDetector.track(this) : null;
236         this.mode = mode;
237         this.clientAuth = isServer() ? checkNotNull(clientAuth, "clientAuth") : ClientAuth.NONE;
238         this.protocols = protocols;
239         this.enableOcsp = enableOcsp;
240 
241         if (mode == SSL.SSL_MODE_SERVER) {
242             rejectRemoteInitiatedRenegotiation =
243                     JDK_REJECT_CLIENT_INITIATED_RENEGOTIATION;
244         }
245         this.keyCertChain = keyCertChain == null ? null : keyCertChain.clone();
246         final List<String> convertedCiphers;
247         if (ciphers == null) {
248             convertedCiphers = null;
249         } else {
250             convertedCiphers = new ArrayList<String>();
251             for (String c : ciphers) {
252                 if (c == null) {
253                     break;
254                 }
255 
256                 String converted = CipherSuiteConverter.toOpenSsl(c);
257                 if (converted != null) {
258                     c = converted;
259                 }
260                 convertedCiphers.add(c);
261             }
262         }
263 
264         unmodifiableCiphers = Arrays.asList(checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
265                 convertedCiphers, DEFAULT_CIPHERS, OpenSsl.availableOpenSslCipherSuites()));
266 
267         this.apn = checkNotNull(apn, "apn");
268 
269         // Create a new SSL_CTX and configure it.
270         boolean success = false;
271         try {
272             try {
273                 ctx = SSLContext.make(SSL.SSL_PROTOCOL_ALL, mode);
274             } catch (Exception e) {
275                 throw new SSLException("failed to create an SSL_CTX", e);
276             }
277 
278             SSLContext.setOptions(ctx, SSLContext.getOptions(ctx) |
279                     SSL.SSL_OP_NO_SSLv2 |
280                     SSL.SSL_OP_NO_SSLv3 |
281                     SSL.SSL_OP_CIPHER_SERVER_PREFERENCE |
282 
283                     // We do not support compression at the moment so we should explicitly disable it.
284                     SSL.SSL_OP_NO_COMPRESSION |
285 
286                     // Disable ticket support by default to be more inline with SSLEngineImpl of the JDK.
287                     // This also let SSLSession.getId() work the same way for the JDK implementation and the
288                     // OpenSSLEngine. If tickets are supported SSLSession.getId() will only return an ID on the
289                     // server-side if it could make use of tickets.
290                     SSL.SSL_OP_NO_TICKET);
291 
292             // We need to enable SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER as the memory address may change between
293             // calling OpenSSLEngine.wrap(...).
294             // See https://github.com/netty/netty-tcnative/issues/100
295             SSLContext.setMode(ctx, SSLContext.getMode(ctx) | SSL.SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
296 
297             if (DH_KEY_LENGTH != null) {
298                 SSLContext.setTmpDHLength(ctx, DH_KEY_LENGTH);
299             }
300 
301                 /* List the ciphers that are permitted to negotiate. */
302             try {
303                 SSLContext.setCipherSuite(ctx, CipherSuiteConverter.toOpenSsl(unmodifiableCiphers));
304             } catch (SSLException e) {
305                 throw e;
306             } catch (Exception e) {
307                 throw new SSLException("failed to set cipher suite: " + unmodifiableCiphers, e);
308             }
309 
310             List<String> nextProtoList = apn.protocols();
311                 /* Set next protocols for next protocol negotiation extension, if specified */
312             if (!nextProtoList.isEmpty()) {
313                 String[] appProtocols = nextProtoList.toArray(new String[nextProtoList.size()]);
314                 int selectorBehavior = opensslSelectorFailureBehavior(apn.selectorFailureBehavior());
315 
316                 switch (apn.protocol()) {
317                     case NPN:
318                         SSLContext.setNpnProtos(ctx, appProtocols, selectorBehavior);
319                         break;
320                     case ALPN:
321                         SSLContext.setAlpnProtos(ctx, appProtocols, selectorBehavior);
322                         break;
323                     case NPN_AND_ALPN:
324                         SSLContext.setNpnProtos(ctx, appProtocols, selectorBehavior);
325                         SSLContext.setAlpnProtos(ctx, appProtocols, selectorBehavior);
326                         break;
327                     default:
328                         throw new Error();
329                 }
330             }
331 
332             /* Set session cache size, if specified */
333             if (sessionCacheSize > 0) {
334                 this.sessionCacheSize = sessionCacheSize;
335                 SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
336             } else {
337                 // Get the default session cache size using SSLContext.setSessionCacheSize()
338                 this.sessionCacheSize = sessionCacheSize = SSLContext.setSessionCacheSize(ctx, 20480);
339                 // Revert the session cache size to the default value.
340                 SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
341             }
342 
343             /* Set session timeout, if specified */
344             if (sessionTimeout > 0) {
345                 this.sessionTimeout = sessionTimeout;
346                 SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
347             } else {
348                 // Get the default session timeout using SSLContext.setSessionCacheTimeout()
349                 this.sessionTimeout = sessionTimeout = SSLContext.setSessionCacheTimeout(ctx, 300);
350                 // Revert the session timeout to the default value.
351                 SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
352             }
353 
354             if (enableOcsp) {
355                 SSLContext.enableOcsp(ctx, isClient());
356             }
357             success = true;
358         } finally {
359             if (!success) {
360                 release();
361             }
362         }
363     }

369             case CHOOSE_MY_LAST_PROTOCOL:
370                 return SSL.SSL_SELECTOR_FAILURE_CHOOSE_MY_LAST_PROTOCOL;
371             default:
372                 throw new Error();
373         }

510     // IMPORTANT: This method must only be called from either the constructor or the finalizer as a user MUST never
511     //            get access to an OpenSslSessionContext after this method was called to prevent the user from
512     //            producing a segfault.
513     private void destroy() {
514         Lock writerLock = ctxLock.writeLock();
515         writerLock.lock();
516         try {
517             if (ctx != 0) {
518                 if (enableOcsp) {
519                     SSLContext.disableOcsp(ctx);
520                 }
521 
522                 SSLContext.free(ctx);
523                 ctx = 0;
524             }
525         } finally {
526             writerLock.unlock();
527         }
528     }

556     /**
557      * Translate a {@link ApplicationProtocolConfig} object to a
558      * {@link OpenSslApplicationProtocolNegotiator} object.
559      *
560      * @param config The configuration which defines the translation
561      * @return The results of the translation
562      */
563     static OpenSslApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config) {
564         if (config == null) {
565             return NONE_PROTOCOL_NEGOTIATOR;
566         }
567 
568         switch (config.protocol()) {
569             case NONE:
570                 return NONE_PROTOCOL_NEGOTIATOR;
571             case ALPN:
572             case NPN:
573             case NPN_AND_ALPN:
574                 switch (config.selectedListenerFailureBehavior()) {
575                     case CHOOSE_MY_LAST_PROTOCOL:
576                     case ACCEPT:
577                         switch (config.selectorFailureBehavior()) {
578                             case CHOOSE_MY_LAST_PROTOCOL:
579                             case NO_ADVERTISE:
580                                 return new OpenSslDefaultApplicationProtocolNegotiator(
581                                         config);
582                             default:
583                                 throw new UnsupportedOperationException(
584                                         new StringBuilder("OpenSSL provider does not support ")
585                                                 .append(config.selectorFailureBehavior())
586                                                 .append(" behavior").toString());
587                         }
588                     default:
589                         throw new UnsupportedOperationException(
590                                 new StringBuilder("OpenSSL provider does not support ")
591                                         .append(config.selectedListenerFailureBehavior())
592                                         .append(" behavior").toString());
593                 }
594             default:
595                 throw new Error();
596         }
597     }

634     abstract static class AbstractCertificateVerifier extends CertificateVerifier {
635         private final OpenSslEngineMap engineMap;
636 
637         AbstractCertificateVerifier(OpenSslEngineMap engineMap) {
638             this.engineMap = engineMap;
639         }
640 
641         @Override
642         public final int verify(long ssl, byte[][] chain, String auth) {
643             X509Certificate[] peerCerts = certificates(chain);
644             final ReferenceCountedOpenSslEngine engine = engineMap.get(ssl);
645             try {
646                 verify(engine, peerCerts, auth);
647                 return CertificateVerifier.X509_V_OK;
648             } catch (Throwable cause) {
649                 logger.debug("verification of certificate failed", cause);
650                 SSLHandshakeException e = new SSLHandshakeException("General OpenSslEngine problem");
651                 e.initCause(cause);
652                 engine.handshakeException = e;
653 
654                 // Try to extract the correct error code that should be used.
655                 if (cause instanceof OpenSslCertificateException) {
656                     // This will never return a negative error code as its validated when constructing the
657                     // OpenSslCertificateException.
658                     return ((OpenSslCertificateException) cause).errorCode();
659                 }
660                 if (cause instanceof CertificateExpiredException) {
661                     return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
662                 }
663                 if (cause instanceof CertificateNotYetValidException) {
664                     return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
665                 }
666                 if (PlatformDependent.javaVersion() >= 7) {
667                     if (cause instanceof CertificateRevokedException) {
668                         return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
669                     }
670 
671                     // The X509TrustManagerImpl uses a Validator which wraps a CertPathValidatorException into
672                     // an CertificateException. So we need to handle the wrapped CertPathValidatorException to be
673                     // able to send the correct alert.
674                     Throwable wrapped = cause.getCause();
675                     while (wrapped != null) {
676                         if (wrapped instanceof CertPathValidatorException) {
677                             CertPathValidatorException ex = (CertPathValidatorException) wrapped;
678                             CertPathValidatorException.Reason reason = ex.getReason();
679                             if (reason == CertPathValidatorException.BasicReason.EXPIRED) {
680                                 return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
681                             }
682                             if (reason == CertPathValidatorException.BasicReason.NOT_YET_VALID) {
683                                 return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
684                             }
685                             if (reason == CertPathValidatorException.BasicReason.REVOKED) {
686                                 return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
687                             }
688                         }
689                         wrapped = wrapped.getCause();
690                     }
691                 }
692 
693                 // Could not detect a specific error code to use, so fallback to a default code.
694                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
695             }
696         }

721     static void setKeyMaterial(long ctx, X509Certificate[] keyCertChain, PrivateKey key, String keyPassword)
722             throws SSLException {
723          /* Load the certificate file and private key. */
724         long keyBio = 0;
725         long keyCertChainBio = 0;
726         long keyCertChainBio2 = 0;
727         PemEncoded encoded = null;
728         try {
729             // Only encode one time
730             encoded = PemX509Certificate.toPEM(ByteBufAllocator.DEFAULT, true, keyCertChain);
731             keyCertChainBio = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
732             keyCertChainBio2 = toBIO(ByteBufAllocator.DEFAULT, encoded.retain());
733 
734             if (key != null) {
735                 keyBio = toBIO(key);
736             }
737 
738             SSLContext.setCertificateBio(
739                     ctx, keyCertChainBio, keyBio,
740                     keyPassword == null ? StringUtil.EMPTY_STRING : keyPassword);
741             // We may have more then one cert in the chain so add all of them now.
742             SSLContext.setCertificateChainBio(ctx, keyCertChainBio2, true);
743         } catch (SSLException e) {
744             throw e;
745         } catch (Exception e) {
746             throw new SSLException("failed to set certificate and key", e);
747         } finally {
748             freeBio(keyBio);
749             freeBio(keyCertChainBio);
750             freeBio(keyCertChainBio2);
751             if (encoded != null) {
752                 encoded.release();
753             }
754         }
755     }

803     static long toBIO(ByteBufAllocator allocator, PemEncoded pem) throws Exception {
804         try {
805             // We can turn direct buffers straight into BIOs. No need to
806             // make a yet another copy.
807             ByteBuf content = pem.content();
808 
809             if (content.isDirect()) {
810                 return newBIO(content.slice().retain());
811             }
812 
813             ByteBuf buffer = allocator.directBuffer(content.readableBytes());
814             try {
815                 buffer.writeBytes(content, content.readerIndex(), content.readableBytes());
816                 return newBIO(buffer.slice().retain());
817             } finally {
818                 try {
819                     // If the contents of the ByteBuf is sensitive (e.g. a PrivateKey) we
820                     // need to zero out the bytes of the copy before we're releasing it.
821                     if (pem.isSensitive()) {
822                         SslUtils.zeroout(buffer);
823                     }
824                 } finally {
825                     buffer.release();
826                 }
827             }
828         } finally {
829             pem.release();
830         }
831     }

833     private static long newBIO(ByteBuf buffer) throws Exception {
834         try {
835             long bio = SSL.newMemBIO();
836             int readable = buffer.readableBytes();
837             if (SSL.bioWrite(bio, OpenSsl.memoryAddress(buffer) + buffer.readerIndex(), readable) != readable) {
838                 SSL.freeBIO(bio);
839                 throw new IllegalStateException("Could not write data to memory BIO");
840             }
841             return bio;
842         } finally {
843             buffer.release();
844         }
