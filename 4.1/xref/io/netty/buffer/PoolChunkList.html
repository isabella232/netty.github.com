30  final class PoolChunkList<T> implements PoolChunkListMetric {
31      private static final Iterator<PoolChunkMetric> EMPTY_METRICS = Collections.<PoolChunkMetric>emptyList().iterator();
32      private final PoolArena<T> arena;
33      private final PoolChunkList<T> nextList;
34      private final int minUsage;
35      private final int maxUsage;
36      private final int maxCapacity;
37      private PoolChunk<T> head;
38  
39      // This is only update once when create the linked like list of PoolChunkList in PoolArena constructor.
40      private PoolChunkList<T> prevList;
41  
42      // TODO: Test if adding padding helps under contention
43      //private long pad0, pad1, pad2, pad3, pad4, pad5, pad6, pad7;

45      PoolChunkList(PoolArena<T> arena, PoolChunkList<T> nextList, int minUsage, int maxUsage, int chunkSize) {
46          assert minUsage <= maxUsage;
47          this.arena = arena;
48          this.nextList = nextList;
49          this.minUsage = minUsage;
50          this.maxUsage = maxUsage;
51          maxCapacity = calculateMaxCapacity(minUsage, chunkSize);
52      }
53  
54      /**
55       * Calculates the maximum capacity of a buffer that will ever be possible to allocate out of the {@link PoolChunk}s
56       * that belong to the {@link PoolChunkList} with the given {@code minUsage} and {@code maxUsage} settings.
57       */
58      private static int calculateMaxCapacity(int minUsage, int chunkSize) {
59          minUsage = minUsage0(minUsage);
60  
61          if (minUsage == 100) {
62              // If the minUsage is 100 we can not allocate anything out of this list.
63              return 0;
64          }
65  
66          // Calculate the maximum amount of bytes that can be allocated from a PoolChunk in this PoolChunkList.
67          //
68          // As an example:
69          // - If a PoolChunkList has minUsage == 25 we are allowed to allocate at most 75% of the chunkSize because
70          //   this is the maximum amount available in any PoolChunk in this PoolChunkList.
71          return  (int) (chunkSize * (100L - minUsage) / 100L);
72      }
73  
74      void prevList(PoolChunkList<T> prevList) {
75          assert this.prevList == null;
76          this.prevList = prevList;
77      }
78  
79      boolean allocate(PooledByteBuf<T> buf, int reqCapacity, int normCapacity) {
80          if (normCapacity > maxCapacity) {
81              // Either this PoolChunkList is empty or the requested capacity is larger then the capacity which can
82              // be handled by the PoolChunks that are contained in this PoolChunkList.
83              return false;
84          }
85  
86          for (PoolChunk<T> cur = head; cur != null; cur = cur.next) {
87              if (cur.allocate(buf, reqCapacity, normCapacity)) {
88                  if (cur.usage() >= maxUsage) {
89                      remove(cur);
90                      nextList.add(cur);
91                  }
92                  return true;
93              }
94          }
95          return false;
96      }
97  
98      boolean free(PoolChunk<T> chunk, long handle, ByteBuffer nioBuffer) {
99          chunk.free(handle, nioBuffer);
100         if (chunk.usage() < minUsage) {
101             remove(chunk);
102             // Move the PoolChunk down the PoolChunkList linked-list.
103             return move0(chunk);
104         }
105         return true;
106     }
107 
108     private boolean move(PoolChunk<T> chunk) {
109         assert chunk.usage() < maxUsage;
110 
111         if (chunk.usage() < minUsage) {
112             // Move the PoolChunk down the PoolChunkList linked-list.
113             return move0(chunk);
114         }
115 
116         // PoolChunk fits into this PoolChunkList, adding it here.
117         add0(chunk);
118         return true;
119     }
120 
121     /**
122      * Moves the {@link PoolChunk} down the {@link PoolChunkList} linked-list so it will end up in the right
123      * {@link PoolChunkList} that has the correct minUsage / maxUsage in respect to {@link PoolChunk#usage()}.
124      */
125     private boolean move0(PoolChunk<T> chunk) {
126         if (prevList == null) {
127             // There is no previous PoolChunkList so return false which result in having the PoolChunk destroyed and
128             // all memory associated with the PoolChunk will be released.
129             assert chunk.usage() == 0;
130             return false;
131         }
132         return prevList.move(chunk);
133     }
134 
135     void add(PoolChunk<T> chunk) {
136         if (chunk.usage() >= maxUsage) {
137             nextList.add(chunk);
138             return;
139         }
140         add0(chunk);
141     }
142 
143     /**
144      * Adds the {@link PoolChunk} to this {@link PoolChunkList}.
145      */
146     void add0(PoolChunk<T> chunk) {
147         chunk.parent = this;
148         if (head == null) {
149             head = chunk;
150             chunk.prev = null;
151             chunk.next = null;
152         } else {
153             chunk.prev = null;
154             chunk.next = head;
155             head.prev = chunk;
156             head = chunk;
157         }
158     }
159 
160     private void remove(PoolChunk<T> cur) {
161         if (cur == head) {
162             head = cur.next;
163             if (head != null) {
164                 head.prev = null;
165             }
166         } else {
167             PoolChunk<T> next = cur.next;
168             cur.prev.next = next;
169             if (next != null) {
170                 next.prev = cur.prev;
171             }
172         }
173     }
174 
175     @Override
176     public int minUsage() {
177         return minUsage0(minUsage);
178     }
179 
180     @Override
181     public int maxUsage() {
182         return min(maxUsage, 100);
183     }
184 
185     private static int minUsage0(int value) {
186         return max(1, value);
187     }
188 
189     @Override
190     public Iterator<PoolChunkMetric> iterator() {
191         synchronized (arena) {
192             if (head == null) {
193                 return EMPTY_METRICS;
194             }
195             List<PoolChunkMetric> metrics = new ArrayList<PoolChunkMetric>();
196             for (PoolChunk<T> cur = head;;) {
197                 metrics.add(cur);
198                 cur = cur.next;
199                 if (cur == null) {
200                     break;
201                 }
202             }
203             return metrics.iterator();
204         }
205     }
206 
207     @Override
208     public String toString() {
209         StringBuilder buf = new StringBuilder();
210         synchronized (arena) {
211             if (head == null) {
212                 return "none";
213             }
214 
215             for (PoolChunk<T> cur = head;;) {
216                 buf.append(cur);
217                 cur = cur.next;
218                 if (cur == null) {
219                     break;
220                 }
221                 buf.append(StringUtil.NEWLINE);
222             }
223         }
224         return buf.toString();
225     }
226 
227     void destroy(PoolArena<T> arena) {
228         PoolChunk<T> chunk = head;
229         while (chunk != null) {
230             arena.destroyChunk(chunk);
231             chunk = chunk.next;
232         }
233         head = null;
234     }
