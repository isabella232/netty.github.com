19  /**
20   * A decorator around another {@link Http2ConnectionEncoder} instance.
21   */
22  public class DecoratingHttp2ConnectionEncoder extends DecoratingHttp2FrameWriter implements Http2ConnectionEncoder {
23      private final Http2ConnectionEncoder delegate;
24  
25      public DecoratingHttp2ConnectionEncoder(Http2ConnectionEncoder delegate) {
26          super(delegate);
27          this.delegate = checkNotNull(delegate, "delegate");
28      }
29  
30      @Override
31      public void lifecycleManager(Http2LifecycleManager lifecycleManager) {
32          delegate.lifecycleManager(lifecycleManager);
33      }
34  
35      @Override
36      public Http2Connection connection() {
37          return delegate.connection();
38      }
39  
40      @Override
41      public Http2RemoteFlowController flowController() {
42          return delegate.flowController();
43      }
44  
45      @Override
46      public Http2FrameWriter frameWriter() {
47          return delegate.frameWriter();
48      }
49  
50      @Override
51      public Http2Settings pollSentSettings() {
52          return delegate.pollSentSettings();
53      }
54  
55      @Override
56      public void remoteSettings(Http2Settings settings) throws Http2Exception {
57          delegate.remoteSettings(settings);
58      }
59  }
