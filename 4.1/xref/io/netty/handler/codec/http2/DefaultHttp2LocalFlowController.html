35  /**
36   * Basic implementation of {@link Http2LocalFlowController}.
37   * <p>
38   * This class is <strong>NOT</strong> thread safe. The assumption is all methods must be invoked from a single thread.
39   * Typically this thread is the event loop thread for the {@link ChannelHandlerContext} managed by this class.
40   */
41  public class DefaultHttp2LocalFlowController implements Http2LocalFlowController {
42      /**
43       * The default ratio of window size to initial window size below which a {@code WINDOW_UPDATE}
44       * is sent to expand the window.
45       */
46      public static final float DEFAULT_WINDOW_UPDATE_RATIO = 0.5f;

59      /**
60       * Constructs a controller with the given settings.
61       *
62       * @param connection the connection state.
63       * @param windowUpdateRatio the window percentage below which to send a {@code WINDOW_UPDATE}.
64       * @param autoRefillConnectionWindow if {@code true}, effectively disables the connection window
65       * in the flow control algorithm as they will always refill automatically without requiring the
66       * application to consume the bytes. When enabled, the maximum bytes you must be prepared to
67       * queue is proportional to {@code maximum number of concurrent streams * the initial window
68       * size per stream}
69       * (<a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_MAX_CONCURRENT_STREAMS</a>
70       * <a href="https://tools.ietf.org/html/rfc7540#section-6.5.2">SETTINGS_INITIAL_WINDOW_SIZE</a>).
71       */
72      public DefaultHttp2LocalFlowController(Http2Connection connection,
73                                             float windowUpdateRatio,
74                                             boolean autoRefillConnectionWindow) {
75          this.connection = checkNotNull(connection, "connection");
76          windowUpdateRatio(windowUpdateRatio);
77  
78          // Add a flow state for the connection.
79          stateKey = connection.newKey();
80          FlowState connectionState = autoRefillConnectionWindow ?
81                  new AutoRefillState(connection.connectionStream(), initialWindowSize) :
82                  new DefaultState(connection.connectionStream(), initialWindowSize);
83          connection.connectionStream().setProperty(stateKey, connectionState);
84  
85          // Register for notification of new streams.
86          connection.addListener(new Http2ConnectionAdapter() {
87              @Override
88              public void onStreamAdded(Http2Stream stream) {
89                  // Unconditionally used the reduced flow control state because it requires no object allocation
90                  // and the DefaultFlowState will be allocated in onStreamActive.
91                  stream.setProperty(stateKey, REDUCED_FLOW_STATE);
92              }
93  
94              @Override
95              public void onStreamActive(Http2Stream stream) {
96                  // Need to be sure the stream's initial window is adjusted for SETTINGS
97                  // frames which may have been exchanged while it was in IDLE
98                  stream.setProperty(stateKey, new DefaultState(stream, initialWindowSize));
99              }
100 
101             @Override
102             public void onStreamClosed(Http2Stream stream) {
103                 try {
104                     // When a stream is closed, consume any remaining bytes so that they
105                     // are restored to the connection window.
106                     FlowState state = state(stream);
107                     int unconsumedBytes = state.unconsumedBytes();
108                     if (ctx != null && unconsumedBytes > 0) {
109                         connectionState().consumeBytes(unconsumedBytes);
110                         state.consumeBytes(unconsumedBytes);
111                     }
112                 } catch (Http2Exception e) {
113                     PlatformDependent.throwException(e);
114                 } finally {
115                     // Unconditionally reduce the amount of memory required for flow control because there is no
116                     // object allocation costs associated with doing so and the stream will not have any more
117                     // local flow control state to keep track of anymore.
118                     stream.setProperty(stateKey, REDUCED_FLOW_STATE);
119                 }
120             }
121         });
122     }

136     public void initialWindowSize(int newWindowSize) throws Http2Exception {
137         assert ctx == null || ctx.executor().inEventLoop();
138         int delta = newWindowSize - initialWindowSize;
139         initialWindowSize = newWindowSize;

161     @Override
162     public void incrementWindowSize(Http2Stream stream, int delta) throws Http2Exception {
163         assert ctx != null && ctx.executor().inEventLoop();
164         FlowState state = state(stream);
165         // Just add the delta to the stream-specific initial window size so that the next time the window
166         // expands it will grow to the new initial size.
167         state.incrementInitialStreamWindow(delta);
168         state.writeWindowUpdateIfNeeded();

171     @Override
172     public boolean consumeBytes(Http2Stream stream, int numBytes) throws Http2Exception {
173         assert ctx != null && ctx.executor().inEventLoop();
174         if (numBytes < 0) {
175             throw new IllegalArgumentException("numBytes must not be negative");
176         }
177         if (numBytes == 0) {
178             return false;
179         }
180 
181         // Streams automatically consume all remaining bytes when they are closed, so just ignore
182         // if already closed.
183         if (stream != null && !isClosed(stream)) {
184             if (stream.id() == CONNECTION_STREAM_ID) {
185                 throw new UnsupportedOperationException("Returning bytes for the connection window is not supported");
186             }
187 
188             boolean windowUpdateSent = connectionState().consumeBytes(numBytes);
189             windowUpdateSent |= state(stream).consumeBytes(numBytes);
190             return windowUpdateSent;
191         }
192         return false;
193     }

197         return state(stream).unconsumedBytes();

228     /**
229      * The window update ratio is used to determine when a window update must be sent. If the ratio
230      * of bytes processed since the last update has meet or exceeded this ratio then a window update will
231      * be sent. This window update ratio will only be applied to {@code streamId}.
232      * <p>
233      * Note it is the responsibly of the caller to ensure that the the
234      * initial {@code SETTINGS} frame is sent before this is called. It would
235      * be considered a {@link Http2Error#PROTOCOL_ERROR} if a {@code WINDOW_UPDATE}
236      * was generated by this method before the initial {@code SETTINGS} frame is sent.
237      * @param stream the stream for which {@code ratio} applies to.
238      * @param ratio the ratio to use when checking if a {@code WINDOW_UPDATE} is determined necessary.
239      * @throws Http2Exception If a protocol-error occurs while generating {@code WINDOW_UPDATE} frames
240      */
241     public void windowUpdateRatio(Http2Stream stream, float ratio) throws Http2Exception {
242         assert ctx != null && ctx.executor().inEventLoop();
243         checkValidRatio(ratio);
244         FlowState state = state(stream);
245         state.windowUpdateRatio(ratio);
246         state.writeWindowUpdateIfNeeded();
247     }

251      * of bytes processed since the last update has meet or exceeded this ratio then a window update will
252      * be sent. This window update ratio will only be applied to {@code streamId}.
253      * @throws Http2Exception If no stream corresponding to {@code stream} could be found.
254      */

256         return state(stream).windowUpdateRatio();

259     @Override
260     public void receiveFlowControlledFrame(Http2Stream stream, ByteBuf data, int padding,
261             boolean endOfStream) throws Http2Exception {
262         assert ctx != null && ctx.executor().inEventLoop();
263         int dataLength = data.readableBytes() + padding;
264 
265         // Apply the connection-level flow control
266         FlowState connectionState = connectionState();
267         connectionState.receiveFlowControlledFrame(dataLength);
268 
269         if (stream != null && !isClosed(stream)) {
270             // Apply the stream-level flow control
271             FlowState state = state(stream);
272             state.endOfStream(endOfStream);
273             state.receiveFlowControlledFrame(dataLength);
274         } else if (dataLength > 0) {
275             // Immediately consume the bytes for the connection window.
276             connectionState.consumeBytes(dataLength);
277         }
278     }

316     /**
317      * Flow control window state for an individual stream.
318      */
319     private class DefaultState implements FlowState {
320         private final Http2Stream stream;
321 
322         /**
323          * The actual flow control window that is decremented as soon as {@code DATA} arrives.
324          */
325         private int window;
326 
327         /**
328          * A view of {@link #window} that is used to determine when to send {@code WINDOW_UPDATE}
329          * frames. Decrementing this window for received {@code DATA} frames is delayed until the
330          * application has indicated that the data has been fully processed. This prevents sending
331          * a {@code WINDOW_UPDATE} until the number of processed bytes drops below the threshold.
332          */
333         private int processedWindow;
334 
335         /**
336          * This is what is used to determine how many bytes need to be returned relative to {@link #processedWindow}.
337          * Each stream has their own initial window size.
338          */
339         private int initialStreamWindowSize;
340 
341         /**
342          * This is used to determine when {@link #processedWindow} is sufficiently far away from
343          * {@link #initialStreamWindowSize} such that a {@code WINDOW_UPDATE} should be sent.
344          * Each stream has their own window update ratio.
345          */
346         private float streamWindowUpdateRatio;
347 
348         private int lowerBound;
349         private boolean endOfStream;
350 
351         public DefaultState(Http2Stream stream, int initialWindowSize) {
352             this.stream = stream;
353             window(initialWindowSize);
354             streamWindowUpdateRatio = windowUpdateRatio;
355         }
356 
357         @Override
358         public void window(int initialWindowSize) {
359             assert ctx == null || ctx.executor().inEventLoop();
360             window = processedWindow = initialStreamWindowSize = initialWindowSize;
361         }
362 
363         @Override
364         public int windowSize() {
365             return window;
366         }
367 
368         @Override
369         public int initialWindowSize() {
370             return initialStreamWindowSize;
371         }
372 
373         @Override
374         public void endOfStream(boolean endOfStream) {
375             this.endOfStream = endOfStream;
376         }
377 
378         @Override
379         public float windowUpdateRatio() {
380             return streamWindowUpdateRatio;
381         }
382 
383         @Override
384         public void windowUpdateRatio(float ratio) {
385             assert ctx == null || ctx.executor().inEventLoop();
386             streamWindowUpdateRatio = ratio;
387         }
388 
389         @Override
390         public void incrementInitialStreamWindow(int delta) {
391             // Clip the delta so that the resulting initialStreamWindowSize falls within the allowed range.
392             int newValue = (int) min(MAX_INITIAL_WINDOW_SIZE,
393                     max(MIN_INITIAL_WINDOW_SIZE, initialStreamWindowSize + (long) delta));
394             delta = newValue - initialStreamWindowSize;
395 
396             initialStreamWindowSize += delta;
397         }
398 
399         @Override
400         public void incrementFlowControlWindows(int delta) throws Http2Exception {
401             if (delta > 0 && window > MAX_INITIAL_WINDOW_SIZE - delta) {
402                 throw streamError(stream.id(), FLOW_CONTROL_ERROR,
403                         "Flow control window overflowed for stream: %d", stream.id());
404             }
405 
406             window += delta;
407             processedWindow += delta;
408             lowerBound = delta < 0 ? delta : 0;
409         }
410 
411         @Override
412         public void receiveFlowControlledFrame(int dataLength) throws Http2Exception {
413             assert dataLength >= 0;
414 
415             // Apply the delta. Even if we throw an exception we want to have taken this delta into account.
416             window -= dataLength;
417 
418             // Window size can become negative if we sent a SETTINGS frame that reduces the
419             // size of the transfer window after the peer has written data frames.
420             // The value is bounded by the length that SETTINGS frame decrease the window.
421             // This difference is stored for the connection when writing the SETTINGS frame
422             // and is cleared once we send a WINDOW_UPDATE frame.
423             if (window < lowerBound) {
424                 throw streamError(stream.id(), FLOW_CONTROL_ERROR,
425                         "Flow control window exceeded for stream: %d", stream.id());
426             }
427         }
428 
429         private void returnProcessedBytes(int delta) throws Http2Exception {
430             if (processedWindow - delta < window) {
431                 throw streamError(stream.id(), INTERNAL_ERROR,
432                         "Attempting to return too many bytes for stream %d", stream.id());
433             }
434             processedWindow -= delta;
435         }
436 
437         @Override
438         public boolean consumeBytes(int numBytes) throws Http2Exception {
439             // Return the bytes processed and update the window.
440             returnProcessedBytes(numBytes);
441             return writeWindowUpdateIfNeeded();
442         }
443 
444         @Override
445         public int unconsumedBytes() {
446             return processedWindow - window;
447         }
448 
449         @Override
450         public boolean writeWindowUpdateIfNeeded() throws Http2Exception {
451             if (endOfStream || initialStreamWindowSize <= 0) {
452                 return false;
453             }
454 
455             int threshold = (int) (initialStreamWindowSize * streamWindowUpdateRatio);
456             if (processedWindow <= threshold) {
457                 writeWindowUpdate();
458                 return true;
459             }
460             return false;
461         }
462 
463         /**
464          * Called to perform a window update for this stream (or connection). Updates the window size back
465          * to the size of the initial window and sends a window update frame to the remote endpoint.
466          */
467         private void writeWindowUpdate() throws Http2Exception {
468             // Expand the window for this stream back to the size of the initial window.
469             int deltaWindowSize = initialStreamWindowSize - processedWindow;
470             try {
471                 incrementFlowControlWindows(deltaWindowSize);
472             } catch (Throwable t) {
473                 throw connectionError(INTERNAL_ERROR, t,
474                         "Attempting to return too many bytes for stream %d", stream.id());
475             }
476 
477             // Send a window update for the stream/connection.
478             frameWriter.writeWindowUpdate(ctx, stream.id(), deltaWindowSize, ctx.newPromise());
479         }
480     }

554     private interface FlowState {
555 
556         int windowSize();
557 
558         int initialWindowSize();
559 
560         void window(int initialWindowSize);
561 
562         /**

564          * @param delta The amount to increase the initial window size by.

567 
568         /**

572          */
573         boolean writeWindowUpdateIfNeeded() throws Http2Exception;
574 
575         /**

580          * @return true if {@code WINDOW_UPDATE} was written, false otherwise.
581          * @throws Http2Exception
582          */
583         boolean consumeBytes(int numBytes) throws Http2Exception;
584 
585         int unconsumedBytes();
586 
587         float windowUpdateRatio();
588 
589         void windowUpdateRatio(float ratio);
590 
591         /**
592          * A flow control event has occurred and we should decrement the amount of available bytes for this stream.
593          * @param dataLength The amount of data to for which this stream is no longer eligible to use for flow control.
594          * @throws Http2Exception If too much data is used relative to how much is available.
595          */
596         void receiveFlowControlledFrame(int dataLength) throws Http2Exception;
597 
598         /**
599          * Increment the windows which are used to determine many bytes have been processed.
600          * @param delta The amount to increment the window by.

602          */
603         void incrementFlowControlWindows(int delta) throws Http2Exception;
604

608     /**
609      * Provides a means to iterate over all active streams and increment the flow control windows.
610      */
611     private final class WindowUpdateVisitor implements Http2StreamVisitor {
612         private CompositeStreamException compositeException;
613         private final int delta;
614 
615         public WindowUpdateVisitor(int delta) {
616             this.delta = delta;
617         }
618 
619         @Override
620         public boolean visit(Http2Stream stream) throws Http2Exception {
621             try {
622                 // Increment flow control window first so state will be consistent if overflow is detected.
623                 FlowState state = state(stream);
624                 state.incrementFlowControlWindows(delta);
625                 state.incrementInitialStreamWindow(delta);
626             } catch (StreamException e) {
627                 if (compositeException == null) {
628                     compositeException = new CompositeStreamException(e.error(), 4);
629                 }
630                 compositeException.add(e);
631             }
632             return true;
633         }
634 
635         public void throwIfError() throws CompositeStreamException {
636             if (compositeException != null) {
637                 throw compositeException;
638             }
639         }
