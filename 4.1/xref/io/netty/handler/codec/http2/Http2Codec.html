16  package io.netty.handler.codec.http2;
17  
18  import io.netty.channel.ChannelDuplexHandler;
19  import io.netty.channel.ChannelHandler;
20  import io.netty.channel.ChannelHandlerContext;
21  import io.netty.util.internal.UnstableApi;
22  
23  import static io.netty.handler.logging.LogLevel.INFO;
24  
25  /**
26   * An HTTP/2 channel handler that adds a {@link Http2FrameCodec} and {@link Http2MultiplexCodec} to the pipeline before
27   * removing itself.
28   */
29  @UnstableApi
30  public final class Http2Codec extends ChannelDuplexHandler {
31      private static final Http2FrameLogger HTTP2_FRAME_LOGGER = new Http2FrameLogger(INFO, Http2Codec.class);
32  
33      private final Http2FrameCodec frameCodec;
34      private final Http2MultiplexCodec multiplexCodec;
35  
36      /**
37       * Construct a new handler whose child channels run in the same event loop as this handler.
38       *
39       * @param server {@code true} this is a server
40       * @param streamHandler the handler added to channels for remotely-created streams. It must be
41       *     {@link ChannelHandler.Sharable}. {@code null} if the event loop from the parent channel should be used.
42       */
43      public Http2Codec(boolean server, ChannelHandler streamHandler) {
44          this(server, new Http2StreamChannelBootstrap().handler(streamHandler), HTTP2_FRAME_LOGGER);
45      }
46  
47      /**
48       * Construct a new handler whose child channels run in the same event loop as this handler.
49       *
50       * @param server {@code true} this is a server
51       * @param streamHandler the handler added to channels for remotely-created streams. It must be
52       *     {@link ChannelHandler.Sharable}. {@code null} if the event loop from the parent channel should be used.
53       * @param initialSettings non default initial settings to send to peer
54       */
55      public Http2Codec(boolean server, ChannelHandler streamHandler, Http2Settings initialSettings) {
56          this(server, new Http2StreamChannelBootstrap().handler(streamHandler), HTTP2_FRAME_LOGGER,
57              initialSettings);
58      }
59  
60      /**
61       * Construct a new handler whose child channels run in a different event loop.
62       *
63       * @param server {@code true} this is a server
64       * @param bootstrap bootstrap used to instantiate child channels for remotely-created streams.
65       */
66      public Http2Codec(boolean server, Http2StreamChannelBootstrap bootstrap, Http2FrameLogger frameLogger) {
67          this(server, bootstrap, new DefaultHttp2FrameWriter(), frameLogger, new Http2Settings());
68      }
69  
70      /**
71       * Construct a new handler whose child channels run in a different event loop.
72       *
73       * @param server {@code true} this is a server
74       * @param bootstrap bootstrap used to instantiate child channels for remotely-created streams.
75       * @param initialSettings non default initial settings to send to peer
76       */
77      public Http2Codec(boolean server, Http2StreamChannelBootstrap bootstrap, Http2FrameLogger frameLogger,
78                        Http2Settings initialSettings) {
79          this(server, bootstrap, new DefaultHttp2FrameWriter(), frameLogger, initialSettings);
80      }
81  
82      // Visible for testing
83      Http2Codec(boolean server, Http2StreamChannelBootstrap bootstrap, Http2FrameWriter frameWriter,
84                 Http2FrameLogger frameLogger, Http2Settings initialSettings) {
85          frameCodec = new Http2FrameCodec(server, frameWriter, frameLogger, initialSettings);
86          multiplexCodec = new Http2MultiplexCodec(server, bootstrap);
87      }
88  
89      Http2FrameCodec frameCodec() {
90          return frameCodec;
91      }
92  
93      @Override
94      public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
95          ctx.pipeline().addBefore(ctx.executor(), ctx.name(), null, frameCodec);
96          ctx.pipeline().addBefore(ctx.executor(), ctx.name(), null, multiplexCodec);
97  
98          ctx.pipeline().remove(this);
99      }
100 }
