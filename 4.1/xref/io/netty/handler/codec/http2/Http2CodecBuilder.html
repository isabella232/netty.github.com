16  package io.netty.handler.codec.http2;
17  
18  import io.netty.channel.ChannelHandler;
19  import io.netty.handler.codec.http2.Http2HeadersEncoder.SensitivityDetector;
20  import io.netty.util.internal.UnstableApi;
21  
22  import static io.netty.handler.logging.LogLevel.INFO;
23  import static io.netty.util.internal.ObjectUtil.checkNotNull;
24  
25  /**
26   * A builder for {@link Http2Codec}.
27   */
28  @UnstableApi
29  public final class Http2CodecBuilder {
30      private static final Http2FrameLogger HTTP2_FRAME_LOGGER = new Http2FrameLogger(INFO, Http2Codec.class);
31  
32      private final Http2StreamChannelBootstrap bootstrap;
33      private final boolean server;
34      private Http2Settings initialSettings;
35      private Http2FrameLogger frameLogger;
36      private SensitivityDetector headersSensitivityDetector;
37  
38      /**
39       * Creates a new {@link Http2Codec} builder.
40       *
41       * @param server {@code true} this is a server
42       * @param streamHandler the handler added to channels for remotely-created streams. It must be
43       *     {@link ChannelHandler.Sharable}. {@code null} if the event loop from the parent channel should be used.
44       */
45      public Http2CodecBuilder(boolean server, ChannelHandler streamHandler) {
46          this(server, new Http2StreamChannelBootstrap().handler(streamHandler));
47      }
48  
49      /**
50       * Creates a new {@link Http2Codec} builder.
51       *
52       * @param server {@code true} this is a server
53       * @param bootstrap bootstrap used to instantiate child channels for remotely-created streams.
54       */
55      public Http2CodecBuilder(boolean server, Http2StreamChannelBootstrap bootstrap) {
56          this.bootstrap = checkNotNull(bootstrap, "bootstrap");
57          this.server = server;
58          this.initialSettings = Http2Settings.defaultSettings();
59          this.frameLogger = HTTP2_FRAME_LOGGER;
60          this.headersSensitivityDetector = null;
61      }
62  
63      /**
64       * Specifies the initial settings to send to peer.
65       *
66       * @param initialSettings non default initial settings to send to peer
67       * @return {@link Http2CodecBuilder} the builder for the {@link Http2Codec}
68       */
69      public Http2CodecBuilder initialSettings(Http2Settings initialSettings) {
70          this.initialSettings = initialSettings;
71          return this;
72      }
73  
74      /**
75       * Returns the initial settings to send to peer.
76       */
77      public Http2Settings initialSettings() {
78          return initialSettings;
79      }
80  
81      /**
82       * Specifies the frame logger to log messages with.
83       *
84       * @param frameLogger handler used to log all frames
85       * @return {@link Http2CodecBuilder} the builder for the {@link Http2Codec}
86       */
87      public Http2CodecBuilder frameLogger(Http2FrameLogger frameLogger) {
88          this.frameLogger = frameLogger;
89          return this;
90      }
91  
92      /**
93       * Returns the frame logger to log messages with.
94       */
95      public Http2FrameLogger frameLogger() {
96          return frameLogger;
97      }
98  
99      /**
100      * Specifies the headers sensitivity detector.
101      *
102      * @param headersSensitivityDetector decides whether headers should be considered sensitive or not
103      * @return {@link Http2CodecBuilder} the builder for the {@link Http2Codec}
104      */
105     public Http2CodecBuilder headersSensitivityDetector(SensitivityDetector headersSensitivityDetector) {
106         this.headersSensitivityDetector = headersSensitivityDetector;
107         return this;
108     }
109 
110     /**
111      * Returns the headers sensitivity detector.
112      */
113     public SensitivityDetector headersSensitivityDetector() {
114         return headersSensitivityDetector;
115     }
116 
117     private Http2FrameWriter frameWriter() {
118         return headersSensitivityDetector() == null ?
119             new DefaultHttp2FrameWriter() :
120             new DefaultHttp2FrameWriter(headersSensitivityDetector());
121     }
122 
123     /**
124      * Builds/creates a new {@link Http2Codec} instance using this builder's current settings.
125      */
126     public Http2Codec build() {
127         return new Http2Codec(server, bootstrap,
128             frameWriter(), frameLogger(), initialSettings());
129     }
130 }
