41  /**
42   * An HTTP/2 handler that creates child channels for each stream. Creating outgoing streams is not
43   * yet supported. Server-side HTTP to HTTP/2 upgrade is supported in conjunction with {@link
44   * Http2ServerUpgradeCodec}; the necessary HTTP-to-HTTP/2 conversion is performed automatically.
45   *
46   * <p><em>This API is very immature.</em> The Http2Connection-based API is currently preferred over
47   * this API. This API is targeted to eventually replace or reduce the need for the
48   * Http2Connection-based API.
49   *
50   * <p>This handler notifies the pipeline of channel events, such as {@link Http2GoAwayFrame}. It
51   * is also capable of writing such messages. Directly writing {@link Http2StreamFrame}s for this
52   * handler is unsupported.
53   *
54   * <h3>Child Channels</h3>
55   *
56   * <p>When a new stream is created, a new {@link Channel} is created for it. Applications send and
57   * receive {@link Http2StreamFrame}s on the created channel. The {@link Http2StreamFrame#stream} is
58   * expected to be {@code null}, but the channel can use the field for its own bookkeeping. {@link
59   * ByteBuf}s cannot be processed by the channel; all writes that reach the head of the pipeline must
60   * be an instance of {@link Http2StreamFrame}.  Writes that reach the head of the pipeline are
61   * processed directly by this handler and cannot be intercepted.
62   *
63   * <p>The child channel will be notified of user events that impact the stream, such as {@link
64   * Http2GoAwayFrame} and {@link Http2ResetFrame}, as soon as they occur. Although {@code
65   * Http2GoAwayFrame} and {@code Http2ResetFrame} signify that the remote is ignoring further
66   * communication, closing of the channel is delayed until any inbound queue is drained with {@link
67   * Channel#read()}, which follows the default behavior of channels in Netty. Applications are
68   * free to close the channel in response to such events if they don't have use for any queued
69   * messages.
70   *
71   * <p>{@link ChannelConfig#setMaxMessagesPerRead(int)} and {@link
72   * ChannelConfig#setAutoRead(boolean)} are supported.
73   */
74  public final class Http2MultiplexCodec extends ChannelDuplexHandler {

109     Http2MultiplexCodec(boolean server, ChannelHandler streamHandler,
110             EventLoopGroup streamGroup, Http2FrameWriter frameWriter) {
111         if (!streamHandler.getClass().isAnnotationPresent(Sharable.class)) {
112             throw new IllegalArgumentException("streamHandler must be Sharable");
113         }
114         this.streamHandler = streamHandler;
115         this.streamGroup = streamGroup;
116         Http2Connection connection = new DefaultHttp2Connection(server);
117         frameWriter = new Http2OutboundFrameLogger(frameWriter, HTTP2_FRAME_LOGGER);
118         Http2ConnectionEncoder encoder = new DefaultHttp2ConnectionEncoder(connection, frameWriter);
119         Http2FrameReader reader = new Http2InboundFrameLogger(new DefaultHttp2FrameReader(), HTTP2_FRAME_LOGGER);
120         Http2ConnectionDecoder decoder = new DefaultHttp2ConnectionDecoder(connection, encoder, reader);
121         decoder.frameListener(new FrameListener());
122         http2Handler = new InternalHttp2ConnectionHandler(decoder, encoder, new Http2Settings());
123         http2Handler.connection().addListener(new ConnectionListener());
124         streamInfoKey = http2Handler.connection().newKey();
125     }

149     /**
150      * Handles the cleartext HTTP upgrade event. If an upgrade occurred, sends a simple response via
151      * HTTP/2 on stream 1 (the stream specifically reserved for cleartext HTTP upgrade).
152      */
153     @Override
154     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
155         if (!(evt instanceof UpgradeEvent)) {
156             super.userEventTriggered(ctx, evt);
157             return;
158         }
159 
160         UpgradeEvent upgrade = (UpgradeEvent) evt;
161         ctx.fireUserEventTriggered(upgrade.retain());
162         try {
163             Http2Stream stream = http2Handler.connection().stream(Http2CodecUtil.HTTP_UPGRADE_STREAM_ID);
164             // TODO: improve handler/stream lifecycle so that stream isn't active before handler added.
165             // The stream was already made active, but ctx may have been null so it wasn't initialized.
166             // https://github.com/netty/netty/issues/4942
167             new ConnectionListener().onStreamActive(stream);
168             upgrade.upgradeRequest().headers().setInt(
169                 HttpConversionUtil.ExtensionHeaderNames.STREAM_ID.text(), Http2CodecUtil.HTTP_UPGRADE_STREAM_ID);
170             new InboundHttpToHttp2Adapter(http2Handler.connection(), http2Handler.decoder().frameListener())
171                 .channelRead(ctx, upgrade.upgradeRequest().retain());
172         } finally {
173             upgrade.release();
174         }
175     }

177     /**
178      * Processes all {@link Http2Frame}s. {@link Http2StreamFrame}s may only originate in child
179      * streams.
180      */
181     @Override
182     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)
183             throws Exception {
184         if (!(msg instanceof Http2Frame)) {
185             super.write(ctx, msg, promise);
186             return;
187         }
188         try {
189             if (msg instanceof Http2StreamFrame) {
190                 Object streamObject = ((Http2StreamFrame) msg).stream();
191                 int streamId = ((Http2StreamChannel) streamObject).stream.id();
192                 if (msg instanceof Http2DataFrame) {
193                     Http2DataFrame frame = (Http2DataFrame) msg;
194                     http2Handler.encoder().writeData(http2HandlerCtx, streamId, frame.content().retain(),
195                         frame.padding(), frame.isEndStream(), promise);
196                 } else if (msg instanceof Http2HeadersFrame) {
197                     Http2HeadersFrame frame = (Http2HeadersFrame) msg;
198                     http2Handler.encoder().writeHeaders(
199                         http2HandlerCtx, streamId, frame.headers(), frame.padding(), frame.isEndStream(), promise);
200                 } else if (msg instanceof Http2ResetFrame) {
201                     Http2ResetFrame frame = (Http2ResetFrame) msg;
202                     http2Handler.resetStream(http2HandlerCtx, streamId, frame.errorCode(), promise);
203                 } else {
204                     throw new UnsupportedMessageTypeException(msg);
205                 }
206             } else if (msg instanceof Http2GoAwayFrame) {
207                 Http2GoAwayFrame frame = (Http2GoAwayFrame) msg;
208                 int lastStreamId = http2Handler.connection().remote().lastStreamCreated()
209                     + frame.extraStreamIds() * 2;
210                 http2Handler.goAway(
211                     http2HandlerCtx, lastStreamId, frame.errorCode(), frame.content().retain(), promise);
212             } else {
213                 throw new UnsupportedMessageTypeException(msg);
214             }
215         } finally {
216             ReferenceCountUtil.release(msg);
217         }
218     }

220     ChannelFuture createStreamChannel(ChannelHandlerContext ctx, Http2Stream stream,
221             ChannelHandler handler) {
222         EventLoopGroup group = streamGroup != null ? streamGroup : ctx.channel().eventLoop();
223         Http2StreamChannel channel = new Http2StreamChannel(stream);
224         channel.pipeline().addLast(handler);
225         ChannelFuture future = group.register(channel);
226         // Handle any errors that occurred on the local thread while registering. Even though
227         // failures can happen after this point, they will be handled by the channel by closing the
228         // channel.
229         if (future.cause() != null) {
230             if (channel.isRegistered()) {
231                 channel.close();
232             } else {
233                 channel.unsafe().closeForcibly();
234             }
235         }
236         return future;
237     }

239     /**
240      * Notifies any child streams of the read completion.
241      */
242     @Override
243     public void channelReadComplete(ChannelHandlerContext ctx) {
244         for (int i = 0; i < streamsToFireChildReadComplete.size(); i++) {
245             final StreamInfo streamInfo = streamsToFireChildReadComplete.get(i);
246             // Clear early in case fireChildReadComplete() causes it to need to be re-processed
247             streamInfo.inStreamsToFireChildReadComplete = false;
248             streamInfo.childChannel.fireChildReadComplete();
249         }
250         streamsToFireChildReadComplete.clear();
251     }
252 
253     void fireChildReadAndRegister(StreamInfo streamInfo, Object msg) {
254         // Can't use childChannel.fireChannelRead() as it would fire independent of whether
255         // channel.read() had been called.
256         streamInfo.childChannel.fireChildRead(msg);
257         if (!streamInfo.inStreamsToFireChildReadComplete) {
258             streamsToFireChildReadComplete.add(streamInfo);
259             streamInfo.inStreamsToFireChildReadComplete = true;
260         }
261     }

263     final class ConnectionListener extends Http2ConnectionAdapter {
264         @Override
265         public void onStreamActive(Http2Stream stream) {
266             if (ctx == null) {
267                 // UPGRADE stream is active before handlerAdded().
268                 return;
269             }
270             // If it is an outgoing stream, then we already created the channel.
271             // TODO: support outgoing streams. https://github.com/netty/netty/issues/4913
272             if (stream.getProperty(streamInfoKey) != null) {
273                 return;
274             }
275             ChannelFuture future = createStreamChannel(ctx, stream, streamHandler);
276             stream.setProperty(streamInfoKey, new StreamInfo((Http2StreamChannel) future.channel()));
277         }
278 
279         @Override
280         public void onStreamClosed(Http2Stream stream) {
281             final StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
282             if (streamInfo != null) {
283                 final EventLoop eventLoop = streamInfo.childChannel.eventLoop();
284                 if (eventLoop.inEventLoop()) {
285                     onStreamClosed0(streamInfo);
286                 } else {
287                     eventLoop.execute(new OneTimeTask() {
288                         @Override
289                         public void run() {
290                             onStreamClosed0(streamInfo);
291                         }
292                     });
293                 }
294             }
295         }
296 
297         private void onStreamClosed0(StreamInfo streamInfo) {
298             streamInfo.childChannel.onStreamClosedFired = true;
299             streamInfo.childChannel.fireChildRead(AbstractHttp2StreamChannel.CLOSE_MESSAGE);
300         }
301 
302         @Override
303         public void onGoAwayReceived(final int lastStreamId, long errorCode, ByteBuf debugData) {
304             final Http2GoAwayFrame goAway = new DefaultHttp2GoAwayFrame(errorCode, debugData);
305             try {
306                 http2Handler.connection().forEachActiveStream(new Http2StreamVisitor() {
307                     @Override
308                     public boolean visit(Http2Stream stream) {
309                         if (stream.id() > lastStreamId
310                                 && http2Handler.connection().local().isValidStreamId(stream.id())) {
311                             StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
312                             // TODO: Can we force a user interaction pattern that doesn't require us to duplicate()?
313                             // https://github.com/netty/netty/issues/4943
314                             streamInfo.childChannel.pipeline().fireUserEventTriggered(goAway.duplicate().retain());
315                         }
316                         return true;
317                     }
318                 });
319             } catch (Throwable t) {
320                 ctx.invoker().invokeExceptionCaught(ctx, t);
321             }
322             ctx.fireUserEventTriggered(goAway.duplicate().retain());
323         }
324     }

332         @Override
333         protected void onStreamError(ChannelHandlerContext ctx, Throwable cause,
334                 Http2Exception.StreamException http2Ex) {
335             try {
336                 Http2Stream stream = http2Handler.connection().stream(http2Ex.streamId());
337                 if (stream == null) {
338                     return;
339                 }
340                 StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
341                 if (streamInfo == null) {
342                     return;
343                 }
344                 streamInfo.childChannel.pipeline().fireExceptionCaught(http2Ex);
345             } finally {
346                 super.onStreamError(ctx, cause, http2Ex);
347             }
348         }

351     class FrameListener extends Http2FrameAdapter {
352         @Override
353         public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode)
354                 throws Http2Exception {
355             Http2Stream stream = http2Handler.connection().stream(streamId);
356             StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
357             // Use a user event in order to circumvent read queue.
358             streamInfo.childChannel.pipeline().fireUserEventTriggered(new DefaultHttp2ResetFrame(errorCode));
359         }
360 
361         @Override
362         public void onHeadersRead(ChannelHandlerContext ctx, int streamId,
363                 Http2Headers headers, int streamDependency, short weight, boolean
364                 exclusive, int padding, boolean endStream) throws Http2Exception {
365             onHeadersRead(ctx, streamId, headers, padding, endStream);
366         }
367 
368         @Override
369         public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers,
370                 int padding, boolean endOfStream) throws Http2Exception {
371             Http2Stream stream = http2Handler.connection().stream(streamId);
372             StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
373             fireChildReadAndRegister(streamInfo, new DefaultHttp2HeadersFrame(headers, endOfStream, padding));
374         }
375 
376         @Override
377         public int onDataRead(ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding,
378                 boolean endOfStream) throws Http2Exception {
379             Http2Stream stream = http2Handler.connection().stream(streamId);
380             StreamInfo streamInfo = (StreamInfo) stream.getProperty(streamInfoKey);
381             fireChildReadAndRegister(streamInfo, new DefaultHttp2DataFrame(data.retain(), endOfStream, padding));
382             // We return the bytes in bytesConsumed() once the stream channel consumed the bytes.
383             return 0;
384         }

399     // This class uses ctx.invoker().invoke* instead of ctx.* to send to the ctx's handler instead
400     // of the 'next' handler.
401     final class Http2StreamChannel extends AbstractHttp2StreamChannel {
402         private final Http2Stream stream;
403         boolean onStreamClosedFired;
404 
405         Http2StreamChannel(Http2Stream stream) {
406             super(ctx.channel());
407             this.stream = stream;
408         }
409 
410         @Override
411         protected void doClose() throws Exception {
412             if (!onStreamClosedFired) {
413                 Http2StreamFrame resetFrame = new DefaultHttp2ResetFrame(Http2Error.CANCEL).setStream(this);
414                 ChannelHandlerInvoker invoker = ctx.invoker();
415                 invoker.invokeWrite(ctx, resetFrame, ctx.newPromise());
416                 invoker.invokeFlush(ctx);
417             }
418             super.doClose();
419         }
420 
421         @Override
422         protected void doWrite(Object msg) {
423             if (!(msg instanceof Http2StreamFrame)) {
424                 ReferenceCountUtil.release(msg);
425                 throw new IllegalArgumentException("Message must be an Http2StreamFrame: " + msg);
426             }
427             Http2StreamFrame frame = (Http2StreamFrame) msg;
428             if (frame.stream() != null) {
429                 ReferenceCountUtil.release(frame);
430                 throw new IllegalArgumentException("Stream must be null on the frame");
431             }
432             frame.setStream(this);
433             ctx.invoker().invokeWrite(ctx, frame, ctx.newPromise());
434         }
435 
436         @Override
437         protected void doWriteComplete() {
438             ctx.invoker().invokeFlush(ctx);
439         }
440 
441         @Override
442         protected EventExecutor preferredEventExecutor() {
443             return ctx.executor();
444         }
445 
446         @Override
447         protected void bytesConsumed(final int bytes) {
448             EventExecutor executor = ctx.executor();
449             if (executor.inEventLoop()) {
450                 bytesConsumed0(bytes);
451             } else {
452                 executor.execute(new OneTimeTask() {
453                     @Override
454                     public void run() {
455                         bytesConsumed0(bytes);
456                     }
457                 });
458             }
459         }
460 
461         private void bytesConsumed0(int bytes) {
462             try {
463                 http2Handler.connection().local().flowController().consumeBytes(stream, bytes);
464             } catch (Throwable t) {
465                 ChannelHandlerInvokerUtil.invokeExceptionCaughtNow(ctx, t);
466             }
467         }
468     }
