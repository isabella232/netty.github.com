39  /**
40   * An HTTP/2 handler that creates child channels for each stream. This handler must be used in combination
41   * with {@link Http2FrameCodec}.
42   *
43   * <p>When a new stream is created, a new {@link Channel} is created for it. Applications send and
44   * receive {@link Http2StreamFrame}s on the created channel. {@link ByteBuf}s cannot be processed by the channel;
45   * all writes that reach the head of the pipeline must be an instance of {@link Http2StreamFrame}. Writes that reach
46   * the head of the pipeline are processed directly by this handler and cannot be intercepted.
47   *
48   * <p>The child channel will be notified of user events that impact the stream, such as {@link
49   * Http2GoAwayFrame} and {@link Http2ResetFrame}, as soon as they occur. Although {@code
50   * Http2GoAwayFrame} and {@code Http2ResetFrame} signify that the remote is ignoring further
51   * communication, closing of the channel is delayed until any inbound queue is drained with {@link
52   * Channel#read()}, which follows the default behavior of channels in Netty. Applications are
53   * free to close the channel in response to such events if they don't have use for any queued
54   * messages. Any connection level events like {@link Http2SettingsFrame} and {@link Http2GoAwayFrame}
55   * will be processed internally and also propagated down the pipeline for other handlers to act on.
56   *
57   * <p>Outbound streams are supported via the {@link Http2StreamChannelBootstrap}.
58   *
59   * <p>{@link ChannelConfig#setMaxMessagesPerRead(int)} and {@link ChannelConfig#setAutoRead(boolean)} are supported.
60   *
61   * <h3>Reference Counting</h3>
62   *
63   * Some {@link Http2StreamFrame}s implement the {@link ReferenceCounted} interface, as they carry
64   * reference counted objects (e.g. {@link ByteBuf}s). The multiplex codec will call {@link ReferenceCounted#retain()}
65   * before propagating a reference counted object through the pipeline, and thus an application handler needs to release
66   * such an object after having consumed it. For more information on reference counting take a look at
67   * http://netty.io/wiki/reference-counted-objects.html
68   *
69   * <h3>Channel Events</h3>
70   *
71   * A child channel becomes active as soon as it is registered to an {@link EventLoop}. Therefore, an active channel
72   * does not map to an active HTTP/2 stream immediately. Only once a {@link Http2HeadersFrame} has been successfully sent
73   * or received, does the channel map to an active HTTP/2 stream. In case it is not possible to open a new HTTP/2 stream
74   * (i.e. due to the maximum number of active streams being exceeded), the child channel receives an exception
75   * indicating the cause and is closed immediately thereafter.
76   *
77   * <h3>Writability and Flow Control</h3>
78   *
79   * A child channel observes outbound/remote flow control via the channel's writability. A channel only becomes writable
80   * when it maps to an active HTTP/2 stream . A child channel does not know about the connection-level flow control
81   * window. {@link ChannelHandler}s are free to ignore the channel's writability, in which case the excessive writes will
82   * be buffered by the parent channel. It's important to note that only {@link Http2DataFrame}s are subject to
83   * HTTP/2 flow control.
84   */
85  @UnstableApi

131     static void registerDone(ChannelFuture future) {
132         // Handle any errors that occurred on the local thread while registering. Even though
133         // failures can happen after this point, they will be handled by the channel by closing the
134         // childChannel.
135         if (!future.isSuccess()) {
136             Channel childChannel = future.channel();
137             if (childChannel.isRegistered()) {
138                 childChannel.close();
139             } else {
140                 childChannel.unsafe().closeForcibly();
141             }
142         }
143     }

158     @Override
159     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
160         parentReadInProgress = true;
161         if (msg instanceof Http2StreamFrame) {
162             if (msg instanceof Http2WindowUpdateFrame) {
163                 // We dont want to propagate update frames to the user
164                 return;
165             }
166             Http2StreamFrame streamFrame = (Http2StreamFrame) msg;
167             DefaultHttp2FrameStream s =
168                     (DefaultHttp2FrameStream) streamFrame.stream();
169 
170             AbstractHttp2StreamChannel channel = (AbstractHttp2StreamChannel) s.attachment;
171             if (msg instanceof Http2ResetFrame) {
172                 // Reset frames needs to be propagated via user events as these are not flow-controlled and so
173                 // must not be controlled by suppressing channel.read() on the child channel.
174                 channel.pipeline().fireUserEventTriggered(msg);
175 
176                 // RST frames will also trigger closing of the streams which then will call
177                 // AbstractHttp2StreamChannel.streamClosed()
178             } else {
179                 channel.fireChildRead(streamFrame);
180             }
181             return;
182         }
183 
184         if (msg instanceof Http2GoAwayFrame) {
185             // goaway frames will also trigger closing of the streams which then will call
186             // AbstractHttp2StreamChannel.streamClosed()
187             onHttp2GoAwayFrame(ctx, (Http2GoAwayFrame) msg);
188         }
189 
190         // Send everything down the pipeline
191         ctx.fireChannelRead(msg);
192     }

196         if (ctx.channel().isWritable()) {
197             // While the writability state may change during iterating of the streams we just set all of the streams

205     @Override
206     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
207         if (evt instanceof Http2FrameStreamEvent) {
208             Http2FrameStreamEvent event = (Http2FrameStreamEvent) evt;
209             DefaultHttp2FrameStream stream = (DefaultHttp2FrameStream) event.stream();
210             if (event.type() == Http2FrameStreamEvent.Type.State) {
211                 switch (stream.state()) {
212                     case HALF_CLOSED_LOCAL:
213                         if (stream.id() != Http2CodecUtil.HTTP_UPGRADE_STREAM_ID) {
214                             // Ignore everything which was not caused by an upgrade
215                             break;
216                         }
217                         // We must have an upgrade handler or else we can't handle the stream
218                         if (upgradeStreamHandler == null) {
219                             throw connectionError(INTERNAL_ERROR,
220                                     "Client is misconfigured for upgrade requests");
221                         }
222                         // fall-trough
223                     case HALF_CLOSED_REMOTE:
224                         // fall-trough
225                     case OPEN:
226                         if (stream.attachment != null) {
227                             // ignore if child channel was already created.
228                             break;
229                         }
230                         final AbstractHttp2StreamChannel ch;
231                         if (stream.state() == Http2Stream.State.HALF_CLOSED_LOCAL) {
232                             ch = new Http2MultiplexHandlerStreamChannel(stream, null);
233                             ch.closeOutbound();
234                             // Add our upgrade handler to the channel and then register the channel.
235                             // The register call fires the channelActive, etc.
236                             ch.pipeline().addLast(upgradeStreamHandler);
237                         } else {
238                             ch = new Http2MultiplexHandlerStreamChannel(stream, inboundStreamHandler);
239                         }
240                         ChannelFuture future = ctx.channel().eventLoop().register(ch);
241                         if (future.isDone()) {
242                             registerDone(future);
243                         } else {
244                             future.addListener(CHILD_CHANNEL_REGISTRATION_LISTENER);
245                         }
246                         break;
247                     case CLOSED:
248                         AbstractHttp2StreamChannel channel = (AbstractHttp2StreamChannel) stream.attachment;
249                         if (channel != null) {
250                             channel.streamClosed();
251                         }
252                         break;
253                     default:
254                         // ignore for now
255                         break;
256                 }
257             }
258             return;
259         }
260         ctx.fireUserEventTriggered(evt);
261     }

268     @Override
269     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
270         if (cause instanceof Http2FrameStreamException) {
271             Http2FrameStreamException exception = (Http2FrameStreamException) cause;
272             Http2FrameStream stream = exception.stream();
273             AbstractHttp2StreamChannel childChannel = (AbstractHttp2StreamChannel)
274                     ((DefaultHttp2FrameStream) stream).attachment;
275             try {
276                 childChannel.pipeline().fireExceptionCaught(cause.getCause());
277             } finally {
278                 childChannel.unsafe().closeForcibly();
279             }
280             return;
281         }
282         ctx.fireExceptionCaught(cause);
283     }

285     private void onHttp2GoAwayFrame(ChannelHandlerContext ctx, final Http2GoAwayFrame goAwayFrame) {
286         try {
287             final boolean server = ctx.channel().parent() instanceof ServerChannel;
288             forEachActiveStream(new Http2FrameStreamVisitor() {
289                 @Override
290                 public boolean visit(Http2FrameStream stream) {
291                     final int streamId = stream.id();
292                     if (streamId > goAwayFrame.lastStreamId() && Http2CodecUtil.isStreamIdValid(streamId, server)) {
293                         final AbstractHttp2StreamChannel childChannel = (AbstractHttp2StreamChannel)
294                                 ((DefaultHttp2FrameStream) stream).attachment;
295                         childChannel.pipeline().fireUserEventTriggered(goAwayFrame.retainedDuplicate());
296                     }
297                     return true;
298                 }
299             });
300         } catch (Http2Exception e) {
301             ctx.fireExceptionCaught(e);
302             ctx.close();
303         }
304     }

315     private void processPendingReadCompleteQueue() {
316         parentReadInProgress = true;
317         // If we have many child channel we can optimize for the case when multiple call flush() in
318         // channelReadComplete(...) callbacks and only do it once as otherwise we will end-up with multiple
319         // write calls on the socket which is expensive.
320         AbstractHttp2StreamChannel childChannel = readCompletePendingQueue.poll();
321         if (childChannel != null) {
322             try {
323                 do {
324                     childChannel.fireChildReadComplete();
325                     childChannel = readCompletePendingQueue.poll();
326                 } while (childChannel != null);
327             } finally {
328                 parentReadInProgress = false;
329                 readCompletePendingQueue.clear();
330                 ctx.flush();
331             }
332         } else {
333             parentReadInProgress = false;
334         }
335     }

343         @Override
344         protected boolean isParentReadInProgress() {
345             return parentReadInProgress;
346         }
347 
348         @Override
349         protected void addChannelToReadCompletePendingQueue() {
350             // If there is no space left in the queue, just keep on processing everything that is already
351             // stored there and try again.
352             while (!readCompletePendingQueue.offer(this)) {
353                 processPendingReadCompleteQueue();
354             }
355         }

357         @Override
358         protected ChannelHandlerContext parentContext() {
359             return ctx;
360         }
