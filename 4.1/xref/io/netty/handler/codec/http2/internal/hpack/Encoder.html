87      /**
88       * Creates a new encoder.
89       */
90      public Encoder(boolean ignoreMaxHeaderListSize, int arraySizeHint) {
91          this.ignoreMaxHeaderListSize = ignoreMaxHeaderListSize;
92          maxHeaderTableSize = DEFAULT_HEADER_TABLE_SIZE;
93          maxHeaderListSize = DEFAULT_HEADER_LIST_SIZE;
94          // Enforce a bound of [2, 128] because hashMask is a byte. The max possible value of hashMask is one less
95          // than the length of this array, and we want the mask to be > 0.
96          headerFields = new HeaderEntry[findNextPositivePowerOfTwo(max(2, min(arraySizeHint, 128)))];
97          hashMask = (byte) (headerFields.length - 1);
98          head.before = head.after = head;
99      }

101     /**
102      * Encode the header field into the header block.
103      *
104      * <strong>The given {@link CharSequence}s must be immutable!</strong>
105      */
106     public void encodeHeaders(int streamId, ByteBuf out, Http2Headers headers, SensitivityDetector sensitivityDetector)
107             throws Http2Exception {
108         if (ignoreMaxHeaderListSize) {
109             encodeHeadersIgnoreMaxHeaderListSize(out, headers, sensitivityDetector);
110         } else {
111             encodeHeadersEnforceMaxHeaderListSize(streamId, out, headers, sensitivityDetector);
112         }
113     }

115     private void encodeHeadersEnforceMaxHeaderListSize(int streamId, ByteBuf out, Http2Headers headers,
116                                                        SensitivityDetector sensitivityDetector)
117             throws Http2Exception {
118         long headerSize = 0;
119         for (Map.Entry<CharSequence, CharSequence> header : headers) {
120             CharSequence name = header.getKey();
121             CharSequence value = header.getValue();
122             long currHeaderSize = HeaderField.sizeOf(name, value);
123             // OK to increment now and check for bounds after because this value is limited to unsigned int and will not
124             // overflow.
125             headerSize += currHeaderSize;
126             if (headerSize > maxHeaderListSize) {
127                 headerListSizeExceeded(streamId, maxHeaderListSize, false);
128             }
129             encodeHeader(out, name, value, sensitivityDetector.isSensitive(name, value), currHeaderSize);
130         }
131     }

133     private void encodeHeadersIgnoreMaxHeaderListSize(ByteBuf out, Http2Headers headers,
134                                                       SensitivityDetector sensitivityDetector) throws Http2Exception {
135         for (Map.Entry<CharSequence, CharSequence> header : headers) {
136             CharSequence name = header.getKey();
137             CharSequence value = header.getValue();
138             encodeHeader(out, name, value, sensitivityDetector.isSensitive(name, value),
139                          HeaderField.sizeOf(name, value));
140         }
141     }

143     /**
144      * Encode the header field into the header block.
145      *
146      * <strong>The given {@link CharSequence}s must be immutable!</strong>
147      */
148     private void encodeHeader(ByteBuf out, CharSequence name, CharSequence value, boolean sensitive, long headerSize) {
149         // If the header value is sensitive then it must never be indexed
150         if (sensitive) {
151             int nameIndex = getNameIndex(name);
152             encodeLiteral(out, name, value, NEVER, nameIndex);
153             return;
154         }
155 
156         // If the peer will only use the static table
157         if (maxHeaderTableSize == 0) {
158             int staticTableIndex = StaticTable.getIndex(name, value);
159             if (staticTableIndex == -1) {
160                 int nameIndex = StaticTable.getIndex(name);
161                 encodeLiteral(out, name, value, NONE, nameIndex);
162             } else {
163                 encodeInteger(out, 0x80, 7, staticTableIndex);
164             }
165             return;
166         }
167 
168         // If the headerSize is greater than the max table size then it must be encoded literally
169         if (headerSize > maxHeaderTableSize) {
170             int nameIndex = getNameIndex(name);
171             encodeLiteral(out, name, value, NONE, nameIndex);
172             return;
173         }
174 
175         HeaderEntry headerField = getEntry(name, value);
176         if (headerField != null) {
177             int index = getIndex(headerField.index) + StaticTable.length;
178             // Section 6.1. Indexed Header Field Representation
179             encodeInteger(out, 0x80, 7, index);
180         } else {
181             int staticTableIndex = StaticTable.getIndex(name, value);
182             if (staticTableIndex != -1) {
183                 // Section 6.1. Indexed Header Field Representation
184                 encodeInteger(out, 0x80, 7, staticTableIndex);
185             } else {
186                 ensureCapacity(headerSize);
187                 encodeLiteral(out, name, value, INCREMENTAL, getNameIndex(name));
188                 add(name, value, headerSize);
189             }
190         }
191     }

193     /**
194      * Set the maximum table size.
195      */
196     public void setMaxHeaderTableSize(ByteBuf out, long maxHeaderTableSize) throws Http2Exception {
197         if (maxHeaderTableSize < MIN_HEADER_TABLE_SIZE || maxHeaderTableSize > MAX_HEADER_TABLE_SIZE) {
198             throw connectionError(PROTOCOL_ERROR, "Header Table Size must be >= %d and <= %d but was %d",
199                     MIN_HEADER_TABLE_SIZE, MAX_HEADER_TABLE_SIZE, maxHeaderTableSize);
200         }
201         if (this.maxHeaderTableSize == maxHeaderTableSize) {
202             return;
203         }
204         this.maxHeaderTableSize = maxHeaderTableSize;
205         ensureCapacity(0);
206         // Casting to integer is safe as we verified the maxHeaderTableSize is a valid unsigned int.
207         encodeInteger(out, 0x20, 5, (int) maxHeaderTableSize);
208     }

229     /**
230      * Encode integer according to Section 5.1.
231      */
232     private static void encodeInteger(ByteBuf out, int mask, int n, int i) {
233         assert n >= 0 && n <= 8 : "N: " + n;
234         int nbits = 0xFF >>> (8 - n);
235         if (i < nbits) {
236             out.writeByte(mask | i);
237         } else {
238             out.writeByte(mask | nbits);
239             int length = i - nbits;
240             for (; (length & ~0x7F) != 0; length >>>= 7) {
241                 out.writeByte((length & 0x7F) | 0x80);
242             }
243             out.writeByte(length);
244         }
245     }

247     /**
248      * Encode string literal according to Section 5.2.
249      */
250     private void encodeStringLiteral(ByteBuf out, CharSequence string) {
251         int huffmanLength = huffmanEncoder.getEncodedLength(string);
252         if (huffmanLength < string.length()) {
253             encodeInteger(out, 0x80, 7, huffmanLength);
254             huffmanEncoder.encode(out, string);
255         } else {
256             encodeInteger(out, 0x00, 7, string.length());
257             if (string instanceof AsciiString) {
258                 // Fast-path
259                 AsciiString asciiString = (AsciiString) string;
260                 out.writeBytes(asciiString.array(), asciiString.arrayOffset(), asciiString.length());
261             } else {
262                 // Only ASCII is allowed in http2 headers, so its fine to use this.
263                 // https://tools.ietf.org/html/rfc7540#section-8.1.2
264                 out.writeCharSequence(string, CharsetUtil.ISO_8859_1);
265             }
266         }
267     }

269     /**
270      * Encode literal header field according to Section 6.2.
271      */
272     private void encodeLiteral(ByteBuf out, CharSequence name, CharSequence value, HpackUtil.IndexType indexType,
273                                int nameIndex) {
274         boolean nameIndexValid = nameIndex != -1;
275         switch (indexType) {
276             case INCREMENTAL:
277                 encodeInteger(out, 0x40, 6, nameIndexValid ? nameIndex : 0);
278                 break;
279             case NONE:
280                 encodeInteger(out, 0x00, 4, nameIndexValid ? nameIndex : 0);
281                 break;
282             case NEVER:
283                 encodeInteger(out, 0x10, 4, nameIndexValid ? nameIndex : 0);
284                 break;
285             default:
286                 throw new Error("should not reach here");
287         }
288         if (!nameIndexValid) {
289             encodeStringLiteral(out, name);
290         }
291         encodeStringLiteral(out, value);
292     }

294     private int getNameIndex(CharSequence name) {
295         int index = StaticTable.getIndex(name);
296         if (index == -1) {
297             index = getIndex(name);
298             if (index >= 0) {
299                 index += StaticTable.length;
300             }
301         }
302         return index;
303     }

305     /**
306      * Ensure that the dynamic table has enough room to hold 'headerSize' more bytes. Removes the
307      * oldest entry from the dynamic table until sufficient space is available.
308      */
309     private void ensureCapacity(long headerSize) {
310         while (maxHeaderTableSize - size < headerSize) {
311             int index = length();
312             if (index == 0) {
313                 break;
314             }
315             remove();
316         }
317     }

344     /**
345      * Returns the header entry with the lowest index value for the header field. Returns null if
346      * header field is not in the dynamic table.
347      */
348     private HeaderEntry getEntry(CharSequence name, CharSequence value) {
349         if (length() == 0 || name == null || value == null) {
350             return null;
351         }
352         int h = AsciiString.hashCode(name);
353         int i = index(h);
354         for (HeaderEntry e = headerFields[i]; e != null; e = e.next) {
355             // To avoid short circuit behavior a bitwise operator is used instead of a boolean operator.
356             if (e.hash == h && (equalsConstantTime(name, e.name) & equalsConstantTime(value, e.value)) != 0) {
357                 return e;
358             }
359         }
360         return null;
361     }
362 
363     /**
364      * Returns the lowest index value for the header field name in the dynamic table. Returns -1 if
365      * the header field name is not in the dynamic table.
366      */
367     private int getIndex(CharSequence name) {
368         if (length() == 0 || name == null) {
369             return -1;
370         }
371         int h = AsciiString.hashCode(name);
372         int i = index(h);
373         for (HeaderEntry e = headerFields[i]; e != null; e = e.next) {
374             if (e.hash == h && equalsConstantTime(name, e.name) != 0) {
375                 return getIndex(e.index);
376             }
377         }
378         return -1;
379     }

381     /**
382      * Compute the index into the dynamic table given the index in the header entry.
383      */
384     private int getIndex(int index) {
385         return index == -1 ? -1 : index - head.before.index + 1;
386     }

388     /**
389      * Add the header field to the dynamic table. Entries are evicted from the dynamic table until
390      * the size of the table and the new header field is less than the table's maxHeaderTableSize. If the size
391      * of the new entry is larger than the table's maxHeaderTableSize, the dynamic table will be cleared.
392      */
393     private void add(CharSequence name, CharSequence value, long headerSize) {
394         // Clear the table if the header field size is larger than the maxHeaderTableSize.
395         if (headerSize > maxHeaderTableSize) {
396             clear();
397             return;
398         }
399 
400         // Evict oldest entries until we have enough maxHeaderTableSize.
401         while (maxHeaderTableSize - size < headerSize) {
402             remove();
403         }
404 
405         int h = AsciiString.hashCode(name);
406         int i = index(h);
407         HeaderEntry old = headerFields[i];
408         HeaderEntry e = new HeaderEntry(h, name, value, head.before.index - 1, old);
409         headerFields[i] = e;
410         e.addBefore(head);
411         size += headerSize;
412     }

414     /**
415      * Remove and return the oldest header field from the dynamic table.
416      */
417     private HeaderField remove() {
418         if (size == 0) {
419             return null;
420         }
421         HeaderEntry eldest = head.after;
422         int h = eldest.hash;
423         int i = index(h);
424         HeaderEntry prev = headerFields[i];
425         HeaderEntry e = prev;
426         while (e != null) {
427             HeaderEntry next = e.next;
428             if (e == eldest) {
429                 if (prev == eldest) {
430                     headerFields[i] = next;
431                 } else {
432                     prev.next = next;
433                 }
434                 eldest.remove();
435                 size -= eldest.size();
436                 return eldest;
437             }
438             prev = e;
439             e = next;
440         }
441         return null;
442     }

444     /**
445      * Remove all entries from the dynamic table.
446      */
447     private void clear() {
448         Arrays.fill(headerFields, null);
449         head.before = head.after = head;
450         size = 0;
451     }

460     /**
461      * A linked hash map HeaderField entry.
462      */
463     private static class HeaderEntry extends HeaderField {
464         // These fields comprise the doubly linked list used for iteration.
465         HeaderEntry before, after;
466 
467         // These fields comprise the chained list for header fields with the same hash.
468         HeaderEntry next;
469         int hash;
470 
471         // This is used to compute the index in the dynamic table.
472         int index;
473 
474         /**
475          * Creates new entry.
476          */
477         HeaderEntry(int hash, CharSequence name, CharSequence value, int index, HeaderEntry next) {
478             super(name, value);
479             this.index = index;
480             this.hash = hash;
481             this.next = next;
482         }
483 
484         /**
485          * Removes this entry from the linked list.
486          */
487         private void remove() {
488             before.after = after;
489             after.before = before;
490             before = null; // null references to prevent nepotism in generational GC.
491             after = null;
492             next = null;
493         }
494 
495         /**
496          * Inserts this entry before the specified existing entry in the list.
497          */
498         private void addBefore(HeaderEntry existingEntry) {
499             after = existingEntry;
500             before = existingEntry.before;
501             before.after = this;
502             after.before = this;
503         }
504     }
