37  /**
38   * A {@link JdkSslEngine} that uses the Conscrypt provider or SSL with ALPN.
39   */
40  abstract class ConscryptAlpnSslEngine extends JdkSslEngine {
41      private static final Class<?> ENGINES_CLASS = getEnginesClass();
42  
43      /**
44       * Indicates whether or not conscrypt is available on the current system.
45       */
46      static boolean isAvailable() {
47          return ENGINES_CLASS != null && PlatformDependent.javaVersion() >= 8;
48      }
49  
50      static boolean isEngineSupported(SSLEngine engine) {
51          return isAvailable() && isConscryptEngine(engine, ENGINES_CLASS);
52      }
53  
54      static ConscryptAlpnSslEngine newClientEngine(SSLEngine engine,
55              JdkApplicationProtocolNegotiator applicationNegotiator) {
56          return new ClientEngine(engine, applicationNegotiator);
57      }
58  
59      static ConscryptAlpnSslEngine newServerEngine(SSLEngine engine,
60              JdkApplicationProtocolNegotiator applicationNegotiator) {
61          return new ServerEngine(engine, applicationNegotiator);
62      }
63  
64      private ConscryptAlpnSslEngine(SSLEngine engine, List<String> protocols) {
65          super(engine);
66  
67          // Set the list of supported ALPN protocols on the engine.
68          Conscrypt.Engines.setAlpnProtocols(engine, protocols.toArray(new String[protocols.size()]));
69      }
70  
71      /**
72       * Calculates the maximum size of the encrypted output buffer required to wrap the given plaintext bytes. Assumes
73       * as a worst case that there is one TLS record per buffer.
74       *
75       * @param plaintextBytes the number of plaintext bytes to be wrapped.
76       * @param numBuffers the number of buffers that the plaintext bytes are spread across.
77       * @return the maximum size of the encrypted output buffer required for the wrap operation.
78       */
79      final int calculateOutNetBufSize(int plaintextBytes, int numBuffers) {
80          // Assuming a max of one frame per component in a composite buffer.
81          long maxOverhead = (long) Conscrypt.Engines.maxSealOverhead(getWrappedEngine()) * numBuffers;
82          // TODO(nmittler): update this to use MAX_ENCRYPTED_PACKET_LENGTH instead of Integer.MAX_VALUE
83          return (int) min(Integer.MAX_VALUE, plaintextBytes + maxOverhead);
84      }
85  
86      final SSLEngineResult unwrap(ByteBuffer[] srcs, ByteBuffer[] dests) throws SSLException {
87          return Conscrypt.Engines.unwrap(getWrappedEngine(), srcs, dests);
88      }
89  
90      private static final class ClientEngine extends ConscryptAlpnSslEngine {
91          private final ProtocolSelectionListener protocolListener;
92  
93          ClientEngine(SSLEngine engine,
94                  JdkApplicationProtocolNegotiator applicationNegotiator) {
95              super(engine, applicationNegotiator.protocols());
96              // Register for completion of the handshake.
97              Conscrypt.Engines.setHandshakeListener(engine, new HandshakeListener() {
98                  @Override
99                  public void onHandshakeFinished() throws SSLException {
100                     selectProtocol();
101                 }
102             });
103 
104             protocolListener = checkNotNull(applicationNegotiator
105                             .protocolListenerFactory().newListener(this, applicationNegotiator.protocols()),
106                     "protocolListener");
107         }
108 
109         private void selectProtocol() throws SSLException {
110             String protocol = Conscrypt.Engines.getAlpnSelectedProtocol(getWrappedEngine());
111             try {
112                 protocolListener.selected(protocol);
113             } catch (Throwable e) {
114                 throw toSSLHandshakeException(e);
115             }
116         }
117     }
118 
119     private static final class ServerEngine extends ConscryptAlpnSslEngine {
120         private final ProtocolSelector protocolSelector;
121 
122         ServerEngine(SSLEngine engine, JdkApplicationProtocolNegotiator applicationNegotiator) {
123             super(engine, applicationNegotiator.protocols());
124 
125             // Register for completion of the handshake.
126             Conscrypt.Engines.setHandshakeListener(engine, new HandshakeListener() {
127                 @Override
128                 public void onHandshakeFinished() throws SSLException {
129                     selectProtocol();
130                 }
131             });
132 
133             protocolSelector = checkNotNull(applicationNegotiator.protocolSelectorFactory()
134                             .newSelector(this,
135                                     new LinkedHashSet<String>(applicationNegotiator.protocols())),
136                     "protocolSelector");
137         }
138 
139         private void selectProtocol() throws SSLException {
140             try {
141                 String protocol = Conscrypt.Engines.getAlpnSelectedProtocol(getWrappedEngine());
142                 protocolSelector.select(protocol != null ? Collections.singletonList(protocol)
143                         : Collections.<String>emptyList());
144             } catch (Throwable e) {
145                 throw toSSLHandshakeException(e);
146             }
147         }
148     }
149 
150     private static Class<?> getEnginesClass() {
151         try {
152             // Always use bootstrap class loader.
153             Class<?> engineClass = Class.forName("org.conscrypt.Conscrypt$Engines", true,
154                     ConscryptAlpnSslEngine.class.getClassLoader());
155             // Ensure that it also has the isConscrypt method.
156             getIsConscryptMethod(engineClass);
157             return engineClass;
158         } catch (Throwable ignore) {
159             // Conscrypt was not loaded.
160             return null;
161         }
162     }
163 
164     private static boolean isConscryptEngine(SSLEngine engine, Class<?> enginesClass) {
165         try {
166             Method method = getIsConscryptMethod(enginesClass);
167             return (Boolean) method.invoke(null, engine);
168         } catch (Throwable ignore) {
169             return false;
170         }
171     }
172 
173     private static Method getIsConscryptMethod(Class<?> enginesClass) throws NoSuchMethodException {
174         return enginesClass.getMethod("isConscrypt", SSLEngine.class);
175     }
