27  import javax.net.ssl.SSLEngine;

30  import io.netty.util.internal.PlatformDependent;
31  import org.eclipse.jetty.alpn.ALPN;

33  abstract class JettyAlpnSslEngine extends JdkSslEngine {
34      private static final boolean available = initAvailable();
35  
36      static boolean isAvailable() {
37          return available;
38      }
39  
40      private static boolean initAvailable() {
41          // TODO: Add support for ALPN when using Java9 and still be able to configure it the Netty way.
42          if (PlatformDependent.javaVersion() <= 8) {
43              try {
44                  // Always use bootstrap class loader.
45                  Class.forName("sun.security.ssl.ALPNExtension", true, null);
46                  return true;
47              } catch (Throwable ignore) {
48                  // alpn-boot was not loaded.
49              }
50          }
51          return false;
52      }
53  
54      static JettyAlpnSslEngine newClientEngine(SSLEngine engine,
55              JdkApplicationProtocolNegotiator applicationNegotiator) {
56          return new ClientEngine(engine, applicationNegotiator);
57      }
58  
59      static JettyAlpnSslEngine newServerEngine(SSLEngine engine,
60              JdkApplicationProtocolNegotiator applicationNegotiator) {
61          return new ServerEngine(engine, applicationNegotiator);
62      }
63  
64      private JettyAlpnSslEngine(SSLEngine engine) {
65          super(engine);
66      }
67  
68      private static final class ClientEngine extends JettyAlpnSslEngine {
69          ClientEngine(SSLEngine engine, final JdkApplicationProtocolNegotiator applicationNegotiator) {
70              super(engine);
71              checkNotNull(applicationNegotiator, "applicationNegotiator");
72              final ProtocolSelectionListener protocolListener = checkNotNull(applicationNegotiator
73                              .protocolListenerFactory().newListener(this, applicationNegotiator.protocols()),
74                      "protocolListener");
75              ALPN.put(engine, new ALPN.ClientProvider() {
76                  @Override
77                  public List<String> protocols() {
78                      return applicationNegotiator.protocols();
79                  }
80  
81                  @Override
82                  public void selected(String protocol) throws SSLException {
83                      try {
84                          protocolListener.selected(protocol);
85                      } catch (Throwable t) {
86                          throw toSSLHandshakeException(t);
87                      }
88                  }
89  
90                  @Override
91                  public void unsupported() {
92                      protocolListener.unsupported();
93                  }
94              });
95          }
96  
97          @Override
98          public void closeInbound() throws SSLException {
99              try {
100                 ALPN.remove(getWrappedEngine());
101             } finally {
102                 super.closeInbound();
103             }
104         }
105 
106         @Override
107         public void closeOutbound() {
108             try {
109                 ALPN.remove(getWrappedEngine());
110             } finally {
111                 super.closeOutbound();
112             }
113         }
114     }
115 
116     private static final class ServerEngine extends JettyAlpnSslEngine {
117         ServerEngine(SSLEngine engine, final JdkApplicationProtocolNegotiator applicationNegotiator) {
118             super(engine);
119             checkNotNull(applicationNegotiator, "applicationNegotiator");
120             final ProtocolSelector protocolSelector = checkNotNull(applicationNegotiator.protocolSelectorFactory()
121                             .newSelector(this, new LinkedHashSet<String>(applicationNegotiator.protocols())),
122                     "protocolSelector");
123             ALPN.put(engine, new ALPN.ServerProvider() {
124                 @Override
125                 public String select(List<String> protocols) throws SSLException {
126                     try {
127                         return protocolSelector.select(protocols);
128                     } catch (Throwable t) {
129                         throw toSSLHandshakeException(t);
130                     }
131                 }
132 
133                 @Override
134                 public void unsupported() {
135                     protocolSelector.unsupported();
136                 }
137             });
138         }
139 
140         @Override
141         public void closeInbound() throws SSLException {
142             try {
143                 ALPN.remove(getWrappedEngine());
144             } finally {
145                 super.closeInbound();
146             }
147         }
148 
149         @Override
150         public void closeOutbound() {
151             try {
152                 ALPN.remove(getWrappedEngine());
153             } finally {
154                 super.closeOutbound();
155             }
156         }
157     }
