224                 }

229             final Set<String> availableJavaCipherSuites = new LinkedHashSet<String>(

348     private static boolean doesSupportOcsp() {
349         boolean supportsOcsp = false;
350         if (version() >= 0x10002000L) {
351             long sslCtx = -1;
352             try {
353                 sslCtx = SSLContext.make(SSL.SSL_PROTOCOL_TLSV1_2, SSL.SSL_MODE_SERVER);
354                 SSLContext.enableOcsp(sslCtx, false);
355                 supportsOcsp = true;
356             } catch (Exception ignore) {
357                 // ignore
358             } finally {
359                 if (sslCtx != -1) {
360                     SSLContext.free(sslCtx);
361                 }
362             }
363         }
364         return supportsOcsp;
365     }
366     private static boolean doesSupportProtocol(int protocol, int opt) {
367         if (opt == 0) {
368             // If the opt is 0 the protocol is not supported. This is for example the case with BoringSSL and SSLv2.
369             return false;
370         }
371         long sslCtx = -1;
372         try {
373             sslCtx = SSLContext.make(protocol, SSL.SSL_MODE_COMBINED);
374             return true;
375         } catch (Exception ignore) {
376             return false;
377         } finally {
378             if (sslCtx != -1) {
379                 SSLContext.free(sslCtx);
380             }
381         }
382     }

509     private static void loadTcNative() throws Exception {
510         String os = PlatformDependent.normalizedOs();
511         String arch = PlatformDependent.normalizedArch();
512 
513         Set<String> libNames = new LinkedHashSet<String>(4);
514         String staticLibName = "netty_tcnative";
515 
516         // First, try loading the platform-specific library. Platform-specific
517         // libraries will be available if using a tcnative uber jar.
518         libNames.add(staticLibName + "_" + os + '_' + arch);
519         if ("linux".equalsIgnoreCase(os)) {
520             // Fedora SSL lib so naming (libssl.so.10 vs libssl.so.1.0.0)..
521             libNames.add(staticLibName + "_" + os + '_' + arch + "_fedora");
522         }
523         libNames.add(staticLibName + "_" + arch);
524         libNames.add(staticLibName);
525 
526         NativeLibraryLoader.loadFirstAvailable(SSL.class.getClassLoader(),
527             libNames.toArray(new String[0]));
528     }
