101     /**
102      * This is the maximum overhead when encrypting plaintext as defined by
103      * <a href="https://www.ietf.org/rfc/rfc5246.txt">rfc5264</a>,
104      * <a href="https://www.ietf.org/rfc/rfc5289.txt">rfc5289</a> and openssl implementation itself.
105      *
106      * Please note that we use a padding of 16 here as openssl uses PKC#5 which uses 16 bytes while the spec itself
107      * allow up to 255 bytes. 16 bytes is the max for PKC#5 (which handles it the same way as PKC#7) as we use a block
108      * size of 16. See <a href="https://tools.ietf.org/html/rfc5652#section-6.3">rfc5652#section-6.3</a>.
109      *
110      * TLS Header (5) + 16 (IV) + 48 (MAC) + 1 (Padding_length field) + 15 (Padding) + 1 (ContentType) +
111      * 2 (ProtocolVersion) + 2 (Length)
112      *
113      * TODO: We may need to review this calculation once TLS 1.3 becomes available.
114      */
115     static final int MAX_TLS_RECORD_OVERHEAD_LENGTH = SSL_RECORD_HEADER_LENGTH + 16 + 48 + 1 + 15 + 1 + 2 + 2;

215     /**
216      * Create a new instance.
217      * @param context Reference count release responsibility is not transferred! The callee still owns this object.
218      * @param alloc The allocator to use.
219      * @param peerHost The peer host name.
220      * @param peerPort The peer port.
221      * @param leakDetection {@code true} to enable leak detection of this object.
222      */
223     ReferenceCountedOpenSslEngine(ReferenceCountedOpenSslContext context, ByteBufAllocator alloc, String peerHost,
224                                   int peerPort, boolean leakDetection) {
225         super(peerHost, peerPort);
226         OpenSsl.ensureAvailability();
227         leak = leakDetection ? leakDetector.track(this) : null;
228         this.alloc = checkNotNull(alloc, "alloc");
229         apn = (OpenSslApplicationProtocolNegotiator) context.applicationProtocolNegotiator();
230         session = new OpenSslSession(context.sessionContext());
231         clientMode = context.isClient();
232         engineMap = context.engineMap;
233         rejectRemoteInitiatedRenegotiation = context.getRejectRemoteInitiatedRenegotiation();
234         localCerts = context.keyCertChain;
235         keyMaterialManager = context.keyMaterialManager();
236         enableOcsp = context.enableOcsp;
237         ssl = SSL.newSSL(context.ctx, !context.isClient());
238         try {
239             networkBIO = SSL.bioNewByteBuffer(ssl, context.getBioNonApplicationBufferSize());
240 
241             // Set the client auth mode, this needs to be done via setClientAuth(...) method so we actually call the
242             // needed JNI methods.
243             setClientAuth(clientMode ? ClientAuth.NONE : context.clientAuth);
244 
245             if (context.protocols != null) {
246                 setEnabledProtocols(context.protocols);
247             }
248 
249             // Use SNI if peerHost was specified
250             // See https://github.com/netty/netty/issues/4746
251             if (clientMode && peerHost != null) {
252                 SSL.setTlsExtHostName(ssl, peerHost);
253             }
254 
255             if (enableOcsp) {
256                 SSL.enableOcsp(ssl);
257             }
258         } catch (Throwable cause) {
259             SSL.freeSSL(ssl);
260             PlatformDependent.throwException(cause);
261         }
262     }

365      */
366     public final synchronized void shutdown() {
367         if (DESTROYED_UPDATER.compareAndSet(this, 0, 1)) {
368             engineMap.remove(ssl);
369             SSL.freeSSL(ssl);
370             ssl = networkBIO = 0;
371 
372             isInboundDone = outboundClosed = true;
373         }

375         // On shutdown clear all errors
376         SSL.clearError();

379     /**
380      * Write plaintext data to the OpenSSL internal BIO
381      *
382      * Calling this function with src.remaining == 0 is undefined.
383      */
384     private int writePlaintextData(final ByteBuffer src, int len) {
385         final int pos = src.position();
386         final int limit = src.limit();
387         final int sslWrote;
388 
389         if (src.isDirect()) {
390             sslWrote = SSL.writeToSSL(ssl, Buffer.address(src) + pos, len);
391             if (sslWrote > 0) {
392                 src.position(pos + sslWrote);
393             }
394         } else {
395             ByteBuf buf = alloc.directBuffer(len);
396             try {
397                 src.limit(pos + len);
398 
399                 buf.setBytes(0, src);
400                 src.limit(limit);
401 
402                 sslWrote = SSL.writeToSSL(ssl, memoryAddress(buf), len);
403                 if (sslWrote > 0) {
404                     src.position(pos + sslWrote);
405                 } else {
406                     src.position(pos);
407                 }
408             } finally {
409                 buf.release();
410             }
411         }
412         return sslWrote;
413     }

415     /**
416      * Write encrypted data to the OpenSSL network BIO.
417      */
418     private ByteBuf writeEncryptedData(final ByteBuffer src, int len) {
419         final int pos = src.position();
420         if (src.isDirect()) {
421             SSL.bioSetByteBuffer(networkBIO, Buffer.address(src) + pos, len, false);
422         } else {
423             final ByteBuf buf = alloc.directBuffer(len);
424             try {
425                 final int limit = src.limit();
426                 src.limit(pos + len);
427                 buf.writeBytes(src);
428                 // Restore the original position and limit because we don't want to consume from `src`.
429                 src.position(pos);
430                 src.limit(limit);
431 
432                 SSL.bioSetByteBuffer(networkBIO, memoryAddress(buf), len, false);
433                 return buf;
434             } catch (Throwable cause) {
435                 buf.release();
436                 PlatformDependent.throwException(cause);
437             }
438         }
439         return null;
440     }

442     /**
443      * Read plaintext data from the OpenSSL internal BIO
444      */
445     private int readPlaintextData(final ByteBuffer dst) {
446         final int sslRead;
447         final int pos = dst.position();
448         if (dst.isDirect()) {
449             sslRead = SSL.readFromSSL(ssl, Buffer.address(dst) + pos, dst.limit() - pos);
450             if (sslRead > 0) {
451                 dst.position(pos + sslRead);
452             }
453         } else {
454             final int limit = dst.limit();
455             final int len = min(MAX_ENCRYPTED_PACKET_LENGTH, limit - pos);
456             final ByteBuf buf = alloc.directBuffer(len);
457             try {
458                 sslRead = SSL.readFromSSL(ssl, memoryAddress(buf), len);
459                 if (sslRead > 0) {
460                     dst.limit(pos + sslRead);
461                     buf.getBytes(buf.readerIndex(), dst);
462                     dst.limit(limit);
463                 }
464             } finally {
465                 buf.release();
466             }
467         }
468 
469         return sslRead;
470     }

472     @Override
473     public final SSLEngineResult wrap(
474             final ByteBuffer[] srcs, int offset, final int length, final ByteBuffer dst) throws SSLException {
475         // Throw required runtime exceptions
476         if (srcs == null) {
477             throw new IllegalArgumentException("srcs is null");
478         }
479         if (dst == null) {
480             throw new IllegalArgumentException("dst is null");
481         }
482 
483         if (offset >= srcs.length || offset + length > srcs.length) {
484             throw new IndexOutOfBoundsException(
485                     "offset: " + offset + ", length: " + length +
486                             " (expected: offset <= offset + length <= srcs.length (" + srcs.length + "))");
487         }
488 
489         if (dst.isReadOnly()) {
490             throw new ReadOnlyBufferException();
491         }
492 
493         synchronized (this) {
494             if (isOutboundDone()) {
495                 // All drained in the outbound buffer
496                 return isInboundDone() || isDestroyed() ? CLOSED_NOT_HANDSHAKING : NEED_UNWRAP_CLOSED;
497             }
498 
499             int bytesProduced = 0;
500             ByteBuf bioReadCopyBuf = null;
501             try {
502                 // Setup the BIO buffer so that we directly write the encryption results into dst.
503                 if (dst.isDirect()) {
504                     SSL.bioSetByteBuffer(networkBIO, Buffer.address(dst) + dst.position(), dst.remaining(),
505                             true);
506                 } else {
507                     bioReadCopyBuf = alloc.directBuffer(dst.remaining());
508                     SSL.bioSetByteBuffer(networkBIO, memoryAddress(bioReadCopyBuf), bioReadCopyBuf.writableBytes(),
509                             true);
510                 }
511 
512                 int bioLengthBefore = SSL.bioLengthByteBuffer(networkBIO);
513 
514                 // Explicit use outboundClosed as we want to drain any bytes that are still present.
515                 if (outboundClosed) {
516                     // There is something left to drain.
517                     // See https://github.com/netty/netty/issues/6260
518                     bytesProduced = SSL.bioFlushByteBuffer(networkBIO);
519                     if (bytesProduced <= 0) {
520                         return newResultMayFinishHandshake(NOT_HANDSHAKING, 0, 0);
521                     }
522                     // It is possible when the outbound was closed there was not enough room in the non-application
523                     // buffers to hold the close_notify. We should keep trying to close until we consume all the data
524                     // OpenSSL can give us.
525                     if (!doSSLShutdown()) {
526                         return newResultMayFinishHandshake(NOT_HANDSHAKING, 0, bytesProduced);
527                     }
528                     bytesProduced = bioLengthBefore - SSL.bioLengthByteBuffer(networkBIO);
529                     return newResultMayFinishHandshake(NEED_WRAP, 0, bytesProduced);
530                 }
531 
532                 // Flush any data that may be implicitly generated by OpenSSL (handshake, close, etc..).
533                 SSLEngineResult.HandshakeStatus status = NOT_HANDSHAKING;
534                 // Prepare OpenSSL to work in server mode and receive handshake
535                 if (handshakeState != HandshakeState.FINISHED) {
536                     if (handshakeState != HandshakeState.STARTED_EXPLICITLY) {
537                         // Update accepted so we know we triggered the handshake via wrap
538                         handshakeState = HandshakeState.STARTED_IMPLICITLY;
539                     }
540 
541                     // Flush any data that may have been written implicitly during the handshake by OpenSSL.
542                     bytesProduced = SSL.bioFlushByteBuffer(networkBIO);
543 
544                     if (bytesProduced > 0 && handshakeException != null) {
545                         // TODO(scott): It is possible that when the handshake failed there was not enough room in the
546                         // non-application buffers to hold the alert. We should get all the data before progressing on.
547                         // However I'm not aware of a way to do this with the OpenSSL APIs.
548                         // See https://github.com/netty/netty/issues/6385.
549 
550                         // We produced / consumed some data during the handshake, signal back to the caller.
551                         // If there is a handshake exception and we have produced data, we should send the data before
552                         // we allow handshake() to throw the handshake exception.
553                         return newResult(NEED_WRAP, 0, bytesProduced);
554                     }
555 
556                     status = handshake();
557 
558                     if (renegotiationPending && status == FINISHED) {
559                         // If renegotiationPending is true that means when we attempted to start renegotiation
560                         // the BIO buffer didn't have enough space to hold the HelloRequest which prompts the
561                         // client to initiate a renegotiation. At this point the HelloRequest has been written
562                         // so we can actually start the handshake process.
563                         renegotiationPending = false;
564                         SSL.setState(ssl, SSL.SSL_ST_ACCEPT);
565                         handshakeState = HandshakeState.STARTED_EXPLICITLY;
566                         status = handshake();
567                     }
568 
569                     // Handshake may have generated more data, for example if the internal SSL buffer is small
570                     // we may have freed up space by flushing above.
571                     bytesProduced = bioLengthBefore - SSL.bioLengthByteBuffer(networkBIO);
572 
573                     if (bytesProduced > 0) {
574                         // It's important we call this before wrapStatus() as wrapStatus() may shutdown the engine.
575                         return newResult(mayFinishHandshake(status != FINISHED ?
576                                          getHandshakeStatus(SSL.bioLengthNonApplication(networkBIO)) : FINISHED),
577                                          0, bytesProduced);
578                     }
579 
580                     if (status == NEED_UNWRAP) {
581                         // Signal if the outbound is done or not.
582                         return isOutboundDone() ? NEED_UNWRAP_CLOSED : NEED_UNWRAP_OK;
583                     }
584 
585                     // Explicit use outboundClosed and not outboundClosed() as we want to drain any bytes that are
586                     // still present.
587                     if (outboundClosed) {
588                         bytesProduced = SSL.bioFlushByteBuffer(networkBIO);
589                         return newResultMayFinishHandshake(status, 0, bytesProduced);
590                     }
591                 }
592 
593                 int srcsLen = 0;
594                 final int endOffset = offset + length;
595                 for (int i = offset; i < endOffset; ++i) {
596                     final ByteBuffer src = srcs[i];
597                     if (src == null) {
598                         throw new IllegalArgumentException("srcs[" + i + "] is null");
599                     }
600                     if (srcsLen == MAX_PLAINTEXT_LENGTH) {
601                         continue;
602                     }
603 
604                     srcsLen += src.remaining();
605                     if (srcsLen > MAX_PLAINTEXT_LENGTH || srcsLen < 0) {
606                         // If srcLen > MAX_PLAINTEXT_LENGTH or secLen < 0 just set it to MAX_PLAINTEXT_LENGTH.
607                         // This also help us to guard against overflow.
608                         // We not break out here as we still need to check for null entries in srcs[].
609                         srcsLen = MAX_PLAINTEXT_LENGTH;
610                     }
611                 }
612 
613                 // we will only produce a single TLS packet, and we don't aggregate src buffers,
614                 // so we always fix the number of buffers to 1 when checking if the dst buffer is large enough.
615                 if (dst.remaining() < calculateOutNetBufSize(srcsLen, 1)) {
616                     return new SSLEngineResult(BUFFER_OVERFLOW, getHandshakeStatus(), 0, 0);
617                 }
618 
619                 // There was no pending data in the network BIO -- encrypt any application data
620                 int bytesConsumed = 0;
621                 // Flush any data that may have been written implicitly by OpenSSL in case a shutdown/alert occurs.
622                 bytesProduced = SSL.bioFlushByteBuffer(networkBIO);
623                 for (; offset < endOffset; ++offset) {
624                     final ByteBuffer src = srcs[offset];
625                     final int remaining = src.remaining();
626                     if (remaining == 0) {
627                         continue;
628                     }
629 
630                     // Write plaintext application data to the SSL engine
631                     int bytesWritten = writePlaintextData(src, min(remaining, MAX_PLAINTEXT_LENGTH - bytesConsumed));
632 
633                     if (bytesWritten > 0) {
634                         bytesConsumed += bytesWritten;
635 
636                         // Determine how much encrypted data was generated:
637                         final int pendingNow = SSL.bioLengthByteBuffer(networkBIO);
638                         bytesProduced += bioLengthBefore - pendingNow;
639                         bioLengthBefore = pendingNow;
640 
641                         return newResultMayFinishHandshake(status, bytesConsumed, bytesProduced);
642                     } else {
643                         int sslError = SSL.getError(ssl, bytesWritten);
644                         if (sslError == SSL.SSL_ERROR_ZERO_RETURN) {
645                             // This means the connection was shutdown correctly, close inbound and outbound
646                             if (!receivedShutdown) {
647                                 closeAll();
648 
649                                 bytesProduced += bioLengthBefore - SSL.bioLengthByteBuffer(networkBIO);
650 
651                                 SSLEngineResult.HandshakeStatus hs = mayFinishHandshake(
652                                         status != FINISHED ? getHandshakeStatus(SSL.bioLengthNonApplication(networkBIO))
653                                                 : FINISHED);
654                                 return newResult(hs, bytesConsumed, bytesProduced);
655                             }
656 
657                             return newResult(NOT_HANDSHAKING, bytesConsumed, bytesProduced);
658                         } else if (sslError == SSL.SSL_ERROR_WANT_READ) {
659                             // If there is no pending data to read from BIO we should go back to event loop and try
660                             // to read more data [1]. It is also possible that event loop will detect the socket has
661                             // been closed. [1] https://www.openssl.org/docs/manmaster/ssl/SSL_write.html
662                             return newResult(NEED_UNWRAP, bytesConsumed, bytesProduced);
663                         } else if (sslError == SSL.SSL_ERROR_WANT_WRITE) {
664                             // SSL_ERROR_WANT_WRITE typically means that the underlying transport is not writable
665                             // and we should set the "want write" flag on the selector and try again when the
666                             // underlying transport is writable [1]. However we are not directly writing to the
667                             // underlying transport and instead writing to a BIO buffer. The OpenSsl documentation
668                             // says we should do the following [1]:
669                             //
670                             // "When using a buffering BIO, like a BIO pair, data must be written into or retrieved
671                             // out of the BIO before being able to continue."
672                             //
673                             // So we attempt to drain the BIO buffer below, but if there is no data this condition
674                             // is undefined and we assume their is a fatal error with the openssl engine and close.
675                             // [1] https://www.openssl.org/docs/manmaster/ssl/SSL_write.html
676                             return newResult(NEED_WRAP, bytesConsumed, bytesProduced);
677                         } else {
678                             // Everything else is considered as error
679                             throw shutdownWithError("SSL_write");
680                         }
681                     }
682                 }
683                 return newResultMayFinishHandshake(status, bytesConsumed, bytesProduced);
684             } finally {
685                 SSL.bioClearByteBuffer(networkBIO);
686                 if (bioReadCopyBuf == null) {
687                     dst.position(dst.position() + bytesProduced);
688                 } else {
689                     assert bioReadCopyBuf.readableBytes() <= dst.remaining() : "The destination buffer " + dst +
690                             " didn't have enough remaining space to hold the encrypted content in " + bioReadCopyBuf;
691                     dst.put(bioReadCopyBuf.internalNioBuffer(bioReadCopyBuf.readerIndex(), bytesProduced));
692                     bioReadCopyBuf.release();
693                 }
694             }
695         }
696     }

702     private SSLEngineResult newResult(SSLEngineResult.Status status, SSLEngineResult.HandshakeStatus hs,
703                                       int bytesConsumed, int bytesProduced) {
704         // If isOutboundDone, then the data from the network BIO
705         // was the close_notify message and all was consumed we are not required to wait
706         // for the receipt the peer's close_notify message -- shutdown.
707         if (isOutboundDone()) {
708             if (isInboundDone()) {
709                 // If the inbound was done as well, we need to ensure we return NOT_HANDSHAKING to signal we are done.
710                 hs = NOT_HANDSHAKING;
711 
712                 // As the inbound and the outbound is done we can shutdown the engine now.
713                 shutdown();
714             }
715             return new SSLEngineResult(CLOSED, hs, bytesConsumed, bytesProduced);
716         }
717         return new SSLEngineResult(status, hs, bytesConsumed, bytesProduced);
718     }

754     public final SSLEngineResult unwrap(
755             final ByteBuffer[] srcs, int srcsOffset, final int srcsLength,
756             final ByteBuffer[] dsts, int dstsOffset, final int dstsLength) throws SSLException {
757 
758         // Throw required runtime exceptions
759         if (srcs == null) {
760             throw new NullPointerException("srcs");
761         }
762         if (srcsOffset >= srcs.length
763                 || srcsOffset + srcsLength > srcs.length) {
764             throw new IndexOutOfBoundsException(
765                     "offset: " + srcsOffset + ", length: " + srcsLength +
766                             " (expected: offset <= offset + length <= srcs.length (" + srcs.length + "))");
767         }
768         if (dsts == null) {
769             throw new IllegalArgumentException("dsts is null");
770         }
771         if (dstsOffset >= dsts.length || dstsOffset + dstsLength > dsts.length) {
772             throw new IndexOutOfBoundsException(
773                     "offset: " + dstsOffset + ", length: " + dstsLength +
774                             " (expected: offset <= offset + length <= dsts.length (" + dsts.length + "))");
775         }
776         long capacity = 0;
777         final int dstsEndOffset = dstsOffset + dstsLength;
778         for (int i = dstsOffset; i < dstsEndOffset; i ++) {
779             ByteBuffer dst = dsts[i];
780             if (dst == null) {
781                 throw new IllegalArgumentException("dsts[" + i + "] is null");
782             }
783             if (dst.isReadOnly()) {
784                 throw new ReadOnlyBufferException();
785             }
786             capacity += dst.remaining();
787         }
788 
789         final int srcsEndOffset = srcsOffset + srcsLength;
790         long len = 0;
791         for (int i = srcsOffset; i < srcsEndOffset; i++) {
792             ByteBuffer src = srcs[i];
793             if (src == null) {
794                 throw new IllegalArgumentException("srcs[" + i + "] is null");
795             }
796             len += src.remaining();
797         }
798 
799         synchronized (this) {
800             if (isInboundDone()) {
801                 return isOutboundDone() || isDestroyed() ? CLOSED_NOT_HANDSHAKING : NEED_WRAP_CLOSED;
802             }
803 
804             SSLEngineResult.HandshakeStatus status = NOT_HANDSHAKING;
805             // Prepare OpenSSL to work in server mode and receive handshake
806             if (handshakeState != HandshakeState.FINISHED) {
807                 if (handshakeState != HandshakeState.STARTED_EXPLICITLY) {
808                     // Update accepted so we know we triggered the handshake via wrap
809                     handshakeState = HandshakeState.STARTED_IMPLICITLY;
810                 }
811 
812                 status = handshake();
813                 if (status == NEED_WRAP) {
814                     return NEED_WRAP_OK;
815                 }
816                 // Check if the inbound is considered to be closed if so let us try to wrap again.
817                 if (isInboundDone) {
818                     return NEED_WRAP_CLOSED;
819                 }
820             }
821 
822             if (len < SSL_RECORD_HEADER_LENGTH) {
823                 return newResultMayFinishHandshake(BUFFER_UNDERFLOW, status, 0, 0);
824             }
825 
826             int packetLength = SslUtils.getEncryptedPacketLength(srcs, srcsOffset);
827 
828             if (packetLength == SslUtils.NOT_ENCRYPTED) {
829                 throw new NotSslRecordException("not an SSL/TLS record");
830             }
831 
832             if (packetLength - SSL_RECORD_HEADER_LENGTH > capacity) {
833                 // No enough space in the destination buffer so signal the caller
834                 // that the buffer needs to be increased.
835                 return newResultMayFinishHandshake(BUFFER_OVERFLOW, status, 0, 0);
836             }
837 
838             if (len < packetLength) {
839                 // We either have no enough data to read the packet length at all or not enough for reading
840                 // the whole packet.
841                 return newResultMayFinishHandshake(BUFFER_UNDERFLOW, status, 0, 0);
842             }
843 
844             // This must always be the case when we reached here as if not we returned BUFFER_UNDERFLOW.
845             assert srcsOffset < srcsEndOffset;
846 
847             // This must always be the case if we reached here.
848             assert capacity > 0;
849 
850             // Number of produced bytes
851             int bytesProduced = 0;
852             int bytesConsumed = 0;
853             try {
854                 for (; srcsOffset < srcsEndOffset; ++srcsOffset) {
855                     ByteBuffer src = srcs[srcsOffset];
856                     int remaining = src.remaining();
857                     if (remaining == 0) {
858                         // We must skip empty buffers as BIO_write will return 0 if asked to write something
859                         // with length 0.
860                         continue;
861                     }
862                     // Write more encrypted data into the BIO. Ensure we only read one packet at a time as
863                     // stated in the SSLEngine javadocs.
864                     int pendingEncryptedBytes = min(packetLength, remaining);
865                     ByteBuf bioWriteCopyBuf = writeEncryptedData(src, pendingEncryptedBytes);
866                     try {
867                         readLoop:
868                         for (; dstsOffset < dstsEndOffset; ++dstsOffset) {
869                             ByteBuffer dst = dsts[dstsOffset];
870                             if (!dst.hasRemaining()) {
871                                 // No space left in the destination buffer, skip it.
872                                 continue;
873                             }
874 
875                             int bytesRead = readPlaintextData(dst);
876                             // We are directly using the ByteBuffer memory for the write, and so we only know what
877                             // has been consumed after we let SSL decrypt the data. At this point we should update
878                             // the number of bytes consumed, update the ByteBuffer position, and release temp
879                             // ByteBuf.
880                             int localBytesConsumed = pendingEncryptedBytes - SSL.bioLengthByteBuffer(networkBIO);
881                             bytesConsumed += localBytesConsumed;
882                             packetLength -= localBytesConsumed;
883                             pendingEncryptedBytes -= localBytesConsumed;
884                             src.position(src.position() + localBytesConsumed);
885 
886                             if (bytesRead > 0) {
887                                 bytesProduced += bytesRead;
888 
889                                 if (!dst.hasRemaining()) {
890                                     // Move to the next dst buffer as this one is full.
891                                     continue;
892                                 }
893                                 if (packetLength == 0) {
894                                     // We read everything return now.
895                                     return newResultMayFinishHandshake(isInboundDone() ? CLOSED : OK, status,
896                                                                        bytesConsumed, bytesProduced);
897                                 }
898                                 // try to write again to the BIO. stop reading from it by break out of the readLoop.
899                                 break;
900                             } else {
901                                 int sslError = SSL.getError(ssl, bytesRead);
902                                 if (sslError == SSL.SSL_ERROR_WANT_READ || sslError == SSL.SSL_ERROR_WANT_WRITE) {
903                                     // break to the outer loop as we want to read more data which means we need to
904                                     // write more to the BIO.
905                                     break readLoop;
906                                 } else if (sslError == SSL.SSL_ERROR_ZERO_RETURN) {
907                                     // This means the connection was shutdown correctly, close inbound and outbound
908                                     if (!receivedShutdown) {
909                                         closeAll();
910                                     }
911                                     return newResultMayFinishHandshake(isInboundDone() ? CLOSED : OK, status,
912                                                                        bytesConsumed, bytesProduced);
913                                 } else {
914                                     return sslReadErrorResult(SSL.getLastErrorNumber(), bytesConsumed,
915                                                               bytesProduced);
916                                 }
917                             }
918                         }
919 
920                         // Either we have no more dst buffers to put the data, or no more data to generate; we are done.
921                         if (dstsOffset >= dstsEndOffset || packetLength == 0) {
922                             break;
923                         }
924                     } finally {
925                         if (bioWriteCopyBuf != null) {
926                             bioWriteCopyBuf.release();
927                         }
928                     }
929                 }
930             } finally {
931                 SSL.bioClearByteBuffer(networkBIO);
932                 rejectRemoteInitiatedRenegotiation();
933             }
934 
935             // Check to see if we received a close_notify message from the peer.
936             if (!receivedShutdown && (SSL.getShutdown(ssl) & SSL.SSL_RECEIVED_SHUTDOWN) == SSL.SSL_RECEIVED_SHUTDOWN) {
937                 closeAll();
938             }
939 
940             return newResultMayFinishHandshake(isInboundDone() ? CLOSED : OK, status, bytesConsumed, bytesProduced);
941         }
942     }

944     private SSLEngineResult sslReadErrorResult(int err, int bytesConsumed, int bytesProduced) throws SSLException {
945         String errStr = SSL.getErrorString(err);
946 
947         // Check if we have a pending handshakeException and if so see if we need to consume all pending data from the
948         // BIO first or can just shutdown and throw it now.
949         // This is needed so we ensure close_notify etc is correctly send to the remote peer.
950         // See https://github.com/netty/netty/issues/3900
951         if (SSL.bioLengthNonApplication(networkBIO) > 0) {
952             if (handshakeException == null && handshakeState != HandshakeState.FINISHED) {
953                 // we seems to have data left that needs to be transfered and so the user needs
954                 // call wrap(...). Store the error so we can pick it up later.
955                 handshakeException = new SSLHandshakeException(errStr);
956             }
957             return new SSLEngineResult(OK, NEED_WRAP, bytesConsumed, bytesProduced);
958         }
959         throw shutdownWithError("SSL_read", errStr);
960     }

1089     /**
1090      * Attempt to call {@link SSL#shutdownSSL(long)}.
1091      * @return {@code false} if the call to {@link SSL#shutdownSSL(long)} was not attempted or returned an error.
1092      */
1093     private boolean doSSLShutdown() {
1094         if (SSL.isInInit(ssl) != 0) {
1095             // Only try to call SSL_shutdown if we are not in the init state anymore.
1096             // Otherwise we will see 'error:140E0197:SSL routines:SSL_shutdown:shutdown while in init' in our logs.
1097             //
1098             // See also http://hg.nginx.org/nginx/rev/062c189fee20
1099             return false;
1100         }
1101         int err = SSL.shutdownSSL(ssl);
1102         if (err < 0) {
1103             int sslErr = SSL.getError(ssl, err);
1104             if (sslErr == SSL.SSL_ERROR_SYSCALL || sslErr == SSL.SSL_ERROR_SSL) {
1105                 if (logger.isDebugEnabled()) {
1106                     logger.debug("SSL_shutdown failed: OpenSSL error: {}", SSL.getLastError());
1107                 }
1108                 // There was an internal error -- shutdown
1109                 shutdown();
1110                 return false;
1111             }
1112             SSL.clearError();
1113         }
1114         return true;
1115     }

1303     @Override
1304     public final synchronized void beginHandshake() throws SSLException {
1305         switch (handshakeState) {
1306             case STARTED_IMPLICITLY:
1307                 checkEngineClosed(BEGIN_HANDSHAKE_ENGINE_CLOSED);
1308 
1309                 // A user did not start handshake by calling this method by him/herself,
1310                 // but handshake has been started already by wrap() or unwrap() implicitly.
1311                 // Because it's the user's first time to call this method, it is unfair to
1312                 // raise an exception.  From the user's standpoint, he or she never asked
1313                 // for renegotiation.
1314 
1315                 handshakeState = HandshakeState.STARTED_EXPLICITLY; // Next time this method is invoked by the user,
1316                 // we should raise an exception.
1317                 break;
1318             case STARTED_EXPLICITLY:
1319                 // Nothing to do as the handshake is not done yet.
1320                 break;
1321             case FINISHED:
1322                 if (clientMode) {
1323                     // Only supported for server mode at the moment.
1324                     throw RENEGOTIATION_UNSUPPORTED;
1325                 }
1326                 // For renegotiate on the server side we need to issue the following command sequence with openssl:
1327                 //
1328                 // SSL_renegotiate(ssl)
1329                 // SSL_do_handshake(ssl)
1330                 // ssl->state = SSL_ST_ACCEPT
1331                 // SSL_do_handshake(ssl)
1332                 //
1333                 // Because of this we fall-through to call handshake() after setting the state, as this will also take
1334                 // care of updating the internal OpenSslSession object.
1335                 //
1336                 // See also:
1337                 // https://github.com/apache/httpd/blob/2.4.16/modules/ssl/ssl_engine_kernel.c#L812
1338                 // http://h71000.www7.hp.com/doc/83final/ba554_90007/ch04s03.html
1339                 int status;
1340                 if ((status = SSL.renegotiate(ssl)) != 1 || (status = SSL.doHandshake(ssl)) != 1) {
1341                     int err = SSL.getError(ssl, status);
1342                     if (err == SSL.SSL_ERROR_WANT_READ || err == SSL.SSL_ERROR_WANT_WRITE) {
1343                         // If the internal SSL buffer is small it is possible that doHandshake may "fail" because
1344                         // there is not enough room to write, so we should wait until the renegotiation has been.
1345                         renegotiationPending = true;
1346                         handshakeState = HandshakeState.STARTED_EXPLICITLY;
1347                         lastAccessed = System.currentTimeMillis();
1348                         return;
1349                     } else {
1350                         throw shutdownWithError("renegotiation failed");
1351                     }
1352                 }
1353 
1354                 SSL.setState(ssl, SSL.SSL_ST_ACCEPT);
1355 
1356                 lastAccessed = System.currentTimeMillis();
1357 
1358                 // fall-through
1359             case NOT_STARTED:
1360                 handshakeState = HandshakeState.STARTED_EXPLICITLY;
1361                 handshake();
1362                 break;
1363             default:
1364                 throw new Error();
1365         }
1366     }

1387     private SSLEngineResult.HandshakeStatus handshake() throws SSLException {
1388         if (handshakeState == HandshakeState.FINISHED) {
1389             return FINISHED;
1390         }
1391         checkEngineClosed(HANDSHAKE_ENGINE_CLOSED);
1392 
1393         // Check if we have a pending handshakeException and if so see if we need to consume all pending data from the
1394         // BIO first or can just shutdown and throw it now.
1395         // This is needed so we ensure close_notify etc is correctly send to the remote peer.
1396         // See https://github.com/netty/netty/issues/3900
1397         SSLHandshakeException exception = handshakeException;
1398         if (exception != null) {
1399             if (SSL.bioLengthNonApplication(networkBIO) > 0) {
1400                 // There is something pending, we need to consume it first via a WRAP so we don't loose anything.
1401                 return NEED_WRAP;
1402             }
1403             // No more data left to send to the remote peer, so null out the exception field, shutdown and throw
1404             // the exception.
1405             handshakeException = null;
1406             shutdown();
1407             throw exception;
1408         }
1409 
1410         // Adding the OpenSslEngine to the OpenSslEngineMap so it can be used in the AbstractCertificateVerifier.
1411         engineMap.add(this);
1412         if (lastAccessed == -1) {
1413             lastAccessed = System.currentTimeMillis();
1414         }
1415 
1416         if (!certificateSet && keyMaterialManager != null) {
1417             certificateSet = true;
1418             keyMaterialManager.setKeyMaterial(this);
1419         }
1420 
1421         int code = SSL.doHandshake(ssl);
1422         if (code <= 0) {
1423             // Check if we have a pending exception that was created during the handshake and if so throw it after
1424             // shutdown the connection.
1425             if (handshakeException != null) {
1426                 exception = handshakeException;
1427                 handshakeException = null;
1428                 shutdown();
1429                 throw exception;
1430             }
1431 
1432             int sslError = SSL.getError(ssl, code);
1433             if (sslError == SSL.SSL_ERROR_WANT_READ || sslError == SSL.SSL_ERROR_WANT_WRITE) {
1434                 return pendingStatus(SSL.bioLengthNonApplication(networkBIO));
1435             } else {
1436                 // Everything else is considered as error
1437                 throw shutdownWithError("SSL_do_handshake");
1438             }
1439         }
1440         // if SSL_do_handshake returns > 0 or sslError == SSL.SSL_ERROR_NAME it means the handshake was finished.
1441         session.handshakeFinished();
1442         engineMap.remove(ssl);
1443         return FINISHED;
1444     }

1814         /**
1815          * Init peer certificates that can be obtained via {@link #getPeerCertificateChain()}
1816          * and {@link #getPeerCertificates()}.
1817          */
1818         private void initPeerCerts() {
1819             // Return the full chain from the JNI layer.
1820             byte[][] chain = SSL.getPeerCertChain(ssl);
1821             if (clientMode) {
1822                 if (isEmpty(chain)) {
1823                     peerCerts = EMPTY_CERTIFICATES;
1824                     x509PeerCerts = EMPTY_JAVAX_X509_CERTIFICATES;
1825                 } else {
1826                     peerCerts = new Certificate[chain.length];
1827                     x509PeerCerts = new X509Certificate[chain.length];
1828                     initCerts(chain, 0);
1829                 }
1830             } else {
1831                 // if used on the server side SSL_get_peer_cert_chain(...) will not include the remote peer
1832                 // certificate. We use SSL_get_peer_certificate to get it in this case and add it to our
1833                 // array later.
1834                 //
1835                 // See https://www.openssl.org/docs/ssl/SSL_get_peer_cert_chain.html
1836                 byte[] clientCert = SSL.getPeerCertificate(ssl);
1837                 if (isEmpty(clientCert)) {
1838                     peerCerts = EMPTY_CERTIFICATES;
1839                     x509PeerCerts = EMPTY_JAVAX_X509_CERTIFICATES;
1840                 } else {
1841                     if (isEmpty(chain)) {
1842                         peerCerts = new Certificate[] {new OpenSslX509Certificate(clientCert)};
1843                         x509PeerCerts = new X509Certificate[] {new OpenSslJavaxX509Certificate(clientCert)};
1844                     } else {
1845                         peerCerts = new Certificate[chain.length + 1];
1846                         x509PeerCerts = new X509Certificate[chain.length + 1];
1847                         peerCerts[0] = new OpenSslX509Certificate(clientCert);
1848                         x509PeerCerts[0] = new OpenSslJavaxX509Certificate(clientCert);
1849                         initCerts(chain, 1);
1850                     }
1851                 }
1852             }
1853         }

1904         private String selectApplicationProtocol(List<String> protocols,
1905                                                  ApplicationProtocolConfig.SelectedListenerFailureBehavior behavior,
1906                                                  String applicationProtocol) throws SSLException {
1907             if (behavior == ApplicationProtocolConfig.SelectedListenerFailureBehavior.ACCEPT) {
1908                 return applicationProtocol;
1909             } else {
1910                 int size = protocols.size();
1911                 assert size > 0;
1912                 if (protocols.contains(applicationProtocol)) {
1913                     return applicationProtocol;
1914                 } else {
1915                     if (behavior == ApplicationProtocolConfig.SelectedListenerFailureBehavior.CHOOSE_MY_LAST_PROTOCOL) {
1916                         return protocols.get(size - 1);
1917                     } else {
1918                         throw new SSLException("unknown protocol " + applicationProtocol);
1919                     }
1920                 }
1921             }
