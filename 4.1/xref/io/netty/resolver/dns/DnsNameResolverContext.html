106     void resolve(Promise<T> promise) {
107         boolean directSearch = parent.searchDomains().length == 0 || StringUtil.endsWith(hostname, '.');
108         if (directSearch) {
109             internalResolve(promise);
110         } else {
111             final Promise<T> original = promise;
112             promise = parent.executor().newPromise();
113             promise.addListener(new FutureListener<T>() {
114                 int count;
115                 @Override
116                 public void operationComplete(Future<T> future) throws Exception {
117                     if (future.isSuccess()) {
118                         original.trySuccess(future.getNow());
119                     } else if (count < parent.searchDomains().length) {
120                         String searchDomain = parent.searchDomains()[count++];
121                         Promise<T> nextPromise = parent.executor().newPromise();
122                         String nextHostname = hostname + '.' + searchDomain;
123                         DnsNameResolverContext<T> nextContext = newResolverContext(parent,
124                             nextHostname, additionals, resolveCache, nameServerAddrs);
125                         nextContext.pristineHostname = hostname;
126                         nextContext.internalResolve(nextPromise);
127                         nextPromise.addListener(this);
128                     } else {
129                         original.tryFailure(future.cause());
130                     }
131                 }
132             });
133             if (parent.ndots() == 0) {
134                 internalResolve(promise);
135             } else {
136                 int dots = 0;
137                 for (int idx = hostname.length() - 1; idx >= 0; idx--) {
138                     if (hostname.charAt(idx) == '.' && ++dots >= parent.ndots()) {
139                         internalResolve(promise);
140                         return;
141                     }
142                 }
143                 promise.tryFailure(new UnknownHostException(hostname));
144             }
145         }
146     }
147 
148     private void internalResolve(Promise<T> promise) {
149         DnsServerAddressStream nameServerAddressStream = getNameServers(hostname);
150 
151         for (DnsRecordType type: parent.resolveRecordTypes()) {
152             if (!query(hostname, type, nameServerAddressStream, promise)) {
153                 return;
154             }
155         }
156     }

158     /**
159      * Add an authoritative nameserver to the cache if its not a root server.
160      */
161     private void addNameServerToCache(
162             AuthoritativeNameServer name, InetAddress resolved, long ttl) {
163         if (!name.isRootServer()) {
164             // Cache NS record if not for a root server as we should never cache for root servers.
165             parent.authoritativeDnsServerCache().cache(name.domainName(),
166                     additionals, resolved, ttl, parent.ch.eventLoop());
167         }
168     }

170     /**
171      * Returns the {@link DnsServerAddressStream} that was cached for the given hostname or {@code null} if non
172      *  could be found.
173      */
174     private DnsServerAddressStream getNameServersFromCache(String hostname) {
175         int len = hostname.length();
176 
177         if (len == 0) {
178             // We never cache for root servers.
179             return null;
180         }
181 
182         // We always store in the cache with a trailing '.'.
183         if (hostname.charAt(len - 1) != '.') {
184             hostname += ".";
185         }
186 
187         int idx = hostname.indexOf('.');
188         if (idx == hostname.length() - 1) {
189             // We are not interested in handling '.' as we should never serve the root servers from cache.
190             return null;
191         }
192 
193         // We start from the closed match and then move down.
194         for (;;) {
195             // Skip '.' as well.
196             hostname = hostname.substring(idx + 1);
197 
198             int idx2 = hostname.indexOf('.');
199             if (idx2 <= 0 || idx2 == hostname.length() - 1) {
200                 // We are not interested in handling '.TLD.' as we should never serve the root servers from cache.
201                 return null;
202             }
203             idx = idx2;
204 
205             List<DnsCacheEntry> entries = parent.authoritativeDnsServerCache().get(hostname, additionals);
206             if (entries != null && !entries.isEmpty()) {
207                 // Found a match in the cache... Also shuffle them so we not always use the same order for lookup.
208                 return DnsServerAddresses.shuffled(new DnsCacheIterable(entries)).stream();
209             }
210         }
211     }

213     private final class DnsCacheIterable implements Iterable<InetSocketAddress> {
214         private final List<DnsCacheEntry> entries;
215 
216         DnsCacheIterable(List<DnsCacheEntry> entries) {
217             this.entries = entries;
218         }
219 
220         @Override
221         public Iterator<InetSocketAddress> iterator() {
222             return new Iterator<InetSocketAddress>() {
223                 Iterator<DnsCacheEntry> entryIterator = entries.iterator();
224 
225                 @Override
226                 public boolean hasNext() {
227                     return entryIterator.hasNext();
228                 }
229 
230                 @Override
231                 public InetSocketAddress next() {
232                     InetAddress address = entryIterator.next().address();
233                     return new InetSocketAddress(address, parent.dnsRedirectPort(address));
234                 }
235 
236                 @Override
237                 public void remove() {
238                     entryIterator.remove();
239                 }
240             };
241         }
242     }

244     private void query(final DnsServerAddressStream nameServerAddrStream, final DnsQuestion question,
245                        final Promise<T> promise) {
246         if (allowedQueries == 0 || promise.isCancelled()) {
247             tryToFinishResolve(promise);
248             return;
249         }
250 
251         allowedQueries --;
252 
253         final Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> f = parent.query0(
254                 nameServerAddrStream.next(), question, additionals,
255                 parent.ch.eventLoop().<AddressedEnvelope<? extends DnsResponse, InetSocketAddress>>newPromise());
256         queriesInProgress.add(f);
257 
258         f.addListener(new FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>>() {
259             @Override
260             public void operationComplete(Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
261                 queriesInProgress.remove(future);
262 
263                 if (promise.isDone() || future.isCancelled()) {
264                     return;
265                 }
266 
267                 try {
268                     if (future.isSuccess()) {
269                         onResponse(nameServerAddrStream, question, future.getNow(), promise);
270                     } else {
271                         // Server did not respond or I/O error occurred; try again.
272                         if (traceEnabled) {
273                             addTrace(future.cause());
274                         }
275                         query(nameServerAddrStream, question, promise);
276                     }
277                 } finally {
278                     tryToFinishResolve(promise);
279                 }
280             }
281         });
282     }

284     void onResponse(final DnsServerAddressStream nameServerAddrStream, final DnsQuestion question,
285                     AddressedEnvelope<DnsResponse, InetSocketAddress> envelope, Promise<T> promise) {
286         try {
287             final DnsResponse res = envelope.content();
288             final DnsResponseCode code = res.code();
289             if (code == DnsResponseCode.NOERROR) {
290                 if (handleRedirect(question, envelope, promise)) {
291                     // Was a redirect so return here as everything else is handled in handleRedirect(...)
292                     return;
293                 }
294                 final DnsRecordType type = question.type();
295 
296                 if (type == DnsRecordType.A || type == DnsRecordType.AAAA) {
297                     onResponseAorAAAA(type, question, envelope, promise);
298                 } else if (type == DnsRecordType.CNAME) {
299                     onResponseCNAME(question, envelope, promise);
300                 }
301                 return;
302             }
303 
304             if (traceEnabled) {
305                 addTrace(envelope.sender(),
306                          "response code: " + code + " with " + res.count(DnsSection.ANSWER) + " answer(s) and " +
307                          res.count(DnsSection.AUTHORITY) + " authority resource(s)");
308             }
309 
310             // Retry with the next server if the server did not tell us that the domain does not exist.
311             if (code != DnsResponseCode.NXDOMAIN) {
312                 query(nameServerAddrStream, question, promise);
313             }
314         } finally {
315             ReferenceCountUtil.safeRelease(envelope);
316         }
317     }

319     /**
320      * Handle redirects if needed and returns {@code true} if there was a redirect handled. If {@code true} is returned
321      * this method took ownership of the {@code promise} otherwise {@code false} is returned.
322      */
323     private boolean handleRedirect(
324             DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> envelope, Promise<T> promise) {
325         final DnsResponse res = envelope.content();
326 
327         // Check if we have answers, if not this may be an non authority NS and so redirects must be handled.
328         if (res.count(DnsSection.ANSWER) == 0) {
329             AuthoritativeNameServerList serverNames = extractAuthoritativeNameServers(question.name(), res);
330 
331             if (serverNames != null) {
332                 List<InetSocketAddress> nameServers = new ArrayList<InetSocketAddress>(serverNames.size());
333                 int additionalCount = res.count(DnsSection.ADDITIONAL);
334 
335                 for (int i = 0; i < additionalCount; i++) {
336                     final DnsRecord r = res.recordAt(DnsSection.ADDITIONAL, i);
337 
338                     if ((r.type() == DnsRecordType.A && !parent.supportsARecords()) ||
339                             r.type() == DnsRecordType.AAAA && !parent.supportsAAAARecords()) {
340                         continue;
341                     }
342 
343                     final String recordName = r.name();
344                     AuthoritativeNameServer authoritativeNameServer =
345                             serverNames.remove(recordName);
346 
347                     if (authoritativeNameServer == null) {
348                         // Not a server we are interested in.
349                         continue;
350                     }
351 
352                     InetAddress resolved = parseAddress(r, recordName);
353                     if (resolved == null) {
354                         // Could not parse it, move to the next.
355                         continue;
356                     }
357 
358                     nameServers.add(new InetSocketAddress(resolved, parent.dnsRedirectPort(resolved)));
359                     addNameServerToCache(authoritativeNameServer, resolved, r.timeToLive());
360                 }
361 
362                 if (nameServers.isEmpty()) {
363                     promise.tryFailure(
364                             new UnknownHostException("Unable to find correct name server for " + hostname));
365                 } else {
366                     // Shuffle as we want to re-distribute the load across name servers.
367                     query(DnsServerAddresses.shuffled(nameServers).stream(), question, promise);
368                 }
369                 return true;
370             }
371         }
372         return false;
373     }

375     /**
376      * Returns the {@code {@link AuthoritativeNameServerList} which were included in {@link DnsSection#AUTHORITY}
377      * or {@code null} if non are found.
378      */
379     private static AuthoritativeNameServerList extractAuthoritativeNameServers(String questionName, DnsResponse res) {
380         int authorityCount = res.count(DnsSection.AUTHORITY);
381         if (authorityCount == 0) {
382             return null;
383         }
384 
385         AuthoritativeNameServerList serverNames = new AuthoritativeNameServerList(questionName);
386         for (int i = 0; i < authorityCount; i++) {
387             serverNames.add(res.recordAt(DnsSection.AUTHORITY, i));
388         }
389         return serverNames;
390     }

392     private void onResponseAorAAAA(
393             DnsRecordType qType, DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> envelope,
394             Promise<T> promise) {
395 
396         // We often get a bunch of CNAMES as well when we asked for A/AAAA.
397         final DnsResponse response = envelope.content();
398         final Map<String, String> cnames = buildAliasMap(response);
399         final int answerCount = response.count(DnsSection.ANSWER);
400 
401         boolean found = false;
402         for (int i = 0; i < answerCount; i ++) {
403             final DnsRecord r = response.recordAt(DnsSection.ANSWER, i);
404             final DnsRecordType type = r.type();
405             if (type != DnsRecordType.A && type != DnsRecordType.AAAA) {
406                 continue;
407             }
408 
409             final String questionName = question.name().toLowerCase(Locale.US);
410             final String recordName = r.name().toLowerCase(Locale.US);
411 
412             // Make sure the record is for the questioned domain.
413             if (!recordName.equals(questionName)) {
414                 // Even if the record's name is not exactly same, it might be an alias defined in the CNAME records.
415                 String resolved = questionName;
416                 do {
417                     resolved = cnames.get(resolved);
418                     if (recordName.equals(resolved)) {
419                         break;
420                     }
421                 } while (resolved != null);
422 
423                 if (resolved == null) {
424                     continue;
425                 }
426             }
427 
428             InetAddress resolved = parseAddress(r, hostname);
429             if (resolved == null) {
430                 continue;
431             }
432 
433             if (resolvedEntries == null) {
434                 resolvedEntries = new ArrayList<DnsCacheEntry>(8);
435             }
436 
437             final DnsCacheEntry e = new DnsCacheEntry(hostname, resolved);
438             resolveCache.cache(hostname, additionals, resolved, r.timeToLive(), parent.ch.eventLoop());
439             resolvedEntries.add(e);
440             found = true;
441 
442             // Note that we do not break from the loop here, so we decode/cache all A/AAAA records.
443         }
444 
445         if (found) {
446             return;
447         }
448 
449         if (traceEnabled) {
450             addTrace(envelope.sender(), "no matching " + qType + " record found");
451         }
452 
453         // We aked for A/AAAA but we got only CNAME.
454         if (!cnames.isEmpty()) {
455             onResponseCNAME(question, envelope, cnames, false, promise);
456         }
457     }

459     private InetAddress parseAddress(DnsRecord r, String name) {
460         if (!(r instanceof DnsRawRecord)) {
461             return null;
462         }
463         final ByteBuf content = ((ByteBufHolder) r).content();
464         final int contentLen = content.readableBytes();
465         if (contentLen != INADDRSZ4 && contentLen != INADDRSZ6) {
466             return null;
467         }
468 
469         final byte[] addrBytes = new byte[contentLen];
470         content.getBytes(content.readerIndex(), addrBytes);
471 
472         try {
473             return InetAddress.getByAddress(
474                     parent.isDecodeIdn() ? IDN.toUnicode(name) : name, addrBytes);
475         } catch (UnknownHostException e) {
476             // Should never reach here.
477             throw new Error(e);
478         }
479     }

481     private void onResponseCNAME(DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> envelope,

483         onResponseCNAME(question, envelope, buildAliasMap(envelope.content()), true, promise);

486     private void onResponseCNAME(
487             DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> response,
488             Map<String, String> cnames, boolean trace, Promise<T> promise) {
489 
490         // Resolve the host name in the question into the real host name.
491         final String name = question.name().toLowerCase(Locale.US);
492         String resolved = name;
493         boolean found = false;
494         while (!cnames.isEmpty()) { // Do not attempt to call Map.remove() when the Map is empty
495                                     // because it can be Collections.emptyMap()
496                                     // whose remove() throws a UnsupportedOperationException.
497             final String next = cnames.remove(resolved);
498             if (next != null) {
499                 found = true;
500                 resolved = next;
501             } else {
502                 break;
503             }
504         }
505 
506         if (found) {
507             followCname(response.sender(), name, resolved, promise);
508         } else if (trace && traceEnabled) {
509             addTrace(response.sender(), "no matching CNAME record found");
510         }
511     }

513     private static Map<String, String> buildAliasMap(DnsResponse response) {
514         final int answerCount = response.count(DnsSection.ANSWER);
515         Map<String, String> cnames = null;
516         for (int i = 0; i < answerCount; i ++) {
517             final DnsRecord r = response.recordAt(DnsSection.ANSWER, i);
518             final DnsRecordType type = r.type();
519             if (type != DnsRecordType.CNAME) {
520                 continue;
521             }
522 
523             if (!(r instanceof DnsRawRecord)) {
524                 continue;
525             }
526 
527             final ByteBuf recordContent = ((ByteBufHolder) r).content();
528             final String domainName = decodeDomainName(recordContent);
529             if (domainName == null) {
530                 continue;
531             }
532 
533             if (cnames == null) {
534                 cnames = new HashMap<String, String>();
535             }
536 
537             cnames.put(r.name().toLowerCase(Locale.US), domainName.toLowerCase(Locale.US));
538         }
539 
540         return cnames != null? cnames : Collections.<String, String>emptyMap();
541     }

543     void tryToFinishResolve(Promise<T> promise) {
544         if (!queriesInProgress.isEmpty()) {
545             // There are still some queries we did not receive responses for.
546             if (gotPreferredAddress()) {
547                 // But it's OK to finish the resolution process if we got a resolved address of the preferred type.
548                 finishResolve(promise);
549             }
550 
551             // We did not get any resolved address of the preferred type, so we can't finish the resolution process.
552             return;
553         }
554 
555         // There are no queries left to try.
556         if (resolvedEntries == null) {
557             // .. and we could not find any A/AAAA records.
558             if (!triedCNAME) {
559                 // As the last resort, try to query CNAME, just in case the name server has it.
560                 triedCNAME = true;
561 
562                 query(hostname, DnsRecordType.CNAME, getNameServers(hostname), promise);
563                 return;
564             }
565         }
566 
567         // We have at least one resolved address or tried CNAME as the last resort..
568         finishResolve(promise);
569     }

571     private boolean gotPreferredAddress() {
572         if (resolvedEntries == null) {
573             return false;
574         }
575 
576         final int size = resolvedEntries.size();
577         switch (parent.preferredAddressType()) {
578         case IPv4:
579             for (int i = 0; i < size; i ++) {
580                 if (resolvedEntries.get(i).address() instanceof Inet4Address) {
581                     return true;
582                 }
583             }
584             break;
585         case IPv6:
586             for (int i = 0; i < size; i ++) {
587                 if (resolvedEntries.get(i).address() instanceof Inet6Address) {
588                     return true;
589                 }
590             }
591             break;
592         default:
593             throw new Error();
594         }
595 
596         return false;
597     }

599     private void finishResolve(Promise<T> promise) {
600         if (!queriesInProgress.isEmpty()) {
601             // If there are queries in progress, we should cancel it because we already finished the resolution.
602             for (Iterator<Future<AddressedEnvelope<DnsResponse, InetSocketAddress>>> i = queriesInProgress.iterator();
603                  i.hasNext();) {
604                 Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> f = i.next();
605                 i.remove();
606 
607                 if (!f.cancel(false)) {
608                     f.addListener(RELEASE_RESPONSE);
609                 }
610             }
611         }
612 
613         if (resolvedEntries != null) {
614             // Found at least one resolved address.
615             for (InternetProtocolFamily f: resolvedInternetProtocolFamilies) {
616                 if (finishResolve(f.addressType(), resolvedEntries, promise)) {
617                     return;
618                 }
619             }
620         }
621 
622         // No resolved address found.
623         final int tries = maxAllowedQueries - allowedQueries;
624         final StringBuilder buf = new StringBuilder(64);
625 
626         buf.append("failed to resolve '");
627         if (pristineHostname != null) {
628           buf.append(pristineHostname);
629         } else {
630           buf.append(hostname);
631         }
632         buf.append('\'');
633         if (tries > 1) {
634             if (tries < maxAllowedQueries) {
635                 buf.append(" after ")
636                    .append(tries)
637                    .append(" queries ");
638             } else {
639                 buf.append(". Exceeded max queries per resolve ")
640                 .append(maxAllowedQueries)
641                 .append(' ');
642             }
643         }
644         if (trace != null) {
645             buf.append(':')
646                .append(trace);
647         }
648         final UnknownHostException cause = new UnknownHostException(buf.toString());
649 
650         resolveCache.cache(hostname, additionals, cause, parent.ch.eventLoop());
651         promise.tryFailure(cause);
652     }

661     static String decodeDomainName(ByteBuf in) {
662         in.markReaderIndex();
663         try {
664             return DefaultDnsRecordDecoder.decodeName(in);
665         } catch (CorruptedFrameException e) {
666             // In this case we just return null.
667             return null;
668         } finally {
669             in.resetReaderIndex();
670         }
671     }

678     private void followCname(InetSocketAddress nameServerAddr, String name, String cname, Promise<T> promise) {
679 
680         if (traceEnabled) {
681             if (trace == null) {
682                 trace = new StringBuilder(128);
683             }
684 
685             trace.append(StringUtil.NEWLINE);
686             trace.append("\tfrom ");
687             trace.append(nameServerAddr);
688             trace.append(": ");
689             trace.append(name);
690             trace.append(" CNAME ");
691             trace.append(cname);
692         }
693 
694         // Use the same server for both CNAME queries
695         DnsServerAddressStream stream = DnsServerAddresses.singleton(getNameServers(cname).next()).stream();
696 
697         if (parent.supportsARecords() && !query(hostname, DnsRecordType.A, stream, promise)) {
698             return;
699         }
700         if (parent.supportsAAAARecords()) {
701             query(hostname, DnsRecordType.AAAA, stream, promise);
702         }
703     }

706         final DnsQuestion question;
707         try {
708             question = new DefaultDnsQuestion(hostname, type);
709         } catch (IllegalArgumentException e) {
710             // java.net.IDN.toASCII(...) may throw an IllegalArgumentException if it fails to parse the hostname
711             promise.tryFailure(e);
712             return false;
713         }
714         query(nextAddr, question, promise);
715         return true;

744     /**
745      * Holds the closed DNS Servers for a domain.
746      */
747     private static final class AuthoritativeNameServerList {
748 
749         private final String questionName;
750 
751         // We not expect the linked-list to be very long so a double-linked-list is overkill.
752         private AuthoritativeNameServer head;
753         private int count;
754 
755         AuthoritativeNameServerList(String questionName) {
756             this.questionName = questionName.toLowerCase(Locale.US);
757         }
758 
759         void add(DnsRecord r) {
760             if (r.type() != DnsRecordType.NS || !(r instanceof DnsRawRecord)) {
761                 return;
762             }
763 
764             // Only include servers that serve the correct domain.
765             if (questionName.length() <  r.name().length()) {
766                 return;
767             }
768 
769             String recordName = r.name().toLowerCase(Locale.US);
770 
771             int dots = 0;
772             for (int a = recordName.length() - 1, b = questionName.length() - 1; a >= 0; a--, b--) {
773                 char c = recordName.charAt(a);
774                 if (questionName.charAt(b) != c) {
775                     return;
776                 }
777                 if (c == '.') {
778                     dots++;
779                 }
780             }
781 
782             if (head != null && head.dots > dots) {
783                 // We already have a closer match so ignore this one, no need to parse the domainName etc.
784                 return;
785             }
786 
787             final ByteBuf recordContent = ((ByteBufHolder) r).content();
788             final String domainName = decodeDomainName(recordContent);
789             if (domainName == null) {
790                 // Could not be parsed, ignore.
791                 return;
792             }
793 
794             // We are only interested in preserving the nameservers which are the closest to our qName, so ensure
795             // we drop servers that have a smaller dots count.
796             if (head == null || head.dots < dots) {
797                 count = 1;
798                 head = new AuthoritativeNameServer(dots, recordName, domainName);
799             } else if (head.dots == dots) {
800                 AuthoritativeNameServer serverName = head;
801                 while (serverName.next != null) {
802                     serverName = serverName.next;
803                 }
804                 serverName.next = new AuthoritativeNameServer(dots, recordName, domainName);
805                 count++;
806             }
807         }
808 
809         // Just walk the linked-list and mark the entry as removed when matched, so next lookup will need to process
810         // one node less.
811         AuthoritativeNameServer remove(String nsName) {
812             AuthoritativeNameServer serverName = head;
813 
814             while (serverName != null) {
815                 if (!serverName.removed && serverName.nsName.equalsIgnoreCase(nsName)) {
816                     serverName.removed = true;
817                     return serverName;
818                 }
819                 serverName = serverName.next;
820             }
821             return null;
822         }
823 
824         int size() {
825             return count;
826         }
827     }

830         final int dots;
831         final String nsName;
832         final String domainName;
833 
834         AuthoritativeNameServer next;
835         boolean removed;
836 
837         AuthoritativeNameServer(int dots, String domainName, String nsName) {
838             this.dots = dots;
839             this.nsName = nsName;
840             this.domainName = domainName;
841         }

843         /**
844          * Returns {@code true} if its a root server.
845          */
846         boolean isRootServer() {
847             return dots == 1;
848         }
849 
850         /**
851          * The domain for which the {@link AuthoritativeNameServer} is responsible.
852          */
853         String domainName() {
854             return domainName;
855         }
