158     void resolve(final Promise<List<T>> promise) {
159         final String[] searchDomains = parent.searchDomains();
160         if (searchDomains.length == 0 || parent.ndots() == 0 || StringUtil.endsWith(hostname, '.')) {
161             internalResolve(promise);
162         } else {
163             final boolean startWithoutSearchDomain = hasNDots();
164             final String initialHostname = startWithoutSearchDomain ? hostname : hostname + '.' + searchDomains[0];
165             final int initialSearchDomainIdx = startWithoutSearchDomain ? 0 : 1;
166 
167             doSearchDomainQuery(initialHostname, new FutureListener<List<T>>() {
168                 private int searchDomainIdx = initialSearchDomainIdx;
169                 @Override
170                 public void operationComplete(Future<List<T>> future) {
171                     Throwable cause = future.cause();
172                     if (cause == null) {
173                         promise.trySuccess(future.getNow());
174                     } else {
175                         if (DnsNameResolver.isTransportOrTimeoutError(cause)) {
176                             promise.tryFailure(new SearchDomainUnknownHostException(cause, hostname));
177                         } else if (searchDomainIdx < searchDomains.length) {
178                             doSearchDomainQuery(hostname + '.' + searchDomains[searchDomainIdx++], this);
179                         } else if (!startWithoutSearchDomain) {
180                             internalResolve(promise);
181                         } else {
182                             promise.tryFailure(new SearchDomainUnknownHostException(cause, hostname));
183                         }
184                     }
185                 }
186             });
187         }
188     }
189 
190     private boolean hasNDots() {
191         for (int idx = hostname.length() - 1, dots = 0; idx >= 0; idx--) {
192             if (hostname.charAt(idx) == '.' && ++dots >= parent.ndots()) {
193                 return true;
194             }
195         }
196         return false;
197     }

216     private void doSearchDomainQuery(String hostname, FutureListener<List<T>> listener) {
217         DnsResolveContext<T> nextContext = newResolverContext(parent, hostname, dnsClass, expectedTypes,
218                                                               additionals, nameServerAddrs);
219         Promise<List<T>> nextPromise = parent.executor().newPromise();
220         nextContext.internalResolve(nextPromise);
221         nextPromise.addListener(listener);
222     }

224     private void internalResolve(Promise<List<T>> promise) {
225         DnsServerAddressStream nameServerAddressStream = getNameServers(hostname);
226 
227         final int end = expectedTypes.length - 1;
228         for (int i = 0; i < end; ++i) {
229             if (!query(hostname, expectedTypes[i], nameServerAddressStream.duplicate(), promise)) {
230                 return;
231             }
232         }
233         query(hostname, expectedTypes[end], nameServerAddressStream, promise);
234     }

239     private void addNameServerToCache(
240             AuthoritativeNameServer name, InetAddress resolved, long ttl) {
241         if (!name.isRootServer()) {
242             // Cache NS record if not for a root server as we should never cache for root servers.
243             parent.authoritativeDnsServerCache().cache(name.domainName(),
244                     additionals, resolved, ttl, parent.ch.eventLoop());
245         }
246     }

248     /**
249      * Returns the {@link DnsServerAddressStream} that was cached for the given hostname or {@code null} if non
250      *  could be found.
251      */
252     private DnsServerAddressStream getNameServersFromCache(String hostname) {
253         int len = hostname.length();
254 
255         if (len == 0) {
256             // We never cache for root servers.
257             return null;
258         }
259 
260         // We always store in the cache with a trailing '.'.
261         if (hostname.charAt(len - 1) != '.') {
262             hostname += ".";
263         }
264 
265         int idx = hostname.indexOf('.');
266         if (idx == hostname.length() - 1) {
267             // We are not interested in handling '.' as we should never serve the root servers from cache.
268             return null;
269         }
270 
271         // We start from the closed match and then move down.
272         for (;;) {
273             // Skip '.' as well.
274             hostname = hostname.substring(idx + 1);
275 
276             int idx2 = hostname.indexOf('.');
277             if (idx2 <= 0 || idx2 == hostname.length() - 1) {
278                 // We are not interested in handling '.TLD.' as we should never serve the root servers from cache.
279                 return null;
280             }
281             idx = idx2;
282 
283             List<? extends DnsCacheEntry> entries = parent.authoritativeDnsServerCache().get(hostname, additionals);
284             if (entries != null && !entries.isEmpty()) {
285                 return DnsServerAddresses.sequential(new DnsCacheIterable(entries)).stream();
286             }
287         }
288     }

290     private final class DnsCacheIterable implements Iterable<InetSocketAddress> {
291         private final List<? extends DnsCacheEntry> entries;
292 
293         DnsCacheIterable(List<? extends DnsCacheEntry> entries) {
294             this.entries = entries;
295         }
296 
297         @Override
298         public Iterator<InetSocketAddress> iterator() {
299             return new Iterator<InetSocketAddress>() {
300                 Iterator<? extends DnsCacheEntry> entryIterator = entries.iterator();
301 
302                 @Override
303                 public boolean hasNext() {
304                     return entryIterator.hasNext();
305                 }
306 
307                 @Override
308                 public InetSocketAddress next() {
309                     InetAddress address = entryIterator.next().address();
310                     return new InetSocketAddress(address, parent.dnsRedirectPort(address));
311                 }
312 
313                 @Override
314                 public void remove() {
315                     entryIterator.remove();
316                 }
317             };
318         }
319     }

322                        final DnsQuestion question,
323                        final Promise<List<T>> promise, Throwable cause) {

328     private void query(final DnsServerAddressStream nameServerAddrStream,
329                        final int nameServerAddrStreamIndex,
330                        final DnsQuestion question,
331                        final DnsQueryLifecycleObserver queryLifecycleObserver,
332                        final Promise<List<T>> promise,
333                        final Throwable cause) {
334         if (nameServerAddrStreamIndex >= nameServerAddrStream.size() || allowedQueries == 0 || promise.isCancelled()) {
335             tryToFinishResolve(nameServerAddrStream, nameServerAddrStreamIndex, question, queryLifecycleObserver,
336                                promise, cause);
337             return;
338         }
339 
340         --allowedQueries;
341         final InetSocketAddress nameServerAddr = nameServerAddrStream.next();
342         final ChannelPromise writePromise = parent.ch.newPromise();
343         final Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> f = parent.query0(
344                 nameServerAddr, question, additionals, writePromise,
345                 parent.ch.eventLoop().<AddressedEnvelope<? extends DnsResponse, InetSocketAddress>>newPromise());
346         queriesInProgress.add(f);
347 
348         queryLifecycleObserver.queryWritten(nameServerAddr, writePromise);
349 
350         f.addListener(new FutureListener<AddressedEnvelope<DnsResponse, InetSocketAddress>>() {
351             @Override
352             public void operationComplete(Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> future) {
353                 queriesInProgress.remove(future);
354 
355                 if (promise.isDone() || future.isCancelled()) {
356                     queryLifecycleObserver.queryCancelled(allowedQueries);
357 
358                     // Check if we need to release the envelope itself. If the query was cancelled the getNow() will
359                     // return null as well as the Future will be failed with a CancellationException.
360                     AddressedEnvelope<DnsResponse, InetSocketAddress> result = future.getNow();
361                     if (result != null) {
362                         result.release();
363                     }
364                     return;
365                 }
366 
367                 final Throwable queryCause = future.cause();
368                 try {
369                     if (queryCause == null) {
370                         onResponse(nameServerAddrStream, nameServerAddrStreamIndex, question, future.getNow(),
371                                    queryLifecycleObserver, promise);
372                     } else {
373                         // Server did not respond or I/O error occurred; try again.
374                         queryLifecycleObserver.queryFailed(queryCause);
375                         query(nameServerAddrStream, nameServerAddrStreamIndex + 1, question, promise, queryCause);
376                     }
377                 } finally {
378                     tryToFinishResolve(nameServerAddrStream, nameServerAddrStreamIndex, question,
379                                        // queryLifecycleObserver has already been terminated at this point so we must
380                                        // not allow it to be terminated again by tryToFinishResolve.
381                                        NoopDnsQueryLifecycleObserver.INSTANCE,
382                                        promise, queryCause);
383                 }
384             }
385         });
386     }

388     private void onResponse(final DnsServerAddressStream nameServerAddrStream, final int nameServerAddrStreamIndex,
389                     final DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> envelope,
390                     final DnsQueryLifecycleObserver queryLifecycleObserver,
391                     Promise<List<T>> promise) {
392         try {
393             final DnsResponse res = envelope.content();
394             final DnsResponseCode code = res.code();
395             if (code == DnsResponseCode.NOERROR) {
396                 if (handleRedirect(question, envelope, queryLifecycleObserver, promise)) {
397                     // Was a redirect so return here as everything else is handled in handleRedirect(...)
398                     return;
399                 }
400                 final DnsRecordType type = question.type();
401 
402                 if (type == DnsRecordType.CNAME) {
403                     onResponseCNAME(question, buildAliasMap(envelope.content()), queryLifecycleObserver, promise);
404                     return;
405                 }
406 
407                 for (DnsRecordType expectedType : expectedTypes) {
408                     if (type == expectedType) {
409                         onExpectedResponse(question, envelope, queryLifecycleObserver, promise);
410                         return;
411                     }
412                 }
413 
414                 queryLifecycleObserver.queryFailed(UNRECOGNIZED_TYPE_QUERY_FAILED_EXCEPTION);
415                 return;
416             }
417 
418             // Retry with the next server if the server did not tell us that the domain does not exist.
419             if (code != DnsResponseCode.NXDOMAIN) {
420                 query(nameServerAddrStream, nameServerAddrStreamIndex + 1, question,
421                       queryLifecycleObserver.queryNoAnswer(code), promise, null);
422             } else {
423                 queryLifecycleObserver.queryFailed(NXDOMAIN_QUERY_FAILED_EXCEPTION);
424             }
425         } finally {
426             ReferenceCountUtil.safeRelease(envelope);
427         }
428     }

430     /**
431      * Handles a redirect answer if needed and returns {@code true} if a redirect query has been made.
432      */
433     private boolean handleRedirect(
434             DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> envelope,
435             final DnsQueryLifecycleObserver queryLifecycleObserver, Promise<List<T>> promise) {
436         final DnsResponse res = envelope.content();
437 
438         // Check if we have answers, if not this may be an non authority NS and so redirects must be handled.
439         if (res.count(DnsSection.ANSWER) == 0) {
440             AuthoritativeNameServerList serverNames = extractAuthoritativeNameServers(question.name(), res);
441 
442             if (serverNames != null) {
443                 List<InetSocketAddress> nameServers = new ArrayList<InetSocketAddress>(serverNames.size());
444                 int additionalCount = res.count(DnsSection.ADDITIONAL);
445 
446                 for (int i = 0; i < additionalCount; i++) {
447                     final DnsRecord r = res.recordAt(DnsSection.ADDITIONAL, i);
448 
449                     if (r.type() == DnsRecordType.A && !parent.supportsARecords() ||
450                         r.type() == DnsRecordType.AAAA && !parent.supportsAAAARecords()) {
451                         continue;
452                     }
453 
454                     final String recordName = r.name();
455                     final AuthoritativeNameServer authoritativeNameServer = serverNames.remove(recordName);
456 
457                     if (authoritativeNameServer == null) {
458                         // Not a server we are interested in.
459                         continue;
460                     }
461 
462                     InetAddress resolved = decodeAddress(r, recordName, parent.isDecodeIdn());
463                     if (resolved == null) {
464                         // Could not parse it, move to the next.
465                         continue;
466                     }
467 
468                     nameServers.add(new InetSocketAddress(resolved, parent.dnsRedirectPort(resolved)));
469                     addNameServerToCache(authoritativeNameServer, resolved, r.timeToLive());
470                 }
471 
472                 if (!nameServers.isEmpty()) {
473                     query(parent.uncachedRedirectDnsServerStream(nameServers), 0, question,
474                           queryLifecycleObserver.queryRedirected(unmodifiableList(nameServers)), promise, null);
475                     return true;
476                 }
477             }
478         }
479         return false;
480     }

482     /**
483      * Returns the {@code {@link AuthoritativeNameServerList} which were included in {@link DnsSection#AUTHORITY}
484      * or {@code null} if non are found.
485      */
486     private static AuthoritativeNameServerList extractAuthoritativeNameServers(String questionName, DnsResponse res) {
487         int authorityCount = res.count(DnsSection.AUTHORITY);
488         if (authorityCount == 0) {
489             return null;
490         }
491 
492         AuthoritativeNameServerList serverNames = new AuthoritativeNameServerList(questionName);
493         for (int i = 0; i < authorityCount; i++) {
494             serverNames.add(res.recordAt(DnsSection.AUTHORITY, i));
495         }
496         return serverNames;
497     }

499     private void onExpectedResponse(
500             DnsQuestion question, AddressedEnvelope<DnsResponse, InetSocketAddress> envelope,
501             final DnsQueryLifecycleObserver queryLifecycleObserver, Promise<List<T>> promise) {
502 
503         // We often get a bunch of CNAMES as well when we asked for A/AAAA.
504         final DnsResponse response = envelope.content();
505         final Map<String, String> cnames = buildAliasMap(response);
506         final int answerCount = response.count(DnsSection.ANSWER);
507 
508         boolean found = false;
509         for (int i = 0; i < answerCount; i ++) {
510             final DnsRecord r = response.recordAt(DnsSection.ANSWER, i);
511             final DnsRecordType type = r.type();
512             boolean matches = false;
513             for (DnsRecordType expectedType : expectedTypes) {
514                 if (type == expectedType) {
515                     matches = true;
516                     break;
517                 }
518             }
519 
520             if (!matches) {
521                 continue;
522             }
523 
524             final String questionName = question.name().toLowerCase(Locale.US);
525             final String recordName = r.name().toLowerCase(Locale.US);
526 
527             // Make sure the record is for the questioned domain.
528             if (!recordName.equals(questionName)) {
529                 // Even if the record's name is not exactly same, it might be an alias defined in the CNAME records.
530                 String resolved = questionName;
531                 do {
532                     resolved = cnames.get(resolved);
533                     if (recordName.equals(resolved)) {
534                         break;
535                     }
536                 } while (resolved != null);
537 
538                 if (resolved == null) {
539                     continue;
540                 }
541             }
542 
543             final T converted = convertRecord(r, hostname, additionals, parent.ch.eventLoop());
544             if (converted == null) {
545                 continue;
546             }
547 
548             if (finalResult == null) {
549                 finalResult = new ArrayList<T>(8);
550             }
551             finalResult.add(converted);
552 
553             cache(hostname, additionals, r, converted);
554             found = true;
555 
556             // Note that we do not break from the loop here, so we decode/cache all A/AAAA records.
557         }
558 
559         if (cnames.isEmpty()) {
560             if (found) {
561                 queryLifecycleObserver.querySucceed();
562                 return;
563             }
564             queryLifecycleObserver.queryFailed(NO_MATCHING_RECORD_QUERY_FAILED_EXCEPTION);
565         } else {
566             queryLifecycleObserver.querySucceed();
567             // We also got a CNAME so we need to ensure we also query it.
568             onResponseCNAME(question, cnames,
569                     parent.dnsQueryLifecycleObserverFactory().newDnsQueryLifecycleObserver(question), promise);
570         }
571     }

573     private void onResponseCNAME(
574             DnsQuestion question, Map<String, String> cnames,
575             final DnsQueryLifecycleObserver queryLifecycleObserver,
576             Promise<List<T>> promise) {
577 
578         // Resolve the host name in the question into the real host name.
579         String resolved = question.name().toLowerCase(Locale.US);
580         boolean found = false;
581         while (!cnames.isEmpty()) { // Do not attempt to call Map.remove() when the Map is empty
582                                     // because it can be Collections.emptyMap()
583                                     // whose remove() throws a UnsupportedOperationException.
584             final String next = cnames.remove(resolved);
585             if (next != null) {
586                 found = true;
587                 resolved = next;
588             } else {
589                 break;
590             }
591         }
592 
593         if (found) {
594             followCname(question, resolved, queryLifecycleObserver, promise);
595         } else {
596             queryLifecycleObserver.queryFailed(CNAME_NOT_FOUND_QUERY_FAILED_EXCEPTION);
597         }
598     }

600     private static Map<String, String> buildAliasMap(DnsResponse response) {
601         final int answerCount = response.count(DnsSection.ANSWER);
602         Map<String, String> cnames = null;
603         for (int i = 0; i < answerCount; i ++) {
604             final DnsRecord r = response.recordAt(DnsSection.ANSWER, i);
605             final DnsRecordType type = r.type();
606             if (type != DnsRecordType.CNAME) {
607                 continue;
608             }
609 
610             if (!(r instanceof DnsRawRecord)) {
611                 continue;
612             }
613 
614             final ByteBuf recordContent = ((ByteBufHolder) r).content();
615             final String domainName = decodeDomainName(recordContent);
616             if (domainName == null) {
617                 continue;
618             }
619 
620             if (cnames == null) {
621                 cnames = new HashMap<String, String>(min(8, answerCount));
622             }
623 
624             cnames.put(r.name().toLowerCase(Locale.US), domainName.toLowerCase(Locale.US));
625         }
626 
627         return cnames != null? cnames : Collections.<String, String>emptyMap();
628     }

630     private void tryToFinishResolve(final DnsServerAddressStream nameServerAddrStream,
631                             final int nameServerAddrStreamIndex,
632                             final DnsQuestion question,
633                             final DnsQueryLifecycleObserver queryLifecycleObserver,
634                             final Promise<List<T>> promise,
635                             final Throwable cause) {
636 
637         // There are no queries left to try.
638         if (!queriesInProgress.isEmpty()) {
639             queryLifecycleObserver.queryCancelled(allowedQueries);
640 
641             // There are still some queries in process, we will try to notify once the next one finishes until
642             // all are finished.
643             return;
644         }
645 
646         // There are no queries left to try.
647         if (finalResult == null) {
648             if (nameServerAddrStreamIndex < nameServerAddrStream.size()) {
649                 if (queryLifecycleObserver == NoopDnsQueryLifecycleObserver.INSTANCE) {
650                     // If the queryLifecycleObserver has already been terminated we should create a new one for this
651                     // fresh query.
652                     query(nameServerAddrStream, nameServerAddrStreamIndex + 1, question, promise, cause);
653                 } else {
654                     query(nameServerAddrStream, nameServerAddrStreamIndex + 1, question, queryLifecycleObserver,
655                           promise, cause);
656                 }
657                 return;
658             }
659 
660             queryLifecycleObserver.queryFailed(NAME_SERVERS_EXHAUSTED_EXCEPTION);
661 
662             // .. and we could not find any expected records.
663 
664             // If cause != null we know this was caused by a timeout / cancel / transport exception. In this case we
665             // won't try to resolve the CNAME as we only should do this if we could not get the expected records
666             // because they do not exist and the DNS server did probably signal it.
667             if (cause == null && !triedCNAME) {
668                 // As the last resort, try to query CNAME, just in case the name server has it.
669                 triedCNAME = true;
670 
671                 query(hostname, DnsRecordType.CNAME, getNameServers(hostname), promise);
672                 return;
673             }
674         } else {
675             queryLifecycleObserver.queryCancelled(allowedQueries);
676         }
677 
678         // We have at least one resolved record or tried CNAME as the last resort..
679         finishResolve(promise, cause);
680     }

682     private void finishResolve(Promise<List<T>> promise, Throwable cause) {
683         if (!queriesInProgress.isEmpty()) {
684             // If there are queries in progress, we should cancel it because we already finished the resolution.
685             for (Iterator<Future<AddressedEnvelope<DnsResponse, InetSocketAddress>>> i = queriesInProgress.iterator();
686                  i.hasNext();) {
687                 Future<AddressedEnvelope<DnsResponse, InetSocketAddress>> f = i.next();
688                 i.remove();
689 
690                 if (!f.cancel(false)) {
691                     f.addListener(RELEASE_RESPONSE);
692                 }
693             }
694         }
695 
696         if (finalResult != null) {
697             // Found at least one resolved record.
698             trySuccess(promise, filterResults(finalResult));
699             return;
700         }
701 
702         // No resolved address found.
703         final int tries = maxAllowedQueries - allowedQueries;
704         final StringBuilder buf = new StringBuilder(64);
705 
706         buf.append("failed to resolve '").append(hostname).append('\'');
707         if (tries > 1) {
708             if (tries < maxAllowedQueries) {
709                 buf.append(" after ")
710                    .append(tries)
711                    .append(" queries ");
712             } else {
713                 buf.append(". Exceeded max queries per resolve ")
714                 .append(maxAllowedQueries)
715                 .append(' ');
716             }
717         }
718         final UnknownHostException unknownHostException = new UnknownHostException(buf.toString());
719         if (cause == null) {
720             // Only cache if the failure was not because of an IO error / timeout that was caused by the query
721             // itself.
722             cache(hostname, additionals, unknownHostException);
723         } else {
724             unknownHostException.initCause(cause);
725         }
726         promise.tryFailure(unknownHostException);
727     }

780     /**
781      * Holds the closed DNS Servers for a domain.
782      */
783     private static final class AuthoritativeNameServerList {
784 
785         private final String questionName;
786 
787         // We not expect the linked-list to be very long so a double-linked-list is overkill.
788         private AuthoritativeNameServer head;
789         private int count;
790 
791         AuthoritativeNameServerList(String questionName) {
792             this.questionName = questionName.toLowerCase(Locale.US);
793         }
794 
795         void add(DnsRecord r) {
796             if (r.type() != DnsRecordType.NS || !(r instanceof DnsRawRecord)) {
797                 return;
798             }
799 
800             // Only include servers that serve the correct domain.
801             if (questionName.length() <  r.name().length()) {
802                 return;
803             }
804 
805             String recordName = r.name().toLowerCase(Locale.US);
806 
807             int dots = 0;
808             for (int a = recordName.length() - 1, b = questionName.length() - 1; a >= 0; a--, b--) {
809                 char c = recordName.charAt(a);
810                 if (questionName.charAt(b) != c) {
811                     return;
812                 }
813                 if (c == '.') {
814                     dots++;
815                 }
816             }
817 
818             if (head != null && head.dots > dots) {
819                 // We already have a closer match so ignore this one, no need to parse the domainName etc.
820                 return;
821             }
822 
823             final ByteBuf recordContent = ((ByteBufHolder) r).content();
824             final String domainName = decodeDomainName(recordContent);
825             if (domainName == null) {
826                 // Could not be parsed, ignore.
827                 return;
828             }
829 
830             // We are only interested in preserving the nameservers which are the closest to our qName, so ensure
831             // we drop servers that have a smaller dots count.
832             if (head == null || head.dots < dots) {
833                 count = 1;
834                 head = new AuthoritativeNameServer(dots, recordName, domainName);
835             } else if (head.dots == dots) {
836                 AuthoritativeNameServer serverName = head;
837                 while (serverName.next != null) {
838                     serverName = serverName.next;
839                 }
840                 serverName.next = new AuthoritativeNameServer(dots, recordName, domainName);
841                 count++;
842             }
843         }
844 
845         // Just walk the linked-list and mark the entry as removed when matched, so next lookup will need to process
846         // one node less.
847         AuthoritativeNameServer remove(String nsName) {
848             AuthoritativeNameServer serverName = head;
849 
850             while (serverName != null) {
851                 if (!serverName.removed && serverName.nsName.equalsIgnoreCase(nsName)) {
852                     serverName.removed = true;
853                     return serverName;
854                 }
855                 serverName = serverName.next;
856             }
857             return null;
858         }
859 
860         int size() {
861             return count;
862         }
863     }

866         final int dots;
867         final String nsName;
868         final String domainName;
869 
870         AuthoritativeNameServer next;
871         boolean removed;
872 
873         AuthoritativeNameServer(int dots, String domainName, String nsName) {
874             this.dots = dots;

876             this.domainName = domainName;

884         }
