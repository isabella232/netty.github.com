112         @Override

118         @Override
119         protected void onRemoval(Stack<T> value) {
120             // Let us remove the WeakOrderQueue from the WeakHashMap directly if its safe to remove some overhead
121             if (value.threadRef.get() == Thread.currentThread()) {
122                if (DELAYED_RECYCLED.isSet()) {
123                    DELAYED_RECYCLED.get().remove(value);
124                }
125             }
126         }

141     protected Recycler(int maxCapacityPerThread, int maxSharedCapacityFactor,
142                        int ratio, int maxDelayedQueuesPerThread) {
143         ratioMask = safeFindNextPositivePowerOfTwo(ratio) - 1;
144         if (maxCapacityPerThread <= 0) {
145             this.maxCapacityPerThread = 0;
146             this.maxSharedCapacityFactor = 1;
147             this.maxDelayedQueuesPerThread = 0;
148         } else {
149             this.maxCapacityPerThread = maxCapacityPerThread;
150             this.maxSharedCapacityFactor = max(1, maxSharedCapacityFactor);
151             this.maxDelayedQueuesPerThread = max(0, maxDelayedQueuesPerThread);
152         }
153     }

155     @SuppressWarnings("unchecked")
156     public final T get() {
157         if (maxCapacityPerThread == 0) {
158             return newObject((Handle<T>) NOOP_HANDLE);
159         }
160         Stack<T> stack = threadLocal.get();
161         DefaultHandle<T> handle = stack.pop();
162         if (handle == null) {
163             handle = stack.newHandle();
164             handle.value = newObject(handle);
165         }
166         return (T) handle.value;
167     }

231     // a queue that makes only moderate guarantees about visibility: items are seen in the correct order,
232     // but we aren't absolutely guaranteed to ever see anything at all, thereby keeping the queue cheap to maintain
233     private static final class WeakOrderQueue {
234 
235         static final WeakOrderQueue DUMMY = new WeakOrderQueue();
236 
237         // Let Link extend AtomicInteger for intrinsics. The Link itself will be used as writerIndex.
238         @SuppressWarnings("serial")
239         private static final class Link extends AtomicInteger {
240             private final DefaultHandle<?>[] elements = new DefaultHandle[LINK_CAPACITY];
241 
242             private int readIndex;
243             private Link next;
244         }
245 
246         // chain of data items
247         private Link head, tail;
248         // pointer to another queue of delayed items for the same stack
249         private WeakOrderQueue next;
250         private final WeakReference<Thread> owner;
251         private final int id = ID_GENERATOR.getAndIncrement();
252         private final AtomicInteger availableSharedCapacity;
253 
254         private WeakOrderQueue() {
255             owner = null;
256             availableSharedCapacity = null;
257         }
258 
259         private WeakOrderQueue(Stack<?> stack, Thread thread) {
260             head = tail = new Link();
261             owner = new WeakReference<Thread>(thread);
262 
263             // Its important that we not store the Stack itself in the WeakOrderQueue as the Stack also is used in
264             // the WeakHashMap as key. So just store the enclosed AtomicInteger which should allow to have the
265             // Stack itself GCed.
266             availableSharedCapacity = stack.availableSharedCapacity;
267         }
268 
269         static WeakOrderQueue newQueue(Stack<?> stack, Thread thread) {
270             WeakOrderQueue queue = new WeakOrderQueue(stack, thread);
271             // Done outside of the constructor to ensure WeakOrderQueue.this does not escape the constructor and so
272             // may be accessed while its still constructed.
273             stack.setHead(queue);
274             return queue;
275         }
276 
277         private void setNext(WeakOrderQueue next) {
278             assert next != this;
279             this.next = next;
280         }

282         /**
283          * Allocate a new {@link WeakOrderQueue} or return {@code null} if not possible.
284          */
285         static WeakOrderQueue allocate(Stack<?> stack, Thread thread) {
286             // We allocated a Link so reserve the space
287             return reserveSpace(stack.availableSharedCapacity, LINK_CAPACITY)
288                     ? newQueue(stack, thread) : null;
289         }
290 
291         private static boolean reserveSpace(AtomicInteger availableSharedCapacity, int space) {
292             assert space >= 0;
293             for (;;) {
294                 int available = availableSharedCapacity.get();
295                 if (available < space) {
296                     return false;
297                 }
298                 if (availableSharedCapacity.compareAndSet(available, available - space)) {
299                     return true;
300                 }
301             }
302         }
303 
304         private void reclaimSpace(int space) {
305             assert space >= 0;
306             availableSharedCapacity.addAndGet(space);
307         }
308 
309         void add(DefaultHandle<?> handle) {
310             handle.lastRecycledId = id;
311 
312             Link tail = this.tail;
313             int writeIndex;
314             if ((writeIndex = tail.get()) == LINK_CAPACITY) {
315                 if (!reserveSpace(availableSharedCapacity, LINK_CAPACITY)) {
316                     // Drop it.
317                     return;
318                 }
319                 // We allocate a Link so reserve the space
320                 this.tail = tail = tail.next = new Link();
321 
322                 writeIndex = tail.get();
323             }
324             tail.elements[writeIndex] = handle;
325             handle.stack = null;
326             // we lazy set to ensure that setting stack to null appears before we unnull it in the owning thread;
327             // this also means we guarantee visibility of an element in the queue if we see the index updated
328             tail.lazySet(writeIndex + 1);
329         }
330 
331         boolean hasFinalData() {
332             return tail.readIndex != tail.get();
333         }
334 
335         // transfer as many items as we can from this queue to the stack, returning true if any were transferred
336         @SuppressWarnings("rawtypes")
337         boolean transfer(Stack<?> dst) {
338             Link head = this.head;
339             if (head == null) {
340                 return false;
341             }
342 
343             if (head.readIndex == LINK_CAPACITY) {
344                 if (head.next == null) {
345                     return false;
346                 }
347                 this.head = head = head.next;
348             }
349 
350             final int srcStart = head.readIndex;
351             int srcEnd = head.get();
352             final int srcSize = srcEnd - srcStart;
353             if (srcSize == 0) {
354                 return false;
355             }
356 
357             final int dstSize = dst.size;
358             final int expectedCapacity = dstSize + srcSize;
359 
360             if (expectedCapacity > dst.elements.length) {
361                 final int actualCapacity = dst.increaseCapacity(expectedCapacity);
362                 srcEnd = min(srcStart + actualCapacity - dstSize, srcEnd);
363             }
364 
365             if (srcStart != srcEnd) {
366                 final DefaultHandle[] srcElems = head.elements;
367                 final DefaultHandle[] dstElems = dst.elements;
368                 int newDstSize = dstSize;
369                 for (int i = srcStart; i < srcEnd; i++) {
370                     DefaultHandle element = srcElems[i];
371                     if (element.recycleId == 0) {
372                         element.recycleId = element.lastRecycledId;
373                     } else if (element.recycleId != element.lastRecycledId) {
374                         throw new IllegalStateException("recycled already");
375                     }
376                     srcElems[i] = null;
377 
378                     if (dst.dropHandle(element)) {
379                         // Drop the object.
380                         continue;
381                     }
382                     element.stack = dst;
383                     dstElems[newDstSize ++] = element;
384                 }
385 
386                 if (srcEnd == LINK_CAPACITY && head.next != null) {
387                     // Add capacity back as the Link is GCed.
388                     reclaimSpace(LINK_CAPACITY);
389 
390                     this.head = head.next;
391                 }
392 
393                 head.readIndex = srcEnd;
394                 if (dst.size == newDstSize) {
395                     return false;
396                 }
397                 dst.size = newDstSize;
398                 return true;
399             } else {
400                 // The destination stack is full already.
401                 return false;
402             }
403         }

405         @Override
406         protected void finalize() throws Throwable {
407             try {
408                 super.finalize();
409             } finally {
410                 // We need to reclaim all space that was reserved by this WeakOrderQueue so we not run out of space in
411                 // the stack. This is needed as we not have a good life-time control over the queue as it is used in a
412                 // WeakHashMap which will drop it at any time.
413                 Link link = head;
414                 while (link != null) {
415                     reclaimSpace(LINK_CAPACITY);
416                     link = link.next;
417                 }
418             }
419         }

422     static final class Stack<T> {
423 
424         // we keep a queue of per-thread queues, which is appended to once only, each time a new thread other
425         // than the stack owner recycles: when we run out of items in our stack we iterate this collection
426         // to scavenge those that can be reused. this permits us to incur minimal thread synchronisation whilst
427         // still recycling all items.
428         final Recycler<T> parent;
429 
430         // We store the Thread in a WeakReference as otherwise we may be the only ones that still hold a strong
431         // Reference to the Thread itself after it died because DefaultHandle will hold a reference to the Stack.
432         //
433         // The biggest issue is if we do not use a WeakReference the Thread may not be able to be collected at all if
434         // the user will store a reference to the DefaultHandle somewhere and never clear this reference (or not clear
435         // it in a timely manner).
436         final WeakReference<Thread> threadRef;
437         final AtomicInteger availableSharedCapacity;
438         final int maxDelayedQueues;
439 
440         private final int maxCapacity;
441         private final int ratioMask;
442         private DefaultHandle<?>[] elements;
443         private int size;
444         private int handleRecycleCount = -1; // Start with -1 so the first one will be recycled.
445         private WeakOrderQueue cursor, prev;
446         private volatile WeakOrderQueue head;
447 
448         Stack(Recycler<T> parent, Thread thread, int maxCapacity, int maxSharedCapacityFactor,
449               int ratioMask, int maxDelayedQueues) {
450             this.parent = parent;
451             threadRef = new WeakReference<Thread>(thread);
452             this.maxCapacity = maxCapacity;
453             availableSharedCapacity = new AtomicInteger(max(maxCapacity / maxSharedCapacityFactor, LINK_CAPACITY));
454             elements = new DefaultHandle[min(INITIAL_CAPACITY, maxCapacity)];
455             this.ratioMask = ratioMask;
456             this.maxDelayedQueues = maxDelayedQueues;
457         }
458 
459         // Marked as synchronized to ensure this is serialized.
460         synchronized void setHead(WeakOrderQueue queue) {
461             queue.setNext(head);
462             head = queue;
463         }
464 
465         int increaseCapacity(int expectedCapacity) {
466             int newCapacity = elements.length;
467             int maxCapacity = this.maxCapacity;
468             do {
469                 newCapacity <<= 1;
470             } while (newCapacity < expectedCapacity && newCapacity < maxCapacity);
471 
472             newCapacity = min(newCapacity, maxCapacity);
473             if (newCapacity != elements.length) {
474                 elements = Arrays.copyOf(elements, newCapacity);
475             }
476 
477             return newCapacity;
478         }
479 
480         @SuppressWarnings({ "unchecked", "rawtypes" })
481         DefaultHandle<T> pop() {
482             int size = this.size;
483             if (size == 0) {
484                 if (!scavenge()) {
485                     return null;
486                 }
487                 size = this.size;
488             }
489             size --;
490             DefaultHandle ret = elements[size];
491             elements[size] = null;
492             if (ret.lastRecycledId != ret.recycleId) {
493                 throw new IllegalStateException("recycled multiple times");
494             }
495             ret.recycleId = 0;
496             ret.lastRecycledId = 0;
497             this.size = size;
498             return ret;
499         }

501         boolean scavenge() {
502             // continue an existing scavenge, if any
503             if (scavengeSome()) {
504                 return true;
505             }
506 
507             // reset our scavenge cursor
508             prev = null;
509             cursor = head;
510             return false;
511         }
512 
513         boolean scavengeSome() {
514             WeakOrderQueue prev;
515             WeakOrderQueue cursor = this.cursor;
516             if (cursor == null) {
517                 prev = null;
518                 cursor = head;
519                 if (cursor == null) {
520                     return false;
521                 }
522             } else {
523                 prev = this.prev;
524             }
525 
526             boolean success = false;
527             do {
528                 if (cursor.transfer(this)) {
529                     success = true;
530                     break;
531                 }
532                 WeakOrderQueue next = cursor.next;
533                 if (cursor.owner.get() == null) {
534                     // If the thread associated with the queue is gone, unlink it, after
535                     // performing a volatile read to confirm there is no data left to collect.
536                     // We never unlink the first queue, as we don't want to synchronize on updating the head.
537                     if (cursor.hasFinalData()) {
538                         for (;;) {
539                             if (cursor.transfer(this)) {
540                                 success = true;
541                             } else {
542                                 break;
543                             }
544                         }
545                     }
546 
547                     if (prev != null) {
548                         prev.setNext(next);
549                     }
550                 } else {
551                     prev = cursor;
552                 }
553 
554                 cursor = next;
555 
556             } while (cursor != null && !success);
557 
558             this.prev = prev;
559             this.cursor = cursor;
560             return success;
561         }

563         void push(DefaultHandle<?> item) {
564             Thread currentThread = Thread.currentThread();
565             if (threadRef.get() == currentThread) {
566                 // The current Thread is the thread that belongs to the Stack, we can try to push the object now.
567                 pushNow(item);
568             } else {
569                 // The current Thread is not the one that belongs to the Stack
570                 // (or the Thread that belonged to the Stack was collected already), we need to signal that the push
571                 // happens later.
572                 pushLater(item, currentThread);
573             }
574         }
575 
576         private void pushNow(DefaultHandle<?> item) {
577             if ((item.recycleId | item.lastRecycledId) != 0) {
578                 throw new IllegalStateException("recycled already");
579             }
580             item.recycleId = item.lastRecycledId = OWN_THREAD_ID;
581 
582             int size = this.size;
583             if (size >= maxCapacity || dropHandle(item)) {
584                 // Hit the maximum capacity or should drop - drop the possibly youngest object.
585                 return;
586             }
587             if (size == elements.length) {
588                 elements = Arrays.copyOf(elements, min(size << 1, maxCapacity));
589             }
590 
591             elements[size] = item;
592             this.size = size + 1;
593         }
594 
595         private void pushLater(DefaultHandle<?> item, Thread thread) {
596             // we don't want to have a ref to the queue as the value in our weak map
597             // so we null it out; to ensure there are no races with restoring it later
598             // we impose a memory ordering here (no-op on x86)
599             Map<Stack<?>, WeakOrderQueue> delayedRecycled = DELAYED_RECYCLED.get();
600             WeakOrderQueue queue = delayedRecycled.get(this);
601             if (queue == null) {
602                 if (delayedRecycled.size() >= maxDelayedQueues) {
603                     // Add a dummy queue so we know we should drop the object
604                     delayedRecycled.put(this, WeakOrderQueue.DUMMY);
605                     return;
606                 }
607                 // Check if we already reached the maximum number of delayed queues and if we can allocate at all.
608                 if ((queue = WeakOrderQueue.allocate(this, thread)) == null) {
609                     // drop object
610                     return;
611                 }
612                 delayedRecycled.put(this, queue);
613             } else if (queue == WeakOrderQueue.DUMMY) {
614                 // drop object
615                 return;
616             }
617 
618             queue.add(item);
619         }
620 
621         boolean dropHandle(DefaultHandle<?> handle) {
622             if (!handle.hasBeenRecycled) {
623                 if ((++handleRecycleCount & ratioMask) != 0) {
624                     // Drop the object.
625                     return true;
626                 }
627                 handle.hasBeenRecycled = true;
628             }
629             return false;
630         }

632         DefaultHandle<T> newHandle() {
633             return new DefaultHandle<T>(this);
634         }
