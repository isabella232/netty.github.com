29  /**
30   * A lock-free alternate implementation of {@link java.util.concurrent.ConcurrentHashMap}
31   * with <strong>primitive long keys</strong>, better scaling properties and
32   * generally lower costs.  The use of {@code long} keys allows for faster
33   * compares and lower memory costs.  The Map provides identical correctness
34   * properties as ConcurrentHashMap.  All operations are non-blocking and
35   * multi-thread safe, including all update operations.  {@link
36   * NonBlockingHashMapLong} scales substatially better than {@link
37   * java.util.concurrent.ConcurrentHashMap} for high update rates, even with a large
38   * concurrency factor.  Scaling is linear up to 768 CPUs on a 768-CPU Azul
39   * box, even with 100% updates or 100% reads or any fraction in-between.
40   * Linear scaling up to all cpus has been observed on a 32-way Sun US2 box,
41   * 32-way Sun Niagra box, 8-way Intel box and a 4-way Power box.
42   *
43   * <p><strong>The main benefit of this class</strong> over using plain {@link
44   * io.netty.util.internal.shaded.org.jctools.maps.NonBlockingHashMap} with {@link Long} keys is
45   * that it avoids the auto-boxing and unboxing costs.  Since auto-boxing is
46   * <em>automatic</em>, it is easy to accidentally cause auto-boxing and negate
47   * the space and speed benefits.
48   *
49   * <p>This class obeys the same functional specification as {@link
50   * java.util.Hashtable}, and includes versions of methods corresponding to
51   * each method of <tt>Hashtable</tt>.  However, even though all operations are
52   * thread-safe, operations do <em>not</em> entail locking and there is
53   * <em>not</em> any support for locking the entire table in a way that
54   * prevents all access.  This class is fully interoperable with
55   * <tt>Hashtable</tt> in programs that rely on its thread safety but not on
56   * its synchronization details.
57   *
58   * <p> Operations (including <tt>put</tt>) generally do not block, so may
59   * overlap with other update operations (including other <tt>puts</tt> and
60   * <tt>removes</tt>).  Retrievals reflect the results of the most recently
61   * <em>completed</em> update operations holding upon their onset.  For
62   * aggregate operations such as <tt>putAll</tt>, concurrent retrievals may
63   * reflect insertion or removal of only some entries.  Similarly, Iterators
64   * and Enumerations return elements reflecting the state of the hash table at
65   * some point at or since the creation of the iterator/enumeration.  They do
66   * <em>not</em> throw {@link ConcurrentModificationException}.  However,
67   * iterators are designed to be used by only one thread at a time.
68   *
69   * <p> Very full tables, or tables with high reprobe rates may trigger an
70   * internal resize operation to move into a larger table.  Resizing is not
71   * terribly expensive, but it is not free either; during resize operations
72   * table throughput may drop somewhat.  All threads that visit the table
73   * during a resize will 'help' the resizing but will still be allowed to
74   * complete their operation before the resize is finished (i.e., a simple
75   * 'get' operation on a million-entry table undergoing resizing will not need
76   * to block until the entire million entries are copied).
77   *
78   * <p>This class and its views and iterators implement all of the
79   * <em>optional</em> methods of the {@link Map} and {@link Iterator}
80   * interfaces.
81   *
82   * <p> Like {@link Hashtable} but unlike {@link HashMap}, this class
83   * does <em>not</em> allow <tt>null</tt> to be used as a value.
84   *
85   *
86   * @since 1.5
87   * @author Cliff Click
88   * @param <TypeV> the type of mapped values
89   */
90

198   // Heuristic to decide if we have reprobed toooo many times.  Running over
199   // the reprobe limit on a 'get' call acts as a 'miss'; on a 'put' call it
200   // can trigger a table resize.  Several places must have exact agreement on
201   // what the reprobe_limit is, so we share it here.

203     return REPROBE_LIMIT + (len>>4);
204   }

303   private TypeV putIfMatch( long key, Object newVal, Object oldVal ) {
304     if (oldVal == null || newVal == null)  throw new NullPointerException();
305     if( key == NO_KEY ) {
306       Object curVal = _val_1;
307       if( oldVal == NO_MATCH_OLD || // Do we care about expected-Value at all?
308           curVal == oldVal ||       // No instant match already?
309           (oldVal == MATCH_ANY && curVal != TOMBSTONE) ||
310           oldVal.equals(curVal) ) { // Expensive equals check
311         if( !CAS(_val_1_offset,curVal,newVal) ) // One shot CAS update attempt
312           curVal = _val_1;                      // Failed; get failing witness
313       }
314       return curVal == TOMBSTONE ? null : (TypeV)curVal; // Return the last value present
315     }
316     final Object res = _chm.putIfMatch( key, newVal, oldVal );
317     assert !(res instanceof Prime);
318     assert res != null;
319     return res == TOMBSTONE ? null : (TypeV)res;
320   }

323   public void clear() {         // Smack a new empty table down
324     CHM newchm = new CHM(this,new ConcurrentAutoTable(),MIN_SIZE_LOG);
325     while( !CAS(_chm_offset,_chm,newchm) ) { /*Spin until the clear works*/}

384   // Help along an existing resize operation.  This is just a fast cut-out
385   // wrapper, to encourage inlining for the fast no-copy-in-progress case.  We
386   // always help the top-most table copy, even if there are nested table
387   // copies in progress.
388   private void help_copy( ) {
389     // Read the top-level CHM only once.  We'll try to help this copy along,
390     // even if it gets promoted out from under us (i.e., the copy completes
391     // and another KVS becomes the top-level copy).
392     CHM topchm = _chm;
393     if( topchm._newchm == null ) return; // No copy in-progress
394     topchm.help_copy_impl(false);
395   }

398   // --- CHM -----------------------------------------------------------------
399   // The control structure for the NonBlockingHashMapLong
400   private static final class CHM implements Serializable {
401     // Back-pointer to top-level structure
402     final NonBlockingHashMapLong _nbhml;
403 
404     // Size in active K,V pairs
405     private final ConcurrentAutoTable _size;
406     public int size () { return (int)_size.get(); }
407 
408     // ---
409     // These next 2 fields are used in the resizing heuristics, to judge when
410     // it is time to resize or copy the table.  Slots is a count of used-up
411     // key slots, and when it nears a large fraction of the table we probably
412     // end up reprobing too much.  Last-resize-milli is the time since the
413     // last resize; if we are running back-to-back resizes without growing
414     // (because there are only a few live keys but many slots full of dead
415     // keys) then we need a larger table to cut down on the churn.
416 
417     // Count of used slots, to tell when table is full of dead unusable slots
418     private final ConcurrentAutoTable _slots;
419     public int slots() { return (int)_slots.get(); }
420 
421     // ---
422     // New mappings, used during resizing.
423     // The 'next' CHM - created during a resize operation.  This represents
424     // the new table being copied from the old one.  It's the volatile
425     // variable that is read as we cross from one table to the next, to get
426     // the required memory orderings.  It monotonically transits from null to
427     // set (once).
428     volatile CHM _newchm;
429     private static final AtomicReferenceFieldUpdater<CHM,CHM> _newchmUpdater =
430       AtomicReferenceFieldUpdater.newUpdater(CHM.class,CHM.class, "_newchm");
431     // Set the _newchm field if we can.  AtomicUpdaters do not fail spuriously.
432     boolean CAS_newchm( CHM newchm ) {
433       return _newchmUpdater.compareAndSet(this,null,newchm);
434     }
435     // Sometimes many threads race to create a new very large table.  Only 1
436     // wins the race, but the losers all allocate a junk large table with
437     // hefty allocation costs.  Attempt to control the overkill here by
438     // throttling attempts to create a new table.  I cannot really block here
439     // (lest I lose the non-blocking property) but late-arriving threads can
440     // give the initial resizing thread a little time to allocate the initial
441     // new table.  The Right Long Term Fix here is to use array-lets and
442     // incrementally create the new very large array.  In C I'd make the array
443     // with malloc (which would mmap under the hood) which would only eat
444     // virtual-address and not real memory - and after Somebody wins then we
445     // could in parallel initialize the array.  Java does not allow
446     // un-initialized array creation (especially of ref arrays!).
447     volatile long _resizers;    // count of threads attempting an initial resize
448     private static final AtomicLongFieldUpdater<CHM> _resizerUpdater =
449       AtomicLongFieldUpdater.newUpdater(CHM.class, "_resizers");
450 
451     // --- key,val -------------------------------------------------------------
452     // Access K,V for a given idx
453     private boolean CAS_key( int idx, long   old, long   key ) {
454       return UNSAFE.compareAndSwapLong  ( _keys, rawIndex(_keys, idx), old, key );
455     }
456     private boolean CAS_val( int idx, Object old, Object val ) {
457       return UNSAFE.compareAndSwapObject( _vals, rawIndex(_vals, idx), old, val );
458     }

460     final long   [] _keys;
461     final Object [] _vals;

463     // Simple constructor
464     CHM( final NonBlockingHashMapLong nbhml, ConcurrentAutoTable size, final int logsize ) {
465       _nbhml = nbhml;
466       _size = size;
467       _slots= new ConcurrentAutoTable();
468       _keys = new long  [1<<logsize];
469       _vals = new Object[1<<logsize];

472     // --- print innards
473     private void print() {
474       for( int i=0; i<_keys.length; i++ ) {
475         long K = _keys[i];
476         if( K != NO_KEY )
477           print_impl(i,K,_vals[i]);
478       }
479       CHM newchm = _newchm;     // New table, if any
480       if( newchm != null ) {
481         System.out.println("----");
482         newchm.print();
483       }
484     }

486     // --- print only the live objects
487     private void print2( ) {
488       for( int i=0; i<_keys.length; i++ ) {
489         long K = _keys[i];
490         if( K != NO_KEY )       // key is sane
491           print2_impl(i,K,_vals[i]);
492       }
493       CHM newchm = _newchm;     // New table, if any
494       if( newchm != null ) {
495         System.out.println("----");
496         newchm.print2();
497       }

500     // --- get_impl ----------------------------------------------------------
501     // Never returns a Prime nor a Tombstone.
502     private Object get_impl ( final long key ) {
503       final int len     = _keys.length;
504       int idx = (int)(key & (len-1)); // First key hash
505 
506       // Main spin/reprobe loop, looking for a Key hit
507       int reprobe_cnt=0;
508       while( true ) {
509         final long   K = _keys[idx]; // Get key   before volatile read, could be NO_KEY
510         final Object V = _vals[idx]; // Get value before volatile read, could be null or Tombstone or Prime
511         if( K == NO_KEY ) return null; // A clear miss
512 
513         // Key-compare
514         if( key == K ) {
515           // Key hit!  Check for no table-copy-in-progress
516           if( !(V instanceof Prime) ) { // No copy?
517             if( V == TOMBSTONE) return null;
518             // We need a volatile-read between reading a newly inserted Value
519             // and returning the Value (so the user might end up reading the
520             // stale Value contents).
521             @SuppressWarnings("unused") final CHM newchm = _newchm; // VOLATILE READ before returning V
522             return V;
523           }
524           // Key hit - but slot is (possibly partially) copied to the new table.
525           // Finish the copy & retry in the new table.
526           return copy_slot_and_check(idx,key).get_impl(key); // Retry in the new table
527         }
528         // get and put must have the same key lookup logic!  But only 'put'
529         // needs to force a table-resize for a too-long key-reprobe sequence.
530         // Check for too-many-reprobes on get.
531         if( ++reprobe_cnt >= reprobe_limit(len) ) // too many probes
532           return _newchm == null // Table copy in progress?
533             ? null               // Nope!  A clear miss
534             : copy_slot_and_check(idx,key).get_impl(key); // Retry in the new table
535 
536         idx = (idx+1)&(len-1);    // Reprobe by 1!  (could now prefetch)
537       }
538     }
539 
540     // --- putIfMatch ---------------------------------------------------------
541     // Put, Remove, PutIfAbsent, etc.  Return the old value.  If the returned
542     // value is equal to expVal (or expVal is NO_MATCH_OLD) then the put can
543     // be assumed to work (although might have been immediately overwritten).
544     // Only the path through copy_slot passes in an expected value of null,
545     // and putIfMatch only returns a null if passed in an expected null.
546     private Object putIfMatch( final long key, final Object putval, final Object expVal ) {
547       assert putval != null;
548       assert !(putval instanceof Prime);
549       assert !(expVal instanceof Prime);
550       final int len      = _keys.length;
551       int idx = (int)(key & (len-1)); // The first key
552 
553       // ---
554       // Key-Claim stanza: spin till we can claim a Key (or force a resizing).
555       int reprobe_cnt=0;
556       long   K;
557       Object V;
558       while( true ) {           // Spin till we get a Key slot
559         V = _vals[idx];         // Get old value
560         K = _keys[idx];         // Get current key
561         if( K == NO_KEY ) {     // Slot is free?
562           // Found an empty Key slot - which means this Key has never been in
563           // this table.  No need to put a Tombstone - the Key is not here!
564           if( putval == TOMBSTONE ) return TOMBSTONE; // Not-now & never-been in this table
565           if( expVal == MATCH_ANY ) return TOMBSTONE; // Will not match, even after K inserts
566           // Claim the zero key-slot
567           if( CAS_key(idx, NO_KEY, key) ) { // Claim slot for Key
568             _slots.add(1);      // Raise key-slots-used count
569             break;              // Got it!
570           }
571           // CAS to claim the key-slot failed.
572           //
573           // This re-read of the Key points out an annoying short-coming of Java
574           // CAS.  Most hardware CAS's report back the existing value - so that
575           // if you fail you have a *witness* - the value which caused the CAS
576           // to fail.  The Java API turns this into a boolean destroying the
577           // witness.  Re-reading does not recover the witness because another
578           // thread can write over the memory after the CAS.  Hence we can be in
579           // the unfortunate situation of having a CAS fail *for cause* but
580           // having that cause removed by a later store.  This turns a
581           // non-spurious-failure CAS (such as Azul has) into one that can
582           // apparently spuriously fail - and we avoid apparent spurious failure
583           // by not allowing Keys to ever change.
584           K = _keys[idx];       // CAS failed, get updated value
585           assert K != NO_KEY ;  // If keys[idx] is NO_KEY, CAS shoulda worked
586         }
587         // Key slot was not null, there exists a Key here
588         if( K == key )
589           break;                // Got it!
590 
591         // get and put must have the same key lookup logic!  Lest 'get' give
592         // up looking too soon.
593         //topmap._reprobes.add(1);
594         if( ++reprobe_cnt >= reprobe_limit(len) ) {
595           // We simply must have a new table to do a 'put'.  At this point a
596           // 'get' will also go to the new table (if any).  We do not need
597           // to claim a key slot (indeed, we cannot find a free one to claim!).
598           final CHM newchm = resize();
599           if( expVal != null ) _nbhml.help_copy(); // help along an existing copy
600           return newchm.putIfMatch(key,putval,expVal);
601         }
602 
603         idx = (idx+1)&(len-1); // Reprobe!
604       } // End of spinning till we get a Key slot
605 
606       // ---
607       // Found the proper Key slot, now update the matching Value slot.  We
608       // never put a null, so Value slots monotonically move from null to
609       // not-null (deleted Values use Tombstone).  Thus if 'V' is null we
610       // fail this fast cutout and fall into the check for table-full.
611       if( putval == V ) return V; // Fast cutout for no-change
612 
613       // See if we want to move to a new table (to avoid high average re-probe
614       // counts).  We only check on the initial set of a Value from null to
615       // not-null (i.e., once per key-insert).
616       if( (V == null && tableFull(reprobe_cnt,len)) ||
617           // Or we found a Prime: resize is already in progress.  The resize
618           // call below will do a CAS on _newchm forcing the read.
619           V instanceof Prime) {
620         resize();               // Force the new table copy to start
621         return copy_slot_and_check(idx,expVal).putIfMatch(key,putval,expVal);
622       }
623 
624       // ---
625       // We are finally prepared to update the existing table
626       //assert !(V instanceof Prime); // always true, so IDE warnings if uncommented
627 
628       // Must match old, and we do not?  Then bail out now.  Note that either V
629       // or expVal might be TOMBSTONE.  Also V can be null, if we've never
630       // inserted a value before.  expVal can be null if we are called from
631       // copy_slot.
632 
633       if( expVal != NO_MATCH_OLD && // Do we care about expected-Value at all?
634           V != expVal &&            // No instant match already?
635           (expVal != MATCH_ANY || V == TOMBSTONE || V == null) &&
636           !(V==null && expVal == TOMBSTONE) &&    // Match on null/TOMBSTONE combo
637           (expVal == null || !expVal.equals(V)) ) // Expensive equals check at the last
638         return (V==null) ? TOMBSTONE : V;         // Do not update!
639 
640       // Actually change the Value in the Key,Value pair
641       if( CAS_val(idx, V, putval ) ) {
642         // CAS succeeded - we did the update!
643         // Both normal put's and table-copy calls putIfMatch, but table-copy
644         // does not (effectively) increase the number of live k/v pairs.
645         if( expVal != null ) {
646           // Adjust sizes - a striped counter
647           if(  (V == null || V == TOMBSTONE) && putval != TOMBSTONE ) _size.add( 1);
648           if( !(V == null || V == TOMBSTONE) && putval == TOMBSTONE ) _size.add(-1);
649         }
650       } else {                  // Else CAS failed
651         V = _vals[idx];         // Get new value
652         // If a Prime'd value got installed, we need to re-run the put on the
653         // new table.  Otherwise we lost the CAS to another racing put.
654         // Simply retry from the start.
655         if( V instanceof Prime )
656           return copy_slot_and_check(idx,expVal).putIfMatch(key,putval,expVal);
657       }
658       // Win or lose the CAS, we are done.  If we won then we know the update
659       // happened as expected.  If we lost, it means "we won but another thread
660       // immediately stomped our update with no chance of a reader reading".
661       return (V==null && expVal!=null) ? TOMBSTONE : V;
662     }

664     // --- tableFull ---------------------------------------------------------
665     // Heuristic to decide if this table is too full, and we should start a
666     // new table.  Note that if a 'get' call has reprobed too many times and
667     // decided the table must be full, then always the estimate_sum must be
668     // high and we must report the table is full.  If we do not, then we might
669     // end up deciding that the table is not full and inserting into the
670     // current table, while a 'get' has decided the same key cannot be in this
671     // table because of too many reprobes.  The invariant is:
672     //   slots.estimate_sum >= max_reprobe_cnt >= reprobe_limit(len)
673     private final boolean tableFull( int reprobe_cnt, int len ) {
674       return
675         // Do the cheap check first: we allow some number of reprobes always
676         reprobe_cnt >= REPROBE_LIMIT &&
677         (reprobe_cnt >= reprobe_limit(len) ||
678          // More expensive check: see if the table is > 1/2 full.
679          _slots.estimate_get() >= (len>>1));
680     }

682     // --- resize ------------------------------------------------------------
683     // Resizing after too many probes.  "How Big???" heuristics are here.
684     // Callers will (not this routine) will 'help_copy' any in-progress copy.
685     // Since this routine has a fast cutout for copy-already-started, callers
686     // MUST 'help_copy' lest we have a path which forever runs through
687     // 'resize' only to discover a copy-in-progress which never progresses.
688     private final CHM resize() {
689       // Check for resize already in progress, probably triggered by another thread
690       CHM newchm = _newchm;     // VOLATILE READ
691       if( newchm != null )      // See if resize is already in progress
692         return newchm;          // Use the new table already
693 
694       // No copy in-progress, so start one.  First up: compute new table size.
695       int oldlen = _keys.length; // Old count of K,V pairs allowed
696       int sz = size();          // Get current table count of active K,V pairs
697       int newsz = sz;           // First size estimate
698 
699       // Heuristic to determine new size.  We expect plenty of dead-slots-with-keys
700       // and we need some decent padding to avoid endless reprobing.
701       if( _nbhml._opt_for_space ) {
702         // This heuristic leads to a much denser table with a higher reprobe rate
703         if( sz >= (oldlen>>1) ) // If we are >50% full of keys then...
704           newsz = oldlen<<1;    // Double size
705       } else {
706         if( sz >= (oldlen>>2) ) { // If we are >25% full of keys then...
707           newsz = oldlen<<1;      // Double size
708           if( sz >= (oldlen>>1) ) // If we are >50% full of keys then...
709             newsz = oldlen<<2;    // Double double size
710         }
711       }
712 
713       // Last (re)size operation was very recent?  Then double again
714       // despite having few live keys; slows down resize operations
715       // for tables subject to a high key churn rate - but do not
716       // forever grow the table.  If there is a high key churn rate
717       // the table needs a steady state of rare same-size resize
718       // operations to clean out the dead keys.
719       long tm = System.currentTimeMillis();
720       if( newsz <= oldlen && // New table would shrink or hold steady?
721           tm <= _nbhml._last_resize_milli+10000)  // Recent resize (less than 10 sec ago)
722         newsz = oldlen<<1;      // Double the existing size
723 
724       // Do not shrink, ever.  If we hit this size once, assume we
725       // will again.
726       if( newsz < oldlen ) newsz = oldlen;
727 
728       // Convert to power-of-2
729       int log2;
730       for( log2=MIN_SIZE_LOG; (1<<log2) < newsz; log2++ ) ; // Compute log2 of size
731       long len = ((1L << log2) << 1) + 2;
732       // prevent integer overflow - limit of 2^31 elements in a Java array
733       // so here, 2^30 + 2 is the largest number of elements in the hash table
734       if ((int)len!=len) {
735         log2 = 30;
736         len = (1L << log2) + 2;
737         if (sz > ((len >> 2) + (len >> 1))) throw new RuntimeException("Table is full.");
738       }
739 
740       // Now limit the number of threads actually allocating memory to a
741       // handful - lest we have 750 threads all trying to allocate a giant
742       // resized array.
743       long r = _resizers;
744       while( !_resizerUpdater.compareAndSet(this,r,r+1) )
745         r = _resizers;
746       // Size calculation: 2 words (K+V) per table entry, plus a handful.  We
747       // guess at 64-bit pointers; 32-bit pointers screws up the size calc by
748       // 2x but does not screw up the heuristic very much.
749       long megs = ((((1L<<log2)<<1)+8)<<3/*word to bytes*/)>>20/*megs*/;
750       if( r >= 2 && megs > 0 ) { // Already 2 guys trying; wait and see
751         newchm = _newchm;        // Between dorking around, another thread did it
752         if( newchm != null )     // See if resize is already in progress
753           return newchm;         // Use the new table already
754         // We could use a wait with timeout, so we'll wakeup as soon as the new table
755         // is ready, or after the timeout in any case.
756         //synchronized( this ) { wait(8*megs); }         // Timeout - we always wakeup
757         // For now, sleep a tad and see if the 2 guys already trying to make
758         // the table actually get around to making it happen.
759         try { Thread.sleep(megs); } catch( Exception e ) { /*empty*/}
760       }
761       // Last check, since the 'new' below is expensive and there is a chance
762       // that another thread slipped in a new thread while we ran the heuristic.
763       newchm = _newchm;
764       if( newchm != null )      // See if resize is already in progress
765         return newchm;          // Use the new table already
766 
767       // New CHM - actually allocate the big arrays
768       newchm = new CHM(_nbhml,_size,log2);
769 
770       // Another check after the slow allocation
771       if( _newchm != null )     // See if resize is already in progress
772         return _newchm;         // Use the new table already
773 
774       // The new table must be CAS'd in so only 1 winner amongst duplicate
775       // racing resizing threads.  Extra CHM's will be GC'd.
776       if( CAS_newchm( newchm ) ) { // NOW a resize-is-in-progress!
777         //notifyAll();            // Wake up any sleepers
778         //long nano = System.nanoTime();
779         //System.out.println(" "+nano+" Resize from "+oldlen+" to "+(1<<log2)+" and had "+(_resizers-1)+" extras" );
780         //System.out.print("["+log2);
781       } else                    // CAS failed?
782         newchm = _newchm;       // Reread new table
783       return newchm;
784     }

787     // The next part of the table to copy.  It monotonically transits from zero
788     // to _keys.length.  Visitors to the table can claim 'work chunks' by
789     // CAS'ing this field up, then copying the indicated indices from the old
790     // table to the new table.  Workers are not required to finish any chunk;
791     // the counter simply wraps and work is copied duplicately until somebody
792     // somewhere completes the count.
793     volatile long _copyIdx = 0;
794     static private final AtomicLongFieldUpdater<CHM> _copyIdxUpdater =
795       AtomicLongFieldUpdater.newUpdater(CHM.class, "_copyIdx");
796 
797     // Work-done reporting.  Used to efficiently signal when we can move to
798     // the new table.  From 0 to len(oldkvs) refers to copying from the old
799     // table to the new.
800     volatile long _copyDone= 0;
801     static private final AtomicLongFieldUpdater<CHM> _copyDoneUpdater =
802       AtomicLongFieldUpdater.newUpdater(CHM.class, "_copyDone");
803 
804     // --- help_copy_impl ----------------------------------------------------
805     // Help along an existing resize operation.  We hope its the top-level
806     // copy (it was when we started) but this CHM might have been promoted out
807     // of the top position.
808     private final void help_copy_impl( final boolean copy_all ) {
809       final CHM newchm = _newchm;
810       assert newchm != null;    // Already checked by caller
811       int oldlen = _keys.length; // Total amount to copy
812       final int MIN_COPY_WORK = Math.min(oldlen,1024); // Limit per-thread work
813 
814       // ---
815       int panic_start = -1;
816       int copyidx=-9999;            // Fool javac to think it's initialized
817       while( _copyDone < oldlen ) { // Still needing to copy?
818         // Carve out a chunk of work.  The counter wraps around so every
819         // thread eventually tries to copy every slot repeatedly.
820 
821         // We "panic" if we have tried TWICE to copy every slot - and it still
822         // has not happened.  i.e., twice some thread somewhere claimed they
823         // would copy 'slot X' (by bumping _copyIdx) but they never claimed to
824         // have finished (by bumping _copyDone).  Our choices become limited:
825         // we can wait for the work-claimers to finish (and become a blocking
826         // algorithm) or do the copy work ourselves.  Tiny tables with huge
827         // thread counts trying to copy the table often 'panic'.
828         if( panic_start == -1 ) { // No panic?
829           copyidx = (int)_copyIdx;
830           while( !_copyIdxUpdater.compareAndSet(this,copyidx,copyidx+MIN_COPY_WORK) )
831             copyidx = (int)_copyIdx;     // Re-read
832           if( !(copyidx < (oldlen<<1)) ) // Panic!
833             panic_start = copyidx;       // Record where we started to panic-copy
834         }
835 
836         // We now know what to copy.  Try to copy.
837         int workdone = 0;
838         for( int i=0; i<MIN_COPY_WORK; i++ )
839           if( copy_slot((copyidx+i)&(oldlen-1)) ) // Made an oldtable slot go dead?
840             workdone++;         // Yes!
841         if( workdone > 0 )      // Report work-done occasionally
842           copy_check_and_promote( workdone );// See if we can promote
843         //for( int i=0; i<MIN_COPY_WORK; i++ )
844         //  if( copy_slot((copyidx+i)&(oldlen-1)) ) // Made an oldtable slot go dead?
845         //    copy_check_and_promote( 1 );// See if we can promote
846 
847         copyidx += MIN_COPY_WORK;
848         // Uncomment these next 2 lines to turn on incremental table-copy.
849         // Otherwise this thread continues to copy until it is all done.
850         if( !copy_all && panic_start == -1 ) // No panic?
851           return;               // Then done copying after doing MIN_COPY_WORK
852       }
853       // Extra promotion check, in case another thread finished all copying
854       // then got stalled before promoting.
855       copy_check_and_promote( 0 ); // See if we can promote
856     }

859     // --- copy_slot_and_check -----------------------------------------------
860     // Copy slot 'idx' from the old table to the new table.  If this thread
861     // confirmed the copy, update the counters and check for promotion.
862     //
863     // Returns the result of reading the volatile _newchm, mostly as a
864     // convenience to callers.  We come here with 1-shot copy requests
865     // typically because the caller has found a Prime, and has not yet read
866     // the _newchm volatile - which must have changed from null-to-not-null
867     // before any Prime appears.  So the caller needs to read the _newchm
868     // field to retry his operation in the new table, but probably has not
869     // read it yet.
870     private CHM copy_slot_and_check( int idx, Object should_help ) {
871       // We're only here because the caller saw a Prime, which implies a
872       // table-copy is in progress.
873       assert _newchm != null;
874       if( copy_slot(idx) )      // Copy the desired slot
875         copy_check_and_promote(1); // Record the slot copied
876       // Generically help along any copy (except if called recursively from a helper)
877       if( should_help != null ) _nbhml.help_copy();
878       return _newchm;
879     }

881     // --- copy_check_and_promote --------------------------------------------
882      private final void copy_check_and_promote( int workdone ) {
883       int oldlen = _keys.length;
884       // We made a slot unusable and so did some of the needed copy work
885       long copyDone = _copyDone;
886       assert (copyDone+workdone) <= oldlen;
887       if( workdone > 0 ) {
888         while( !_copyDoneUpdater.compareAndSet(this,copyDone,copyDone+workdone) ) {
889           copyDone = _copyDone; // Reload, retry
890           assert (copyDone+workdone) <= oldlen; 
891         }
892       }
893 
894       // Check for copy being ALL done, and promote.  Note that we might have
895       // nested in-progress copies and manage to finish a nested copy before
896       // finishing the top-level copy.  We only promote top-level copies.
897       if( copyDone+workdone == oldlen && // Ready to promote this table?
898           _nbhml._chm == this &&         // Looking at the top-level table?
899           // Attempt to promote
900           _nbhml.CAS(_chm_offset,this,_newchm) ) {
901         _nbhml._last_resize_milli = System.currentTimeMillis();  // Record resize time for next check
902       }
903     }

905     // --- copy_slot ---------------------------------------------------------
906     // Copy one K/V pair from oldkvs[i] to newkvs.  Returns true if we can
907     // confirm that we set an old-table slot to TOMBPRIME, and only returns after
908     // updating the new table.  We need an accurate confirmed-copy count so
909     // that we know when we can promote (if we promote the new table too soon,
910     // other threads may 'miss' on values not-yet-copied from the old table).
911     // We don't allow any direct updates on the new table, unless they first
912     // happened to the old table - so that any transition in the new table from
913     // null to not-null must have been from a copy_slot (or other old-table
914     // overwrite) and not from a thread directly writing in the new table.
915     private boolean copy_slot( int idx ) {
916       // Blindly set the key slot from NO_KEY to some key which hashes here,
917       // to eagerly stop fresh put's from inserting new values in the old
918       // table when the old table is mid-resize.  We don't need to act on the
919       // results here, because our correctness stems from box'ing the Value
920       // field.  Slamming the Key field is a minor speed optimization.
921       long key;
922       while( (key=_keys[idx]) == NO_KEY )
923         CAS_key(idx, NO_KEY, (idx+_keys.length)/*a non-zero key which hashes here*/);
924 
925       // ---
926       // Prevent new values from appearing in the old table.
927       // Box what we see in the old table, to prevent further updates.
928       Object oldval = _vals[idx]; // Read OLD table
929       while( !(oldval instanceof Prime) ) {
930         final Prime box = (oldval == null || oldval == TOMBSTONE) ? TOMBPRIME : new Prime(oldval);
931         if( CAS_val(idx,oldval,box) ) { // CAS down a box'd version of oldval
932           // If we made the Value slot hold a TOMBPRIME, then we both
933           // prevented further updates here but also the (absent) oldval is
934           // vaccuously available in the new table.  We return with true here:
935           // any thread looking for a value for this key can correctly go
936           // straight to the new table and skip looking in the old table.
937           if( box == TOMBPRIME )
938             return true;
939           // Otherwise we boxed something, but it still needs to be
940           // copied into the new table.
941           oldval = box;         // Record updated oldval
942           break;                // Break loop; oldval is now boxed by us
943         }
944         oldval = _vals[idx];    // Else try, try again
945       }
946       if( oldval == TOMBPRIME ) return false; // Copy already complete here!
947 
948       // ---
949       // Copy the value into the new table, but only if we overwrite a null.
950       // If another value is already in the new table, then somebody else
951       // wrote something there and that write is happens-after any value that
952       // appears in the old table.
953       Object old_unboxed = ((Prime)oldval)._V;
954       assert old_unboxed != TOMBSTONE;
955       _newchm.putIfMatch(key, old_unboxed, null);
956 
957       // ---
958       // Finally, now that any old value is exposed in the new table, we can
959       // forever hide the old-table value by slapping a TOMBPRIME down.  This
960       // will stop other threads from uselessly attempting to copy this slot
961       // (i.e., it's a speed optimization not a correctness issue).
962       while( oldval != TOMBPRIME && !CAS_val(idx,oldval,TOMBPRIME) )
963         oldval = _vals[idx];
964 
965       return oldval != TOMBPRIME; // True if we slammed the TOMBPRIME down
966     } // end copy_slot
967   } // End of CHM

970   // --- Snapshot ------------------------------------------------------------
971   // The main class for iterating over the NBHM.  It "snapshots" a clean
972   // view of the K/V array.
973   private class SnapshotV implements Iterator<TypeV>, Enumeration<TypeV> {
974     final CHM _sschm;
975     public SnapshotV() {
976       CHM topchm;
977       while( true ) {           // Verify no table-copy-in-progress
978         topchm = _chm;

999     public TypeV next() {

1001       // figure that out last go 'round lest 'hasNext' report true and
1002       // some other thread deleted the last value.  Instead, 'next'
1003       // spends all its effort finding the key that comes after the
1004       // 'next' key.
1005       if( _idx != -1 && _nextV == null ) throw new NoSuchElementException();

1010       // _nextV is the trigger: stop searching when it is != null

1012         _idx = 0;               // Setup for next phase of search
1013         _nextK = NO_KEY;
1014         if( (_nextV=get(_nextK)) != null ) return _prevV;
1015       }

1017         _nextK = key(_idx++); // Get a key that definitely is in the set (for the moment!)
1018         if( _nextK != NO_KEY && // Found something?
1019             (_nextV=get(_nextK)) != null )
1020           break;                // Got it!  _nextK is a valid Key
1021       }                         // Else keep scanning
1022       return _prevV;            // Return current value.
1023     }
