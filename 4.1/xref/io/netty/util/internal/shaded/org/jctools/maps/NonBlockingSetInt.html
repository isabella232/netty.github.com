26  /**
27   * A multi-threaded bit-vector set, implemented as an array of primitive
28   * {@code longs}.  All operations are non-blocking and multi-threaded safe.
29   * {@link #contains(int)} calls are roughly the same speed as a {load, mask}
30   * sequence.  {@link #add(int)} and {@link #remove(int)} calls are a tad more
31   * expensive than a {load, mask, store} sequence because they must use a CAS.
32   * The bit-vector is auto-sizing.
33   *
34   * <p><em>General note of caution:</em> The Set API allows the use of {@link Integer}
35   * with silent autoboxing - which can be very expensive if many calls are
36   * being made.  Since autoboxing is silent you may not be aware that this is
37   * going on.  The built-in API takes lower-case {@code ints} and is much more
38   * efficient.
39   *
40   * <p>Space: space is used in proportion to the largest element, as opposed to
41   * the number of elements (as is the case with hash-table based Set
42   * implementations).  Space is approximately (largest_element/8 + 64) bytes.
43   *
44   * The implementation is a simple bit-vector using CAS for update.
45   *
46   * @since 1.5
47   * @author Cliff Click
48   */
49  public class NonBlockingSetInt extends AbstractSet<Integer> implements Serializable {

156   private class iter implements Iterator<Integer> {
157     NBSI _nbsi2;
158     int _idx  = -1;
159     int _prev = -1;
160     iter() { _nbsi2 = _nbsi; advance(); }
161     public boolean hasNext() { return _idx != -2; }
162     private void advance() {     
163       while( true ) {
164         _idx++;                 // Next index
165         while( (_idx>>6) >= _nbsi2._bits.length ) { // Index out of range?
166           if( _nbsi2._new == null ) { // New table?
167             _idx = -2;          // No, so must be all done
168             return;             // 
169           }
170           _nbsi2 = _nbsi2._new; // Carry on, in the new table
171         }
172         if( _nbsi2.contains(_idx) ) return;
173       }
174     }
175     public Integer next() { 
176       if( _idx == -1 ) throw new NoSuchElementException();
177       _prev = _idx;
178       advance();
179       return _prev;
180     }
181     public void remove() { 
182       if( _prev == -1 ) throw new IllegalStateException();
183       _nbsi2.remove(_prev);
184       _prev = -1;
185     }
186   }

220     // --- Bits to allow Unsafe access to arrays
221     private static final int _Lbase  = UNSAFE.arrayBaseOffset(long[].class);
222     private static final int _Lscale = UNSAFE.arrayIndexScale(long[].class);
223     private static long rawIndex(final long[] ary, final int idx) {

228       return UNSAFE.compareAndSwapLong( _bits, rawIndex(_bits, idx), old, nnn );

232     // The New Table, only set once to non-zero during a resize.
233     // Must be atomically set.
234     private NBSI _new;

248     private transient final AtomicInteger _copyIdx;   // Used to count bits started copying
249     private transient final AtomicInteger _copyDone;  // Used to count words copied in a resize operation
250     private transient final int _sum_bits_length; // Sum of all nested _bits.lengths
251 
252     private static final long mask( int i ) { return 1L<<(i&63); }
253 
254     // I need 1 free bit out of 64 to allow for resize.  I do this by stealing
255     // the high order bit - but then I need to do something with adding element
256     // number 63 (and friends).  I could use a mod63 function but it's more
257     // efficient to handle the mod-64 case as an exception.
258     //
259     // Every 64th bit is put in it's own recursive bitvector.  If the low 6 bits
260     // are all set, we shift them off and recursively operate on the _nbsi64 set.
261     private final NBSI _nbsi64;
262     
263     private NBSI( int max_elem, ConcurrentAutoTable ctr, NonBlockingSetInt nonb ) { 
264       super(); 
265       _non_blocking_set_int = nonb;
266       _size = ctr;
267       _copyIdx  = ctr == null ? null : new AtomicInteger();
268       _copyDone = ctr == null ? null : new AtomicInteger();
269       // The main array of bits
270       _bits = new long[(int)(((long)max_elem+63)>>>6)];
271       // Every 64th bit is moved off to it's own subarray, so that the
272       // sign-bit is free for other purposes
273       _nbsi64 = ((max_elem+1)>>>6) == 0 ? null : new NBSI((max_elem+1)>>>6, null, null);
274       _sum_bits_length = _bits.length + (_nbsi64==null ? 0 : _nbsi64._sum_bits_length);
275     }

277     // Lower-case 'int' versions - no autoboxing, very fast.
278     // 'i' is known positive.
279     public boolean add( final int i ) {
280       // Check for out-of-range for the current size bit vector.
281       // If so we need to grow the bit vector.
282       if( (i>>6) >= _bits.length ) 
283         return install_larger_new_bits(i). // Install larger pile-o-bits (duh)
284           help_copy().add(i);              // Finally, add to the new table
285 
286       // Handle every 64th bit via using a nested array
287       NBSI nbsi = this;         // The bit array being added into
288       int j = i;                // The bit index being added
289       while( (j&63) == 63 ) {   // Bit 64? (low 6 bits are all set)
290         nbsi = nbsi._nbsi64;    // Recurse
291         j = j>>6;               // Strip off low 6 bits (all set)
292       }
293 
294       final long mask = mask(j);
295       long old;
296       do { 
297         old = nbsi._bits[j>>6]; // Read old bits
298         if( old < 0 )           // Not mutable?
299           // Not mutable: finish copy of word, and retry on copied word
300           return help_copy_impl(i).help_copy().add(i);
301         if( (old & mask) != 0 ) return false; // Bit is already set?
302       } while( !nbsi.CAS( j>>6, old, old | mask ) );
303       _size.add(1);
304       return true;
305     }
306 
307     public boolean remove( final int i ) {
308       if( (i>>6) >= _bits.length ) // Out of bounds?  Not in this array!
309         return _new != null && help_copy().remove(i);
310 
311       // Handle every 64th bit via using a nested array
312       NBSI nbsi = this;         // The bit array being added into
313       int j = i;                // The bit index being added
314       while( (j&63) == 63 ) {   // Bit 64? (low 6 bits are all set)
315         nbsi = nbsi._nbsi64;    // Recurse
316         j = j>>6;               // Strip off low 6 bits (all set)
317       }
318 
319       final long mask = mask(j);
320       long old;
321       do { 
322         old = nbsi._bits[j>>6]; // Read old bits
323         if( old < 0 )           // Not mutable?
324           // Not mutable: finish copy of word, and retry on copied word
325           return help_copy_impl(i).help_copy().remove(i);
326         if( (old & mask) == 0 ) return false; // Bit is already clear?
327       } while( !nbsi.CAS( j>>6, old, old & ~mask ) );
328       _size.add(-1);
329       return true;
330     }

332     public boolean contains( final int i ) { 
333       if( (i>>6) >= _bits.length ) // Out of bounds?  Not in this array!
334         return _new != null && help_copy().contains(i);
335 
336       // Handle every 64th bit via using a nested array
337       NBSI nbsi = this;         // The bit array being added into
338       int j = i;                // The bit index being added
339       while( (j&63) == 63 ) {   // Bit 64? (low 6 bits are all set)
340         nbsi = nbsi._nbsi64;    // Recurse
341         j = j>>6;               // Strip off low 6 bits (all set)
342       }
343 
344       final long mask = mask(j);
345       long old = nbsi._bits[j>>6]; // Read old bits
346       if( old < 0 )             // Not mutable?
347         // Not mutable: finish copy of word, and retry on copied word
348         return help_copy_impl(i).help_copy().contains(i);
349       // Yes mutable: test & return bit
350       return (old & mask) != 0; 
351     }

355     // Must grow the current array to hold an element of size i
356     private NBSI install_larger_new_bits( final int i ) {
357       if( _new == null ) {
358         // Grow by powers of 2, to avoid minor grow-by-1's.
359         // Note: must grow by exact powers-of-2 or the by-64-bit trick doesn't work right
360         int sz = (_bits.length<<6)<<1;
361         // CAS to install a new larger size.  Did it work?  Did it fail?  We
362         // don't know and don't care.  Only One can be installed, so if
363         // another thread installed a too-small size, we can't help it - we
364         // must simply install our new larger size as a nested-resize table.
365         CAS_new(new NBSI(sz, _size, _non_blocking_set_int));
366       }
367       // Return self for 'fluid' programming style
368       return this;
369     }

371     // Help any top-level NBSI to copy until completed.
372     // Always return the _new version of *this* NBSI, in case we're nested.
373     private NBSI help_copy() {
374       // Pick some words to help with - but only help copy the top-level NBSI.
375       // Nested NBSI waits until the top is done before we start helping.
376       NBSI top_nbsi = _non_blocking_set_int._nbsi;
377       final int HELP = 8;       // Tuning number: how much copy pain are we willing to inflict?
378       // We "help" by forcing individual bit indices to copy.  However, bits
379       // come in lumps of 64 per word, so we just advance the bit counter by 64's.
380       int idx = top_nbsi._copyIdx.getAndAdd(64*HELP);
381       for( int i=0; i<HELP; i++ ) {
382         int j = idx+i*64;
383         j %= (top_nbsi._bits.length<<6); // Limit, wrap to array size; means we retry indices
384         top_nbsi.help_copy_impl(j   );
385         top_nbsi.help_copy_impl(j+63); // Also force the nested-by-64 bit
386       }
387 
388       // Top level guy ready to promote?
389       // Note: WE may not be the top-level guy!
390       if( top_nbsi._copyDone.get() == top_nbsi._sum_bits_length )
391         // One shot CAS to promote - it may fail since we are racing; others
392         // may promote as well
393         if( _non_blocking_set_int.CAS_nbsi( top_nbsi, top_nbsi._new ) ) {
394           //System.out.println("Promote at top level to size "+(_non_blocking_set_int._nbsi._bits.length<<6));
395         }
396 
397       // Return the new bitvector for 'fluid' programming style
398       return _new;
399     }

401     // Help copy this one word.  State Machine.  
402     // (1) If not "made immutable" in the old array, set the sign bit to make
403     //     it immutable.
404     // (2) If non-zero in old array & zero in new, CAS new from 0 to copy-of-old
405     // (3) If non-zero in old array & non-zero in new, CAS old to zero
406     // (4) Zero in old, new is valid
407     // At this point, old should be immutable-zero & new has a copy of bits
408     private NBSI help_copy_impl( int i ) {
409       // Handle every 64th bit via using a nested array
410       NBSI old = this;          // The bit array being copied from
411       NBSI nnn = _new;          // The bit array being copied to
412       if( nnn == null ) return this; // Promoted already
413       int j = i;                // The bit index being added
414       while( (j&63) == 63 ) {   // Bit 64? (low 6 bits are all set)
415         old = old._nbsi64;      // Recurse
416         nnn = nnn._nbsi64;      // Recurse
417         j = j>>6;               // Strip off low 6 bits (all set)
418       }
419 
420       // Transit from state 1: word is not immutable yet
421       // Immutable is in bit 63, the sign bit.
422       long bits = old._bits[j>>6];
423       while( bits >= 0 ) {      // Still in state (1)?
424         long oldbits = bits;
425         bits |= mask(63);       // Target state of bits: sign-bit means immutable
426         if( old.CAS( j>>6, oldbits, bits ) ) {
427           if( oldbits == 0 ) _copyDone.addAndGet(1);
428           break;                // Success - old array word is now immutable
429         } 
430         bits = old._bits[j>>6]; // Retry if CAS failed
431       }
432 
433       // Transit from state 2: non-zero in old and zero in new
434       if( bits != mask(63) ) {  // Non-zero in old?
435         long new_bits = nnn._bits[j>>6];
436         if( new_bits == 0 ) {   // New array is still zero
437           new_bits = bits & ~mask(63); // Desired new value: a mutable copy of bits
438           // One-shot CAS attempt, no loop, from 0 to non-zero.
439           // If it fails, somebody else did the copy for us
440           if( !nnn.CAS( j>>6, 0, new_bits ) )
441             new_bits = nnn._bits[j>>6]; // Since it failed, get the new value
442           assert new_bits != 0;
443         }
444 
445         // Transit from state 3: non-zero in old and non-zero in new
446         // One-shot CAS attempt, no loop, from non-zero to 0 (but immutable)
447         if( old.CAS( j>>6, bits, mask(63) ) )
448           _copyDone.addAndGet(1); // One more word finished copying
449       }
450       
451       // Now in state 4: zero (and immutable) in old
452 
453       // Return the self bitvector for 'fluid' programming style
454       return this;
455     }
