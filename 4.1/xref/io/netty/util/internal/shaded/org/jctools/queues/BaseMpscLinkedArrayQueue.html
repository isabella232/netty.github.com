126 /**
127  * An MPSC array queue which starts at <i>initialCapacity</i> and grows to <i>maxCapacity</i> in linked chunks
128  * of the initial size. The queue grows only when the current buffer is full and elements are not copied on
129  * resize, instead a link to the new buffer is stored in the old buffer for the consumer to follow.
130  */
131 abstract class BaseMpscLinkedArrayQueue<E> extends BaseMpscLinkedArrayQueueColdProducerFields<E>
132     implements MessagePassingQueue<E>, QueueProgressIndicators

134     // No post padding here, subclasses must add
135     private static final Object JUMP = new Object();
136     private static final Object BUFFER_CONSUMED = new Object();
137     private static final int CONTINUE_TO_P_INDEX_CAS = 0;
138     private static final int RETRY = 1;
139     private static final int QUEUE_FULL = 2;
140     private static final int QUEUE_RESIZE = 3;

143     /**
144      * @param initialCapacity the queue initial capacity. If chunk size is fixed this will be the chunk size.
145      *                        Must be 2 or more.
146      */
147     public BaseMpscLinkedArrayQueue(final int initialCapacity)
148     {
149         RangeUtil.checkGreaterThanOrEqual(initialCapacity, 2, "initialCapacity");
150 
151         int p2capacity = Pow2.roundToPowerOfTwo(initialCapacity);
152         // leave lower bit of mask clear
153         long mask = (p2capacity - 1) << 1;
154         // need extra element to point at next array
155         E[] buffer = allocateRefArray(p2capacity + 1);
156         producerBuffer = buffer;
157         producerMask = mask;
158         consumerBuffer = buffer;
159         consumerMask = mask;
160         soProducerLimit(mask); // we know it's all empty to start with
161     }

163     @Override
164     public int size()
165     {
166         // NOTE: because indices are on even numbers we cannot use the size util.
167 
168         /*
169          * It is possible for a thread to be interrupted or reschedule between the read of the producer and
170          * consumer indices, therefore protection is required to ensure size is within valid range. In the
171          * event of concurrent polls/offers to this method the size is OVER estimated as we read consumer
172          * index BEFORE the producer index.
173          */
174         long after = lvConsumerIndex();
175         long size;
176         while (true)
177         {
178             final long before = after;
179             final long currentProducerIndex = lvProducerIndex();
180             after = lvConsumerIndex();
181             if (before == after)
182             {
183                 size = ((currentProducerIndex - after) >> 1);
184                 break;
185             }
186         }
187         // Long overflow is impossible, so size is always positive. Integer overflow is possible for the unbounded
188         // indexed queues.
189         if (size > Integer.MAX_VALUE)
190         {
191             return Integer.MAX_VALUE;
192         }
193         else
194         {
195             return (int) size;
196         }
197     }

199     @Override
200     public boolean isEmpty()
201     {
202         // Order matters!
203         // Loading consumer before producer allows for producer increments after consumer index is read.
204         // This ensures this method is conservative in it's estimate. Note that as this is an MPMC there is
205         // nothing we can do to make this an exact method.
206         return (this.lvConsumerIndex() == this.lvProducerIndex());

215     @Override
216     public boolean offer(final E e)
217     {
218         if (null == e)
219         {
220             throw new NullPointerException();
221         }
222 
223         long mask;
224         E[] buffer;
225         long pIndex;
226 
227         while (true)
228         {
229             long producerLimit = lvProducerLimit();
230             pIndex = lvProducerIndex();
231             // lower bit is indicative of resize, if we see it we spin until it's cleared
232             if ((pIndex & 1) == 1)
233             {
234                 continue;
235             }
236             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1)
237 
238             // mask/buffer may get changed by resizing -> only use for array access after successful CAS.
239             mask = this.producerMask;
240             buffer = this.producerBuffer;
241             // a successful CAS ties the ordering, lv(pIndex) - [mask/buffer] -> cas(pIndex)
242 
243             // assumption behind this optimization is that queue is almost always empty or near empty
244             if (producerLimit <= pIndex)
245             {
246                 int result = offerSlowPath(mask, pIndex, producerLimit);
247                 switch (result)
248                 {
249                     case CONTINUE_TO_P_INDEX_CAS:
250                         break;
251                     case RETRY:
252                         continue;
253                     case QUEUE_FULL:
254                         return false;
255                     case QUEUE_RESIZE:
256                         resize(mask, buffer, pIndex, e, null);
257                         return true;
258                 }
259             }
260 
261             if (casProducerIndex(pIndex, pIndex + 2))
262             {
263                 break;
264             }
265         }
266         // INDEX visible before ELEMENT
267         final long offset = modifiedCalcCircularRefElementOffset(pIndex, mask);
268         soRefElement(buffer, offset, e); // release element e
269         return true;
270     }

278     @Override
279     public E poll()
280     {
281         final E[] buffer = consumerBuffer;
282         final long index = lpConsumerIndex();
283         final long mask = consumerMask;
284 
285         final long offset = modifiedCalcCircularRefElementOffset(index, mask);
286         Object e = lvRefElement(buffer, offset);
287         if (e == null)
288         {
289             if (index != lvProducerIndex())
290             {
291                 // poll() == null iff queue is empty, null element is not strong enough indicator, so we must
292                 // check the producer index. If the queue is indeed not empty we spin until element is
293                 // visible.
294                 do
295                 {
296                     e = lvRefElement(buffer, offset);
297                 }
298                 while (e == null);
299             }
300             else
301             {
302                 return null;
303             }
304         }
305 
306         if (e == JUMP)
307         {
308             final E[] nextBuffer = nextBuffer(buffer, mask);
309             return newBufferPoll(nextBuffer, index);
310         }
311 
312         soRefElement(buffer, offset, null); // release element null
313         soConsumerIndex(index + 2); // release cIndex
314         return (E) e;
315     }

349     /**
350      * We do not inline resize into this method because we do not resize on fill.
351      */
352     private int offerSlowPath(long mask, long pIndex, long producerLimit)
353     {
354         final long cIndex = lvConsumerIndex();
355         long bufferCapacity = getCurrentBufferCapacity(mask);
356 
357         if (cIndex + bufferCapacity > pIndex)
358         {
359             if (!casProducerLimit(producerLimit, cIndex + bufferCapacity))
360             {
361                 // retry from top
362                 return RETRY;
363             }
364             else
365             {
366                 // continue to pIndex CAS
367                 return CONTINUE_TO_P_INDEX_CAS;
368             }
369         }
370         // full and cannot grow
371         else if (availableInQueue(pIndex, cIndex) <= 0)
372         {
373             // offer should return false;
374             return QUEUE_FULL;
375         }
376         // grab index for resize -> set lower bit
377         else if (casProducerIndex(pIndex, pIndex + 1))
378         {
379             // trigger a resize
380             return QUEUE_RESIZE;
381         }
382         else
383         {
384             // failed resize attempt, retry from top
385             return RETRY;
386         }
387     }

396     {

398         final E[] nextBuffer = (E[]) lvRefElement(buffer, offset);

498     {
499         long result = 0;// result is a long because we want to have a safepoint check at regular intervals
500         final int capacity = capacity();
501         do
502         {
503             final int filled = fill(s, PortableJvmInfo.RECOMENDED_OFFER_BATCH);
504             if (filled == 0)

514     @Override
515     public int fill(Supplier<E> s, int limit)
516     {
517         if (null == s)
518             throw new IllegalArgumentException("supplier is null");
519         if (limit < 0)
520             throw new IllegalArgumentException("limit is negative:" + limit);
521         if (limit == 0)
522             return 0;
523 
524         long mask;
525         E[] buffer;
526         long pIndex;
527         int claimedSlots;
528         while (true)
529         {
530             long producerLimit = lvProducerLimit();
531             pIndex = lvProducerIndex();
532             // lower bit is indicative of resize, if we see it we spin until it's cleared
533             if ((pIndex & 1) == 1)
534             {
535                 continue;
536             }
537             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1)
538 
539             // NOTE: mask/buffer may get changed by resizing -> only use for array access after successful CAS.
540             // Only by virtue offloading them between the lvProducerIndex and a successful casProducerIndex are they
541             // safe to use.
542             mask = this.producerMask;
543             buffer = this.producerBuffer;
544             // a successful CAS ties the ordering, lv(pIndex) -> [mask/buffer] -> cas(pIndex)
545 
546             // we want 'limit' slots, but will settle for whatever is visible to 'producerLimit'
547             long batchIndex = Math.min(producerLimit, pIndex + 2l * limit); //  -> producerLimit >= batchIndex
548 
549             if (pIndex >= producerLimit)
550             {
551                 int result = offerSlowPath(mask, pIndex, producerLimit);
552                 switch (result)
553                 {
554                     case CONTINUE_TO_P_INDEX_CAS:
555                         // offer slow path verifies only one slot ahead, we cannot rely on indication here
556                     case RETRY:
557                         continue;
558                     case QUEUE_FULL:
559                         return 0;
560                     case QUEUE_RESIZE:
561                         resize(mask, buffer, pIndex, null, s);
562                         return 1;
563                 }
564             }
565 
566             // claim limit slots at once
567             if (casProducerIndex(pIndex, batchIndex))
568             {
569                 claimedSlots = (int) ((batchIndex - pIndex) / 2);
570                 break;
571             }
572         }
573 
574         for (int i = 0; i < claimedSlots; i++)
575         {
576             final long offset = modifiedCalcCircularRefElementOffset(pIndex + 2l * i, mask);
577             soRefElement(buffer, offset, s.get());
578         }
579         return claimedSlots;
580     }

620     private static class WeakIterator<E> implements Iterator<E>
621     {
622         private final long pIndex;
623         private long nextIndex;
624         private E nextElement;
625         private E[] currentBuffer;
626         private int mask;
627 
628         WeakIterator(E[] currentBuffer, long cIndex, long pIndex)
629         {
630             this.pIndex = pIndex >> 1;
631             this.nextIndex = cIndex >> 1;
632             setBuffer(currentBuffer);
633             nextElement = getNext();

651             if (e == null)

653                 throw new NoSuchElementException();
654             }
655             nextElement = getNext();
656             return e;
657         }

659         private void setBuffer(E[] buffer)
660         {
661             this.currentBuffer = buffer;
662             this.mask = length(buffer) - 2;
663         }

665         private E getNext()
666         {
667             while (nextIndex < pIndex)
668             {
669                 long index = nextIndex++;
670                 E e = lvRefElement(currentBuffer, calcCircularRefElementOffset(index, mask));
671                 // skip removed/not yet visible elements
672                 if (e == null)
673                 {
674                     continue;
675                 }
676 
677                 // not null && not JUMP -> found next element
678                 if (e != JUMP)
679                 {
680                     return e;
681                 }
682 
683                 // need to jump to the next buffer
684                 int nextBufferIndex = mask + 1;
685                 Object nextBuffer = lvRefElement(currentBuffer,
686                                               calcRefElementOffset(nextBufferIndex));
687 
688                 if (nextBuffer == BUFFER_CONSUMED || nextBuffer == null)
689                 {
690                     // Consumer may have passed us, or the next buffer is not visible yet: drop out early
691                     return null;
692                 }
693 
694                 setBuffer((E[]) nextBuffer);
695                 // now with the new array retry the load, it can't be a JUMP, but we need to repeat same index
696                 e = lvRefElement(currentBuffer, calcCircularRefElementOffset(index, mask));
697                 // skip removed/not yet visible elements
698                 if (e == null)
699                 {
700                     continue;
701                 }
702                 else
703                 {
704                     return e;
705                 }
706 
707             }
708             return null;
709         }

712     private void resize(long oldMask, E[] oldBuffer, long pIndex, E e, Supplier<E> s)
713     {
714         assert (e != null && s == null) || (e == null || s != null);
715         int newBufferLength = getNextBufferSize(oldBuffer);
716         final E[] newBuffer;
717         try
718         {
719             newBuffer = allocateRefArray(newBufferLength);
720         }
721         catch (OutOfMemoryError oom)
722         {
723             assert lvProducerIndex() == pIndex + 1;
724             soProducerIndex(pIndex);
725             throw oom;
726         }
727 
728         producerBuffer = newBuffer;
729         final int newMask = (newBufferLength - 2) << 1;
730         producerMask = newMask;
731 
732         final long offsetInOld = modifiedCalcCircularRefElementOffset(pIndex, oldMask);
733         final long offsetInNew = modifiedCalcCircularRefElementOffset(pIndex, newMask);
734 
735         soRefElement(newBuffer, offsetInNew, e == null ? s.get() : e);// element in new array
736         soRefElement(oldBuffer, nextArrayOffset(oldMask), newBuffer);// buffer linked
737 
738         // ASSERT code
739         final long cIndex = lvConsumerIndex();
740         final long availableInQueue = availableInQueue(pIndex, cIndex);
741         RangeUtil.checkPositive(availableInQueue, "availableInQueue");
742 
743         // Invalidate racing CASs
744         // We never set the limit beyond the bounds of a buffer
745         soProducerLimit(pIndex + Math.min(newMask, availableInQueue));
746 
747         // make resize visible to the other producers
748         soProducerIndex(pIndex + 2);
749 
750         // INDEX visible before ELEMENT, consistent with consumer expectation
751 
752         // make resize visible to consumer
753         soRefElement(oldBuffer, offsetInOld, JUMP);
754     }
