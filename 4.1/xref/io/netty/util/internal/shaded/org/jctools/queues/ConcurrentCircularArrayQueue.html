29  /**
30   * Common functionality for array backed queues. The class is pre-padded and the array is padded on either side to help
31   * with False Sharing prevention. It is expected that subclasses handle post padding.
32   *
33   * @param <E>
34   * @author nitsanw
35   */
36  public abstract class ConcurrentCircularArrayQueue<E> extends ConcurrentCircularArrayQueueL0Pad<E>
37  {
38      protected final long mask;
39      protected final E[] buffer;
40  
41      public ConcurrentCircularArrayQueue(int capacity)
42      {
43          int actualCapacity = Pow2.roundToPowerOfTwo(capacity);
44          mask = actualCapacity - 1;
45          buffer = CircularArrayOffsetCalculator.allocate(actualCapacity);
46      }
47  
48      /**
49       * @param index desirable element index
50       * @param mask (length - 1)
51       * @return the offset in bytes within the array for a given index.
52       */
53      protected static long calcElementOffset(long index, long mask)
54      {
55          return CircularArrayOffsetCalculator.calcElementOffset(index, mask);
56      }
57  
58      /**
59       * @param index desirable element index
60       * @return the offset in bytes within the array for a given index.
61       */
62      protected final long calcElementOffset(long index)
63      {
64          return calcElementOffset(index, mask);
65      }
66  
67      @Override
68      public Iterator<E> iterator()
69      {
70          throw new UnsupportedOperationException();
71      }
72  
73      @Override
74      public final int size()
75      {
76          return IndexedQueueSizeUtil.size(this);
77      }
78  
79      @Override
80      public final boolean isEmpty()
81      {
82          return IndexedQueueSizeUtil.isEmpty(this);
83      }
84  
85      @Override
86      public String toString()
87      {
88          return this.getClass().getName();
89      }
90  
91      @Override
92      public void clear()
93      {
94          while (poll() != null)
95          {
96              // if you stare into the void
97          }
98      }
99  
100     @Override
101     public int capacity()
102     {
103         return (int) (mask + 1);
104     }
105 
106     @Override
107     public final long currentProducerIndex()
108     {
109         return lvProducerIndex();
110     }
111 
112     @Override
113     public final long currentConsumerIndex()
114     {
115         return lvConsumerIndex();
116     }
