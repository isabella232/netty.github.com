229     public MpscBlockingConsumerArrayQueue(final int capacity)
230     {
231         // leave lower bit of mask clear
232         super((Pow2.roundToPowerOfTwo(capacity) - 1) << 1, (E[]) allocateRefArray(Pow2.roundToPowerOfTwo(capacity)));
233 
234         RangeUtil.checkGreaterThanOrEqual(capacity, 1, "capacity");
235         soProducerLimit((Pow2.roundToPowerOfTwo(capacity) - 1) << 1); // we know it's all empty to start with
236     }

244     @Override
245     public final int size()
246     {
247         // NOTE: because indices are on even numbers we cannot use the size util.
248 
249         /*
250          * It is possible for a thread to be interrupted or reschedule between the read of the producer and
251          * consumer indices, therefore protection is required to ensure size is within valid range. In the
252          * event of concurrent polls/offers to this method the size is OVER estimated as we read consumer
253          * index BEFORE the producer index.
254          */
255         long after = lvConsumerIndex();
256         long size;
257         while (true)
258         {
259             final long before = after;
260             final long currentProducerIndex = lvProducerIndex();
261             after = lvConsumerIndex();
262             if (before == after)
263             {
264                 size = ((currentProducerIndex - after) >> 1);
265                 break;
266             }
267         }
268         // Long overflow is impossible, so size is always positive. Integer overflow is possible for the unbounded
269         // indexed queues.
270         if (size > Integer.MAX_VALUE)
271         {
272             return Integer.MAX_VALUE;
273         }
274         else
275         {
276             return (int) size;
277         }
278     }

280     @Override

282     {
283         // Order matters!
284         // Loading consumer before producer allows for producer increments after consumer index is read.
285         // This ensures this method is conservative in it's estimate. Note that as this is an MPMC there is
286         // nothing we can do to make this an exact method.
287         return ((this.lvConsumerIndex()/2) == (this.lvProducerIndex()/2));
288     }

296     /**
297      * {@link #offer} if {@link #size()} is less than threshold.
298      *
299      * @param e         the object to offer onto the queue, not null
300      * @param threshold the maximum allowable size
301      * @return true if the offer is successful, false if queue size exceeds threshold
302      * @since 3.0.1
303      */
304     public boolean offerIfBelowThreshold(final E e, int threshold)
305     {
306         if (null == e)
307         {
308             throw new NullPointerException();
309         }
310 
311         final long mask = this.producerMask;
312         final long capacity = mask + 2;
313         threshold = threshold << 1;
314         final E[] buffer = this.producerBuffer;
315         long pIndex;
316         while (true)
317         {
318             pIndex = lvProducerIndex();
319             // lower bit is indicative of blocked consumer
320             if ((pIndex & 1) == 1)
321             {
322                 if (offerAndWakeup(buffer, mask, pIndex, e)) {
323                     return true;
324                 }
325                 continue;
326             }
327             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1), consumer is awake
328             final long producerLimit = lvProducerLimit();
329 
330             // Use producer limit to save a read of the more rapidly mutated consumer index.
331             // Assumption: queue is usually empty or near empty
332 
333             // available is also << 1
334             final long available = producerLimit - pIndex;
335             // sizeEstimate <= size
336             final long sizeEstimate = capacity - available;
337 
338             if (sizeEstimate >= threshold ||
339                 // producerLimit check allows for threshold >= capacity
340                 producerLimit <= pIndex)
341             {
342                 if (!recalculateProducerLimit(pIndex, producerLimit, lvConsumerIndex(), capacity, threshold))
343                 {
344                     return false;
345                 }
346             }
347 
348             // Claim the index
349             if (casProducerIndex(pIndex, pIndex + 2))
350             {
351                 break;
352             }
353         }
354         final long offset = modifiedCalcCircularRefElementOffset(pIndex, mask);
355         // INDEX visible before ELEMENT
356         soRefElement(buffer, offset, e); // release element e
357         return true;
358     }

360     @Override
361     public boolean offer(final E e)
362     {
363         if (null == e)
364         {
365             throw new NullPointerException();
366         }
367 
368         final long mask = this.producerMask;
369         final E[] buffer = this.producerBuffer;
370         long pIndex;
371         while (true)
372         {
373             pIndex = lvProducerIndex();
374             // lower bit is indicative of blocked consumer
375             if ((pIndex & 1) == 1)
376             {
377                 if (offerAndWakeup(buffer, mask, pIndex, e))
378                     return true;
379                 continue;
380             }
381             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1), consumer is awake
382             final long producerLimit = lvProducerLimit();
383 
384             // Use producer limit to save a read of the more rapidly mutated consumer index.
385             // Assumption: queue is usually empty or near empty
386             if (producerLimit <= pIndex)
387             {
388                 if (!recalculateProducerLimit(mask, pIndex, producerLimit))
389                 {
390                     return false;
391                 }
392             }
393 
394             // Claim the index
395             if (casProducerIndex(pIndex, pIndex + 2))
396             {
397                 break;
398             }
399         }
400         final long offset = modifiedCalcCircularRefElementOffset(pIndex, mask);
401         // INDEX visible before ELEMENT
402         soRefElement(buffer, offset, e); // release element e
403         return true;
404     }

421     private boolean offerAndWakeup(E[] buffer, long mask, long pIndex, E e)
422     {
423         final long offset = modifiedCalcCircularRefElementOffset(pIndex, mask);
424         final Thread consumerThread = lvBlocked();
425 
426         // We could see a null here through a race with the consumer not yet storing the reference, or through a race
427         // with another producer. Just retry.
428         if (consumerThread == null)
429         {
430             return false;
431         }
432 
433         // Claim the slot and the responsibility of unparking
434         if(!casProducerIndex(pIndex, pIndex + 1))
435         {
436             return false;
437         }
438 
439         soRefElement(buffer, offset, e);
440         releaseParkedConsumer(consumerThread);
441         return true;
442     }

444     private boolean recalculateProducerLimit(long mask, long pIndex, long producerLimit)
445     {
446         return recalculateProducerLimit(pIndex, producerLimit, lvConsumerIndex(), mask + 2, mask + 2);
447     }

449     private boolean recalculateProducerLimit(long pIndex, long producerLimit, long cIndex, long bufferCapacity, long threshold)
450     {
451         // try to update the limit with our new found knowledge on cIndex
452         if (cIndex + bufferCapacity > pIndex)
453         {
454             casProducerLimit(producerLimit, cIndex + bufferCapacity);
455         }
456         // full and cannot grow, or hit threshold
457         long size = pIndex - cIndex;
458         return size < threshold && size < bufferCapacity;
459     }

461     private void wakeupConsumer()
462     {
463         Thread consumerThread;
464         do
465         {
466             consumerThread = lvBlocked();
467         } while (consumerThread == null);
468 
469         releaseParkedConsumer(consumerThread);
470     }

472     /**
473      * A consumer is trapped in `parking` until the field is nulled, and this code can only be executed after it
474      * is no longer in that state. This in effect means there can be no racing here, whoever wins the CAS to make
475      * pIndex even again owns the field.
476      */
477     private void releaseParkedConsumer(Thread consumerThread) {
478         soBlocked(null);
479         LockSupport.unpark(consumerThread);
480     }

487     public E take() throws InterruptedException
488     {
489         final E[] buffer = consumerBuffer;
490         final long mask = consumerMask;
491 
492         final long cIndex = lpConsumerIndex();
493         final long offset = modifiedCalcCircularRefElementOffset(cIndex, mask);
494         E e = lvRefElement(buffer, offset);
495         if (e == null)
496         {
497             return parkUntilNext(buffer, cIndex, offset, Long.MAX_VALUE);
498         }
499 
500         soRefElement(buffer, offset, null); // release element null
501         soConsumerIndex(cIndex + 2); // release cIndex
502 
503         return e;
504     }

517         final long offset = modifiedCalcCircularRefElementOffset(cIndex, mask);

530         soConsumerIndex(cIndex + 2); // release cIndex
531 
532         return e;
533     }

535     private E parkUntilNext(E[] buffer, long cIndex, long offset, long timeoutNs) throws InterruptedException {
536         E e;
537         final long pIndex = lvProducerIndex();
538         if (cIndex == pIndex && // queue is empty
539             casProducerIndex(pIndex, pIndex + 1)) // we announce ourselves as parked
540         {
541             // producers only try a wakeup when both the index and the blocked thread are visible, otherwise they spin
542             soBlocked(Thread.currentThread());
543             // ignore deadline when it's forever
544             final long deadlineNs = timeoutNs == Long.MAX_VALUE ? 0 : System.nanoTime() + timeoutNs;
545             while (true)
546             {
547                 LockSupport.parkNanos(this, timeoutNs);
548                 if (Thread.interrupted())
549                 {
550                     // revert blocking state
551                     revertParkedState(pIndex);
552                     throw new InterruptedException();
553                 }
554                 if (lvBlocked() == null)
555                 {
556                     break;
557                 }
558                 // ignore deadline when it's forever
559                 timeoutNs = timeoutNs == Long.MAX_VALUE ? Long.MAX_VALUE : deadlineNs - System.nanoTime();
560                 if (timeoutNs <= 0)
561                 {
562                     if (revertParkedState(pIndex)) {
563                         // ran out of time and the producer has not moved the index
564                         return null;
565                     }
566                     else {
567                         break; // just in the nick of time
568                     }
569                 }
570             }
571         }
572         // producer index is visible before element, so if we wake up between the index moving and the element
573         // store we could see a null.
574         e = spinWaitForElement(buffer, offset);
575 
576         soRefElement(buffer, offset, null); // release element null
577         soConsumerIndex(cIndex + 2); // release cIndex
578 
579         return e;
580     }
581 
582     /**
583      * Consumer must revert the `parked` state when interrupted or thewhen returning after a timeout. This means
584      * reverting the pIndex state to it's pre-parking value AND nulling out the blocked field.
585      *
586      * If a producer has beat us to moving the pIndex that may mean an unparking is in flight, in which case we wait for
587      * the unparking producer to null out the blocker field.
588      */
589     private boolean revertParkedState(long pIndex) {
590         if (casProducerIndex(pIndex + 1, pIndex))
591         {
592             soBlocked(null);
593             return true;
594         }
595         spinWaitForUnblock();
596         return false;
597     }

624     {
625         final E[] buffer = consumerBuffer;
626         final long mask = consumerMask;
627 
628         final long index = lpConsumerIndex();
629         final long offset = modifiedCalcCircularRefElementOffset(index, mask);
630         E e = lvRefElement(buffer, offset);
631         if (e == null)
632         {
633             // consumer can't see the odd producer index
634             if (index != lvProducerIndex())
635             {
636                 // poll() == null iff queue is empty, null element is not strong enough indicator, so we must
637                 // check the producer index. If the queue is indeed not empty we spin until element is
638                 // visible.
639                 e = spinWaitForElement(buffer, offset);
640             }
641             else
642             {
643                 return null;
644             }
645         }
646 
647         soRefElement(buffer, offset, null); // release element null
648         soConsumerIndex(index + 2); // release cIndex
649         return e;
650     }

746     @Override
747     public int fill(Supplier<E> s, int limit)
748     {
749         if (null == s)
750             throw new IllegalArgumentException("supplier is null");
751         if (limit < 0)
752             throw new IllegalArgumentException("limit is negative:" + limit);
753         if (limit == 0)
754             return 0;
755 
756         final long mask = this.producerMask;
757 
758         long pIndex;
759         int claimedSlots;
760         boolean wakeup = false;
761         long batchIndex = 0;
762         final long shiftedBatchSize = 2L * limit;
763 
764         while (true)
765         {
766             pIndex = lvProducerIndex();
767             long producerLimit = lvProducerLimit();
768 
769             // lower bit is indicative of blocked consumer
770             if ((pIndex & 1) == 1)
771             {
772                 if(!casProducerIndex(pIndex, pIndex + 1))
773                 {
774                     continue;
775                 }
776                 // We've claimed pIndex, now we need to wake up consumer and set the element
777                 wakeup = true;
778                 batchIndex = pIndex + 1;
779                 pIndex = pIndex - 1;
780                 break;
781             }
782             // pIndex is even (lower bit is 0) -> actual index is (pIndex >> 1), consumer is awake
783 
784             // we want 'limit' slots, but will settle for whatever is visible to 'producerLimit'
785             batchIndex = Math.min(producerLimit, pIndex + shiftedBatchSize); //  -> producerLimit >= batchIndex
786 
787             // Use producer limit to save a read of the more rapidly mutated consumer index.
788             // Assumption: queue is usually empty or near empty
789             if (pIndex >= producerLimit)
790             {
791                 if (!recalculateProducerLimit(mask, pIndex, producerLimit))
792                 {
793                     return 0;
794                 }
795                 batchIndex = Math.min(lvProducerLimit(), pIndex + shiftedBatchSize);
796             }
797 
798             // Claim the index
799             if (casProducerIndex(pIndex, batchIndex))
800             {
801                 break;
802             }
803         }
804         claimedSlots = (int) ((batchIndex - pIndex) / 2);
805 
806         final E[] buffer = this.producerBuffer;
807         // first element offset might be a wakeup, so peeled from loop
808         for (int i = 0; i < claimedSlots; i++)
809         {
810             long offset = modifiedCalcCircularRefElementOffset(pIndex + 2L * i, mask);
811             soRefElement(buffer, offset, s.get());
812         }
813 
814         if (wakeup)
815         {
816             wakeupConsumer();
817         }
818 
819         return claimedSlots;
820     }
