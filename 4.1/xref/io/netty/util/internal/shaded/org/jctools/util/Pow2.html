14  package io.netty.util.internal.shaded.org.jctools.util;
15  
16  /**
17   * Power of 2 utility functions.
18   */
19  @InternalAPI
20  public final class Pow2 {
21      public static final int MAX_POW2 = 1 << 30;
22  
23      /**
24       * @param value from which next positive power of two will be found.
25       * @return the next positive power of 2, this value if it is a power of 2. Negative values are mapped to 1.
26       * @throws IllegalArgumentException is value is more than MAX_POW2 or less than 0
27       */
28      public static int roundToPowerOfTwo(final int value) {
29          if (value > MAX_POW2) {
30              throw new IllegalArgumentException("There is no larger power of 2 int for value:"+value+" since it exceeds 2^31.");
31          }
32          if (value < 0) {
33              throw new IllegalArgumentException("Given value:"+value+". Expecting value >= 0.");
34          }
35          final int nextPow2 = 1 << (32 - Integer.numberOfLeadingZeros(value - 1));
36          return nextPow2;
37      }
38  
39      /**
40       * @param value to be tested to see if it is a power of two.
41       * @return true if the value is a power of 2 otherwise false.
42       */
43      public static boolean isPowerOfTwo(final int value) {
44          return (value & (value - 1)) == 0;
45      }
46  
47      /**
48       * Align a value to the next multiple up of alignment. If the value equals an alignment multiple then it
49       * is returned unchanged.
50       *
51       * @param value to be aligned up.
52       * @param alignment to be used, must be a power of 2.
53       * @return the value aligned to the next boundary.
54       */
55      public static long align(final long value, final int alignment) {
56          if (!isPowerOfTwo(alignment)) {
57              throw new IllegalArgumentException("alignment must be a power of 2:" + alignment);
58          }
59          return (value + (alignment - 1)) & ~(alignment - 1);
60      }
61  }
